\HW{Two-Step Gauge Transformation}{2026-04-22}



% \summary{Starting from the Lax pair, we expand the formal solution, obtain recursive relations for the coefficients, and use them to derive the evolution equations. This yields the mKdV equation as a member of the AKNS hierarchy, with all steps carried out explicitly.} 


Consider the system
\begin{align}
\mathrm{i}\Psi_t
&=
\frac{1}{2}Q^2\Psi
-
(\Psi^\dagger\Psi)\Psi
+
\frac{\Delta}{2}\sigma_3\Psi,
\qquad
\Psi=(\Psi_1,\Psi_2)^\top,
\label{eq:initial-system}
\\
Q
&=
-\mathrm{i}\partial_x+\alpha A(x),
\qquad
A(x)
=
\begin{pmatrix}
0 & {\mathrm e}^{-2\mathrm{i}kx}
\\
{\mathrm e}^{2\mathrm{i}kx} & 0
\end{pmatrix}.
\label{eq:A-matrix}
\end{align}
Here
\begin{equation}
\sigma_1=
\begin{pmatrix}
0 & 1\\
1 & 0
\end{pmatrix},
\qquad
\sigma_2=
\begin{pmatrix}
0 & -\mathrm{i}\\
\mathrm{i} & 0
\end{pmatrix},
\qquad
\sigma_3=
\begin{pmatrix}
1 & 0\\
0 & -1
\end{pmatrix}
\label{eq:pauli-matrices}
\end{equation}
denote the Pauli matrices.

We first compute the square of $Q$. Since
\begin{equation}
A(x)^2=\mathbb{I}_2,
\qquad
A_x(x)=
\begin{pmatrix}
0 & -2\mathrm{i}k\,{\mathrm e}^{-2\mathrm{i}kx}
\\
2\mathrm{i}k\,{\mathrm e}^{2\mathrm{i}kx} & 0
\end{pmatrix},
\label{eq:A-square}
\end{equation}
we obtain
\begin{align}
Q^2
&=
\bigl(-\mathrm{i}\partial_x+\alpha A\bigr)^2
\nonumber\\
&=
-\partial_x^2
-\mathrm{i}\alpha A\partial_x
-\mathrm{i}\alpha\partial_x(A\,\cdot)
+\alpha^2A^2
\nonumber\\
&=
-\partial_x^2
-\mathrm{i}\alpha A\partial_x
-\mathrm{i}\alpha(A_x+A\partial_x)
+\alpha^2A^2
\nonumber\\
&=
-\partial_x^2
-2\mathrm{i}\alpha A\partial_x
-\mathrm{i}\alpha A_x
+\alpha^2.
\label{eq:Q-square}
\end{align}
Substituting \eqref{eq:Q-square} into \eqref{eq:initial-system} yields
\begin{equation}
\mathrm{i}\Psi_t
=
-\frac{1}{2}\Psi_{xx}
-\mathrm{i}\alpha A\Psi_x
-\frac{\mathrm{i}}{2}\alpha A_x\Psi
+\frac{1}{2}\alpha^2\Psi
-|\Psi|^2\Psi
+\frac{\Delta}{2}\sigma_3\Psi.
\label{eq:expanded-system}
\end{equation}

Define the gauge transformation
\begin{equation}
\Psi(x,t)
=
\Lambda(x,t)\psi(x,t),
\qquad
\Lambda(x,t)
=
{\mathrm e}^{-\frac{\mathrm{i}}{2}(\alpha^2+k^2)t}
{\mathrm e}^{-\mathrm{i}kx\sigma_3}.
\label{eq:first-gauge}
\end{equation}
Since $\sigma_3^2=\mathbb{I}_2$, the matrix exponential may be written as
\begin{equation}
{\mathrm e}^{-\mathrm{i}kx\sigma_3}
=
\begin{pmatrix}
{\mathrm e}^{-\mathrm{i}kx} & 0
\\
0 & {\mathrm e}^{\mathrm{i}kx}
\end{pmatrix},
\end{equation}
and therefore $\Lambda$ is unitary:
\begin{equation}
\Lambda^\dagger\Lambda=\mathbb{I}_2.
\label{eq:Lambda-unitary}
\end{equation}
Consequently,
\begin{equation}
|\Psi|^2
=
\Psi^\dagger\Psi
=
\psi^\dagger\Lambda^\dagger\Lambda\psi
=
\psi^\dagger\psi
=
|\psi|^2.
\label{eq:norm-invariant}
\end{equation}

Differentiating \eqref{eq:first-gauge} gives
\begin{equation}
\Lambda_t
=
-\frac{\mathrm{i}}{2}(\alpha^2+k^2)\Lambda,
\qquad
\Lambda_x
=
-\mathrm{i}k\sigma_3\Lambda.
\label{eq:Lambda-derivatives}
\end{equation}
Hence
\begin{align}
\Psi_t
&=
\Lambda_t\psi+\Lambda\psi_t
=
\Lambda\left(
-\frac{\mathrm{i}}{2}(\alpha^2+k^2)\psi+\psi_t
\right),
\label{eq:Psi-t}
\\
\Psi_x
&=
\Lambda_x\psi+\Lambda\psi_x
=
\Lambda\left(
-\mathrm{i}k\sigma_3\psi+\psi_x
\right),
\label{eq:Psi-x}
\end{align}
and
\begin{align}
\Psi_{xx}
&=
\partial_x\!\left[
\Lambda\left(
-\mathrm{i}k\sigma_3\psi+\psi_x
\right)
\right]
\nonumber\\
&=
\Lambda_x\left(
-\mathrm{i}k\sigma_3\psi+\psi_x
\right)
+
\Lambda\left(
-\mathrm{i}k\sigma_3\psi_x+\psi_{xx}
\right)
\nonumber\\
&=
\Lambda\left[
-\mathrm{i}k\sigma_3
\left(
-\mathrm{i}k\sigma_3\psi+\psi_x
\right)
-\mathrm{i}k\sigma_3\psi_x
+\psi_{xx}
\right]
\nonumber\\
&=
\Lambda\left(
-k^2\psi
-2\mathrm{i}k\sigma_3\psi_x
+\psi_{xx}
\right).
\label{eq:Psi-xx}
\end{align}

Next we compute the conjugation of $A(x)$ by $\Lambda$. Since the scalar phase
\(
{\mathrm e}^{-\frac{\mathrm{i}}{2}(\alpha^2+k^2)t}
\)
commutes with every matrix, we have
\begin{align}
\Lambda^{-1}A(x)\Lambda
&=
{\mathrm e}^{\mathrm{i}kx\sigma_3}
A(x)
{\mathrm e}^{-\mathrm{i}kx\sigma_3}
\nonumber\\
&=
\begin{pmatrix}
{\mathrm e}^{\mathrm{i}kx} & 0
\\
0 & {\mathrm e}^{-\mathrm{i}kx}
\end{pmatrix}
\begin{pmatrix}
0 & {\mathrm e}^{-2\mathrm{i}kx}
\\
{\mathrm e}^{2\mathrm{i}kx} & 0
\end{pmatrix}
\begin{pmatrix}
{\mathrm e}^{-\mathrm{i}kx} & 0
\\
0 & {\mathrm e}^{\mathrm{i}kx}
\end{pmatrix}
\nonumber\\
&=
\begin{pmatrix}
0 & 1
\\
1 & 0
\end{pmatrix}
=
\sigma_1.
\label{eq:A-conjugation}
\end{align}
Similarly,
\begin{align}
\Lambda^{-1}A_x(x)\Lambda
=
\begin{pmatrix}
0 & -2\mathrm{i}k
\\
2\mathrm{i}k & 0
\end{pmatrix}
=
2k
\begin{pmatrix}
0 & -\mathrm{i}
\\
\mathrm{i} & 0
\end{pmatrix}
=
2k\sigma_2
=
2\mathrm{i}k\sigma_1\sigma_3.
\label{eq:Ax-conjugation}
\end{align}
Moreover,
\begin{equation}
\Lambda^{-1}\sigma_3\Lambda
=
\sigma_3,
\label{eq:sigma3-conjugation}
\end{equation}
since $\sigma_3$ commutes with ${\mathrm e}^{-\mathrm{i}kx\sigma_3}$.

Substituting \eqref{eq:Psi-t}, \eqref{eq:Psi-x}, and \eqref{eq:Psi-xx} into \eqref{eq:expanded-system}, we obtain
\begin{align}
&\mathrm{i}\Lambda
\left(
-\frac{\mathrm{i}}{2}(\alpha^2+k^2)\psi+\psi_t
\right)
\nonumber\\
&=
-\frac12
\Lambda
\left(
-k^2\psi
-2\mathrm{i}k\sigma_3\psi_x
+\psi_{xx}
\right)
\nonumber\\
&\quad
-\mathrm{i}\alpha A\Lambda
\left(
-\mathrm{i}k\sigma_3\psi+\psi_x
\right)
-\frac{\mathrm{i}}{2}\alpha A_x\Lambda\psi
+\frac12\alpha^2\Lambda\psi
\nonumber\\
&\quad
-|\psi|^2\Lambda\psi
+\frac{\Delta}{2}\sigma_3\Lambda\psi.
\label{eq:substituted-system}
\end{align}
Multiplying \eqref{eq:substituted-system} from the left by $\Lambda^{-1}$ and using
\eqref{eq:A-conjugation}--\eqref{eq:sigma3-conjugation} gives
\begin{align}
\frac12(\alpha^2+k^2)\psi+\mathrm{i}\psi_t
&=
\frac12k^2\psi
+\mathrm{i}k\sigma_3\psi_x
-\frac12\psi_{xx}
\nonumber\\
&\quad
-\mathrm{i}\alpha\sigma_1
\left(
-\mathrm{i}k\sigma_3\psi+\psi_x
\right)
-\mathrm{i}\alpha k\sigma_2\psi
\nonumber\\
&\quad
+\frac12\alpha^2\psi
-|\psi|^2\psi
+\frac{\Delta}{2}\sigma_3\psi.
\label{eq:before-cancellation}
\end{align}
Using the identity
\begin{equation}
\sigma_1\sigma_3=-\mathrm{i}\sigma_2,
\label{eq:pauli-identity}
\end{equation}
we find
\begin{equation}
-\mathrm{i}\alpha\sigma_1(-\mathrm{i}k\sigma_3\psi)
=
\mathrm{i}\alpha k\sigma_2\psi,
\end{equation}
which exactly cancels the term
\(
-\mathrm{i}\alpha k\sigma_2\psi
\)
in \eqref{eq:before-cancellation}. After cancellation of the constant terms, we arrive at
\begin{equation}
\mathrm{i}\psi_t
=
-\frac12\psi_{xx}
+\mathrm{i}(k\sigma_3-\alpha\sigma_1)\psi_x
-|\psi|^2\psi
+\frac{\Delta}{2}\sigma_3\psi.
\label{eq:first-transformed-system}
\end{equation}


We now introduce the second gauge transformation
\begin{equation}
\psi(x,t)
=
V(x,t)q(x,t),
\qquad
V(x,t)
=
{\mathrm e}^{\frac{\mathrm{i}}{2}(\alpha^2+k^2)t}\,
B\,
{\mathrm e}^{-\mathrm{i}\kappa x\sigma_3},
\qquad
\kappa=\sqrt{\alpha^2+k^2},
\label{eq:second-gauge}
\end{equation}
where
\begin{equation}
B=
\begin{pmatrix}
-\sin\varphi & -\cos\varphi
\\
\cos\varphi & -\sin\varphi
\end{pmatrix}
=
-\sin\varphi\,\mathbb{I}_2
+
\cos\varphi\,\sigma_1\sigma_3.
\label{eq:B-matrix}
\end{equation}

The angle $\varphi$ is chosen so that
\begin{equation}
2\varphi
=
\arctan\frac{k}{\alpha}-\frac{\pi}{2}.
\label{eq:phi-definition}
\end{equation}
Hence
\begin{equation}
\tan(2\varphi)
=
-\frac{\alpha}{k},
\end{equation}
and therefore
\begin{equation}
\sin(2\varphi)
=
-\frac{\alpha}{\sqrt{\alpha^2+k^2}},
\qquad
\cos(2\varphi)
=
\frac{k}{\sqrt{\alpha^2+k^2}}.
\label{eq:trig-identities}
\end{equation}

Since $B$ is independent of $x$ and $t$, differentiating
\eqref{eq:second-gauge} yields
\begin{align}
\psi_t
&=
V
\left(
\frac{\mathrm{i}}{2}\kappa^2q+q_t
\right),
\label{eq:psi-t-q}
\\
\psi_x
&=
V
\left(
-\mathrm{i}\kappa\sigma_3q+q_x
\right),
\label{eq:psi-x-q}
\\
\psi_{xx}
&=
V
\left(
-\kappa^2q
-2\mathrm{i}\kappa\sigma_3q_x
+q_{xx}
\right).
\label{eq:q-derivatives}
\end{align}

Next we compute several algebraic identities associated with $B$. Since
\begin{equation}
B^\top
=
\begin{pmatrix}
-\sin\varphi & \cos\varphi
\\
-\cos\varphi & -\sin\varphi
\end{pmatrix},
\end{equation}
a direct computation gives
\begin{equation}
B^\top B
=
\mathbb{I}_2.
\label{eq:B-orthogonal}
\end{equation}
Consequently,
\begin{equation}
V^\dagger V
=
\mathbb{I}_2,
\end{equation}
and hence the nonlinear term is preserved:
\begin{equation}
|\psi|^2
=
\psi^\dagger\psi
=
q^\dagger V^\dagger V q
=
q^\dagger q
=
|q|^2.
\label{eq:q-norm}
\end{equation}

We now compute the conjugation of $\sigma_3$ by $B$. Using
\eqref{eq:trig-identities}, one finds
\begin{align}
B^\top\sigma_3B
=
\begin{pmatrix}
-\cos(2\varphi) & \sin(2\varphi)
\\
\sin(2\varphi) & \cos(2\varphi)
\end{pmatrix}
=
-\frac{1}{\sqrt{\alpha^2+k^2}}
\begin{pmatrix}
k & \alpha
\\
\alpha & -k
\end{pmatrix}.
\label{eq:B-conjugation}
\end{align}

Using
\eqref{eq:B-matrix}, we compute
\begin{align}
B^\top(k\sigma_3-\alpha\sigma_1)B
&=
kB^\top\sigma_3B-\alpha B^\top\sigma_1B
\nonumber\\
&=
\begin{pmatrix}
-k\cos(2\varphi)+\alpha\sin(2\varphi)
&
k\sin(2\varphi)+\alpha\cos(2\varphi)
\\
k\sin(2\varphi)+\alpha\cos(2\varphi)
&
k\cos(2\varphi)-\alpha\sin(2\varphi)
\end{pmatrix}.
\end{align}
Substituting \eqref{eq:trig-identities} gives
\begin{equation}
B^\top(k\sigma_3-\alpha\sigma_1)B
=
\begin{pmatrix}
-\kappa & 0
\\
0 & \kappa
\end{pmatrix}
=
-\kappa\sigma_3.
\label{eq:B-diagonalization}
\end{equation}

Substituting \eqref{eq:psi-t-q}--\eqref{eq:q-derivatives} into
\eqref{eq:first-transformed-system}, we obtain
\begin{align}
\mathrm{i}V
\left(
\frac{\mathrm{i}}{2}\kappa^2q+q_t
\right)
&=
-\frac12
V
\left(
-\kappa^2q
-2\mathrm{i}\kappa\sigma_3q_x
+q_{xx}
\right)
\nonumber\\
&\quad
+
\mathrm{i}(k\sigma_3-\alpha\sigma_1)
V
\left(
-\mathrm{i}\kappa\sigma_3q+q_x
\right)
-|q|^2Vq
+\frac{\Delta}{2}\sigma_3Vq.
\label{eq:q-substitution}
\end{align}
Multiplying \eqref{eq:q-substitution} from the left by $V^{-1}$ yields
\begin{align}
-\frac12\kappa^2q+\mathrm{i}q_t
&=
\frac12\kappa^2q
+\mathrm{i}\kappa\sigma_3q_x
-\frac12q_{xx}
\nonumber\\
&\quad
+
\mathrm{i}V^{-1}(k\sigma_3-\alpha\sigma_1)V
\left(
-\mathrm{i}\kappa\sigma_3q+q_x
\right)
\nonumber\\
&\quad
-|q|^2q
+\frac{\Delta}{2}V^{-1}\sigma_3Vq.
\label{eq:q-before-cancel}
\end{align}

Since
\begin{equation}
V^{-1}(k\sigma_3-\alpha\sigma_1)V
=
{\mathrm e}^{\mathrm{i}\kappa x\sigma_3}
B^\top(k\sigma_3-\alpha\sigma_1)B
{\mathrm e}^{-\mathrm{i}\kappa x\sigma_3},
\end{equation}
equation \eqref{eq:B-diagonalization} implies
\begin{equation}
V^{-1}(k\sigma_3-\alpha\sigma_1)V
=
-\kappa\sigma_3.
\label{eq:diagonalized-operator}
\end{equation}
Substituting \eqref{eq:diagonalized-operator} into
\eqref{eq:q-before-cancel}, the first-order derivative terms cancel:
\begin{align}
\mathrm{i}q_t
&=
-\frac12q_{xx}
-|q|^2q
+\frac{\Delta}{2}V^{-1}\sigma_3Vq.
\label{eq:q-equation-pre-final}
\end{align}

Finally,
\begin{align}
V^{-1}\sigma_3V
&=
{\mathrm e}^{\mathrm{i}\kappa x\sigma_3}
B^\top\sigma_3B
{\mathrm e}^{-\mathrm{i}\kappa x\sigma_3}
\nonumber\\
&=
{\mathrm e}^{\mathrm{i}\kappa x\sigma_3}
\frac{-1}{\sqrt{\alpha^2+k^2}}
\begin{pmatrix}
k & \alpha
\\
\alpha & -k
\end{pmatrix}
{\mathrm e}^{-\mathrm{i}\kappa x\sigma_3}.
\label{eq:sigma3-final-transform}
\end{align}
Therefore,
\begin{align}
\mathrm{i}q_t
&=
-\frac{1}{2}q_{xx}
-|q|^2q
+\frac{\Delta}{2}
{\mathrm e}^{\mathrm{i}\kappa x\sigma_3}
\frac{-1}{\sqrt{\alpha^2+k^2}}
\begin{pmatrix}
k & \alpha
\\
\alpha & -k
\end{pmatrix}
{\mathrm e}^{-\mathrm{i}\kappa x\sigma_3}q.
\label{eq:final-system}
\end{align}


\HW{The squared eigenfunctions on Weierstrass points}{2026-04-29}

Let \(B\in\mathbb{C}^{g\times g}\) be a symmetric matrix with positive definite imaginary part. The associated Riemann theta function is defined by
\begin{equation}
\Theta(\bm{z},B)
=
\sum_{\bm{n}\in\mathbb{Z}^{g}}
\exp\!\left(
\pi\mathrm{i}\,\bm{n}^{\top}B\bm{n}
+
2\pi\mathrm{i}\,\bm{n}^{\top}\bm{z}
\right),
\qquad
\bm{z}\in\mathbb{C}^{g}.
\label{eq:Riemann-theta}
\end{equation}

For \(k=1,\ldots,g\), let
\[
\bm{e}_k
=
(0,\ldots,0,1,0,\ldots,0)^{\top}
\]
denote the \(k\)-th standard basis vector of \(\mathbb{R}^{g}\). The theta function satisfies the quasi-periodicity relations
\begin{equation}
\Theta(\bm{z}+\bm{e}_k,B)
=
\Theta(\bm{z},B),
\label{eq:theta-periodicity-1}
\end{equation}
and
\begin{equation}
\Theta(\bm{z}+B\bm{e}_k,B)
=
\exp\!\left(
-2\pi\mathrm{i}\,z_k
-
\pi\mathrm{i}\,B_{kk}
\right)
\Theta(\bm{z},B),
\label{eq:theta-periodicity-2}
\end{equation}
where \(z_k\) denotes the \(k\)-th component of \(\bm{z}\).

Consider the genus-two hyperelliptic curve
\begin{equation}
\mathscr{F}_2
=
\left\{
P=(\lambda,y)
\;\middle|\;
y^2
=
\prod_{j=1}^{3}
(\lambda-\lambda_j)(\lambda-\lambda_j^*)
\right\},
\label{eq:genus-two-curve}
\end{equation}
with \(\lambda\in\mathbb{CP}^1\).

In Case~1, we assume that
\[
\lambda_1,\lambda_2,\lambda_3\in \mathrm{i}\mathbb{R},
\qquad
\Im \lambda_3>\Im \lambda_2>\Im \lambda_1>0 .
\]

Let \(\{a_1,a_2,b_1,b_2\}\) be a canonical homology basis satisfying
\[
a_i\circ b_j
=
-
b_j\circ a_i
=
\delta_{ij},
\qquad
a_i\circ a_j
=
0,
\qquad
b_i\circ b_j
=
0 .
\]
Let \(\{w_1\,d\lambda,w_2\,d\lambda\}\) be the normalized basis of holomorphic differentials determined by
\begin{equation}
\oint_{a_j} w_i\,d\lambda
=
\delta_{ij},
\qquad
\oint_{b_j} w_i\,d\lambda
=
B_{ij},
\label{eq:normalized-differentials}
\end{equation}
for \(i,j=1,2\).

For a fixed base point \(P_0\in\mathscr{F}_2\), the Abel map is defined by
\begin{equation}
\mathcal{A}_{P_0}(P)
=
\bigl(
\mathcal{A}_{P_0,1}(P),
\mathcal{A}_{P_0,2}(P)
\bigr)^{\top}
=
\Biggl(
\int_{P_0}^{P} w_1\,d\lambda,
\;
\int_{P_0}^{P} w_2\,d\lambda
\Biggr)^{\top}.
\label{eq:Abel-map}
\end{equation}

Let \(d\Omega_1,d\Omega_2,d\Omega_3\) be normalized Abelian differentials satisfying
\begin{equation}
\oint_{a_j} d\Omega_k
=
0,
\qquad
j=1,2,
\quad
k=1,2,3.
\label{eq:a-normalization}
\end{equation}
Their corresponding Abelian integrals admit the asymptotic expansions
\begin{equation}
\Omega_1(P)
=
\pm\left(
\ln \lambda
+
\ln \omega_1
+
o(1)
\right),
\qquad
P\to\infty^\pm,
\label{eq:Omega1-asymptotic}
\end{equation}
\begin{equation}
\Omega_2(P)
=
\pm\left(
\lambda
+
\omega_2
+
o(1)
\right),
\qquad
P\to\infty^\pm,
\label{eq:Omega2-asymptotic}
\end{equation}
and
\begin{equation}
\Omega_3(P)
=
\pm\left(
2\lambda^2
+
\omega_3
+
o(1)
\right),
\qquad
P\to\infty^\pm.
\label{eq:Omega3-asymptotic}
\end{equation}

Fix the base point \(P_0=(\lambda_3,0)\), and define
\begin{equation}
\Omega_k(P)
=
\int_{P_0}^{P} d\Omega_k,
\qquad
k=1,2,3.
\label{eq:Omega-definition}
\end{equation}

The corresponding \(b\)-period vectors are defined by
\begin{equation}
U_k
=
\frac{1}{2\pi\mathrm{i}}
\oint_{b_k} d\Omega_2,
\qquad
V_k
=
\frac{1}{2\pi\mathrm{i}}
\oint_{b_k} d\Omega_3,
\qquad
\Delta_k
=
-
\frac{1}{2\pi\mathrm{i}}
\oint_{b_k} d\Omega_1,
\label{eq:period-vectors}
\end{equation}
for \(k=1,2\). If \(\ell\) is a positively oriented circle surrounding \(\infty^+\), then
\begin{equation}
\oint_{\ell} d\Omega_1
=
2\pi\mathrm{i}.
\label{eq:Omega1-cycle}
\end{equation}

The vectors \(U=(U_1,U_2)\) and \(V=(V_1,V_2)\) satisfy
\begin{equation}
U_1
=
U_2
\in
\mathrm{i}\mathbb{R},
\label{eq:U-reality}
\end{equation}
and
\begin{equation}
V_1
=
-
V_2
\in
\mathbb{R}.
\label{eq:V-reality}
\end{equation}
Moreover,
\begin{equation}
\Delta_k\in\mathbb{R},
\qquad
\Delta_1+\Delta_2=1,
\label{eq:reality-conditions}
\end{equation}
and the symmetry relation
\begin{equation}
\sigma_1\Delta
=
-
\Delta
+
\bm{1},
\qquad
\bm{1}
=
\begin{pmatrix}
1\\
1
\end{pmatrix},
\qquad
\sigma_1
=
\begin{pmatrix}
0 & 1\\
1 & 0
\end{pmatrix},
\label{eq:Delta-symmetry}
\end{equation}
holds.

Define
\begin{equation}
\Omega
=
D+\mathrm{i}Ux+\mathrm{i}Vt
=
\mathrm{i}U(x-x_0)+\mathrm{i}V(t-t_0),
\label{eq:Omega-vector}
\end{equation}
where \(D=-\mathrm{i}Ux_0-\mathrm{i}Vt_0\), and let
\begin{equation}
\Delta
=
\mathcal{A}_{P_0}(\infty^+)
-
\mathcal{A}_{P_0}(\infty^-)
=
\mathcal{A}_{\infty^-}(\infty^+).
\label{eq:Delta-divisor}
\end{equation}
Then
\begin{equation}
\sigma_1\Omega^*
=
\Omega,
\qquad
\mathcal{A}_{\infty^-}(P_0)
=
\frac{\Delta}{2},
\qquad
\mathcal{A}_{\infty^+}(P_0)
=
-
\frac{\Delta}{2}.
\label{eq:Abel-symmetry}
\end{equation}

The Baker--Akhiezer function associated with the two-phase solution is given by
\begin{equation}
\eta(P;x,t)
=
\omega_2x+\omega_3t+\frac{\theta}{2}
+\frac{\mathrm{i}}{2}\Omega_1(P),
\label{eq:BA-phase-eta}
\end{equation}
and
\begin{equation}
\begin{aligned}
\binom{\Phi_{11}(P;x,t)}{\Phi_{21}(P;x,t)}
&=
\frac{\exp\!\bigl(\mathrm{i}\Omega_2(P)x+\mathrm{i}\Omega_3(P)t\bigr)}
{\Theta(\Omega)}
\exp\!\bigl(-\mathrm{i}\eta(P;x,t)\sigma_3\bigr)
\\
&\quad\times
\binom{
\Theta\bigl(\Omega+\mathcal{A}_{\infty^+}(P)\bigr)
}{
\Theta\bigl(\Omega+\Delta+\mathcal{A}_{\infty^+}(P)\bigr)
}.
\end{aligned}
\label{eq:BA-function}
\end{equation}
Here \(\omega_2=0\) and \(\omega_3\in\mathbb{R}\).

Using \eqref{eq:Abel-symmetry}, we obtain
\begin{equation}
\Omega+\mathcal{A}_{\infty^+}(P)
=
\Omega
-
\frac{\Delta}{2}
+
\mathcal{A}_{P_0}(P),
\label{eq:shift-plus}
\end{equation}
and
\begin{equation}
\Omega+\Delta+\mathcal{A}_{\infty^+}(P)
=
\Omega
+
\frac{\Delta}{2}
+
\mathcal{A}_{P_0}(P).
\label{eq:shift-minus}
\end{equation}


Define the branch points
\begin{equation}
P_j
=
(\lambda_j,0),
\qquad
P_{-j}
=
(\lambda_j^*,0),
\qquad
j=1,2,3.
\label{eq:branch-points}
\end{equation}

For \(P\in\mathscr{F}_2\), define
\begin{equation}
\bm{P}_1(P)
=
\begin{pmatrix}
\Phi_{11}(P)\\[4pt]
\Phi_{21}(P)
\end{pmatrix}
\begin{pmatrix}
\Phi_{11}^*\bigl(\tau_R(\iota(P))\bigr),
&
\Phi_{21}^*\bigl(\tau_R(\iota(P))\bigr)
\end{pmatrix},
\label{eq:P1-definition}
\end{equation}
and
\begin{equation}
\bm{S}_1(P)
=
\begin{pmatrix}
\Phi_{21}(P)\,
\Phi_{11}^*\bigl(\tau_R(\iota(P))\bigr)
\\[6pt]
\Phi_{11}(P)\,
\Phi_{21}^*\bigl(\tau_R(\iota(P))\bigr)
\end{pmatrix}.
\label{eq:S1-definition}
\end{equation}

Here \(\tau_R\) and \(\iota\) denote the involutions
\begin{equation}
\tau_R:(\lambda,y)\mapsto(\lambda^*,y^*),
\qquad
\iota:(\lambda,y)\mapsto(\lambda,-y).
\label{eq:involutions}
\end{equation}

Let
\begin{equation}
\mathscr{P}:\mathscr{F}_2\to\mathbb{CP}^1,
\qquad
\mathscr{P}(\lambda,y)=\lambda,
\label{eq:projection-map}
\end{equation}
be the canonical projection map. We consider \(\bm{S}_1(P)\) at the Weierstrass points satisfying
\[
\mathscr{P}(P)
=
\pm\lambda_1,
\pm\lambda_2,
\pm\lambda_3.
\]

Since \(P\) is a branch point, one has \(\iota(P)=P\). Hence
\begin{equation}
\bm{S}_1(P)
=
\begin{pmatrix}
\Phi_{21}(P)\,
\Phi_{11}^*\bigl(\tau_R(P)\bigr)
\\[6pt]
\Phi_{11}(P)\,
\Phi_{21}^*\bigl(\tau_R(P)\bigr)
\end{pmatrix}.
\label{eq:S1-branch-points}
\end{equation}

We now establish the following result concerning the squared eigenfunctions evaluated at the Weierstrass points.

\begin{prop}\label{prop:hw2}
The vectors
\[
\bm{S}_1(P_j),
\qquad
j=\pm1,\pm2,\pm3,
\]
span a three-dimensional subspace. More precisely,
\begin{equation}
\dim
\Bigl(
\operatorname{span}
\{
\bm{S}_1(P_j)
:\,
j=\pm1,\pm2,\pm3
\}
\Bigr)
=
3.
\label{eq:span-dimension-prop}
\end{equation}
\end{prop}


Define
\begin{equation}
\begin{pmatrix}
\widetilde{\Phi}_{11}(P)\\[4pt]
\widetilde{\Phi}_{21}(P)
\end{pmatrix}
=
\exp\!\left[
-
\mathrm{i}
\left(
\omega_2x+\omega_3t+\frac{\theta}{2}
+\frac{\mathrm{i}}{2}\Omega_1(P)
\right)
\sigma_3
\right]
\begin{pmatrix}
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P)
\right)
\\[8pt]
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P)
\right)
\end{pmatrix}.
\label{eq:tilde-Phi}
\end{equation}

Taking complex conjugates at the reflected point \(\tau_R(P)\), we obtain
\begin{align}
&
\begin{pmatrix}
\widetilde{\Phi}_{11}^*\bigl(\tau_R(P)\bigr)
&
\widetilde{\Phi}_{21}^*\bigl(\tau_R(P)\bigr)
\end{pmatrix}
\nonumber\\[4pt]
&=
\begin{pmatrix}
\Theta^*\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}\bigl(\tau_R(P)\bigr)
\right),
&
\Theta^*\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}\bigl(\tau_R(P)\bigr)
\right)
\end{pmatrix}
\nonumber\\[4pt]
&\hspace{2cm}\times
\exp\!\left[
\mathrm{i}
\left(
\omega_2x+\omega_3t+\frac{\theta}{2}
-
\frac{\mathrm{i}}{2}\Omega_1^*\bigl(\tau_R(P)\bigr)
\right)
\sigma_3
\right].
\label{eq:tilde-Phi-conjugate}
\end{align}

Substituting these expressions into the definition of \(\bm{S}_1(P)\) yields
\begin{equation}
\bm{S}_1(P)
=
\widetilde{\bm{S}}_1(P)
\frac{
\exp\!\left(
\mathrm{i}
\bigl(
\Omega_2(P)-\Omega_2^*(\tau_R(P))
\bigr)x
+
\mathrm{i}
\bigl(
\Omega_3(P)-\Omega_3^*(\tau_R(P))
\bigr)t
\right)
}{
|\Theta(\Omega)|^2
}.
\label{eq:S1-factorization}
\end{equation}
Here
\begin{equation}
\widetilde{\bm{S}}_1(P)
=
\begin{pmatrix}
\widetilde{\Phi}_{21}(P)\,
\widetilde{\Phi}_{11}^*\bigl(\tau_R(P)\bigr)
\\[8pt]
\widetilde{\Phi}_{11}(P)\,
\widetilde{\Phi}_{21}^*\bigl(\tau_R(P)\bigr)
\end{pmatrix}.
\label{eq:tilde-S1}
\end{equation}

Using \eqref{eq:tilde-Phi} and \eqref{eq:tilde-Phi-conjugate}, we further obtain
\begin{align}
\widetilde{\bm{S}}_1(P)
&=
\exp\!\left[
\mathrm{i}
\left(
2\omega_2x
+
2\omega_3t
+
\theta
+
\frac{\mathrm{i}}{2}
\bigl(
\Omega_1(P)-\Omega_1^*\bigl(\tau_R(P)\bigr)
\bigr)
\right)
\sigma_3
\right]
\nonumber\\[4pt]
&\quad\times
\begin{pmatrix}
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P)
\right)
\Theta^*\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}\bigl(\tau_R(P)\bigr)
\right)
\\[10pt]
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P)
\right)
\Theta^*\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}\bigl(\tau_R(P)\bigr)
\right)
\end{pmatrix}.
\label{eq:tilde-S1-expanded}
\end{align}

Since
\begin{equation}
\mathcal{A}_{P_0}(P)
=
\begin{pmatrix}
\displaystyle \int_{P_0}^{P} w_1\,d\lambda \\[10pt]
\displaystyle \int_{P_0}^{P} w_2\,d\lambda
\end{pmatrix},
\qquad
P_0=P_3=(\lambda_3,0),
\label{eq:Abel-map-components}
\end{equation}
we choose all integration paths in the sheet \(\mathscr{F}^-\). Then
\begin{equation}
\mathcal{A}_{P_0}(P_3)=0.
\label{eq:A-P3}
\end{equation}

For the branch point \(P_2\), the integration path corresponds to one half of the cycle \(b_1\). Hence
\begin{equation}
\mathcal{A}_{P_0}(P_2)
=
\frac{1}{2}
\oint_{b_1}
\begin{pmatrix}
w_1\,d\lambda\\
w_2\,d\lambda
\end{pmatrix}
=
\frac{B\bm{e}_1}{2},
\label{eq:A-P2}
\end{equation}
where
\[
\bm{e}_1
=
\begin{pmatrix}
1\\
0
\end{pmatrix},
\qquad
\bm{e}_2
=
\begin{pmatrix}
0\\
1
\end{pmatrix}.
\]

Similarly, the path from \(P_3\) to \(P_1\) is homologous to one half of the cycle \(-a_1+b_1\), and therefore
\begin{equation}
\mathcal{A}_{P_0}(P_1)
=
\frac{1}{2}
\oint_{-a_1+b_1}
\begin{pmatrix}
w_1\,d\lambda\\
w_2\,d\lambda
\end{pmatrix}
=
\frac{B\bm{e}_1-\bm{e}_1}{2}.
\label{eq:A-P1}
\end{equation}

For the conjugate branch point \(P_{-3}\), the integration path is homologous to one half of the cycle \(-a_1-a_2\). Thus
\begin{equation}
\mathcal{A}_{P_0}(P_{-3})
=
\frac{1}{2}
\oint_{-a_1-a_2}
\begin{pmatrix}
w_1\,d\lambda\\
w_2\,d\lambda
\end{pmatrix}
=
-
\frac{\bm{e}_1+\bm{e}_2}{2}
=
-
\frac{\bm{1}}{2},
\label{eq:A-Pminus3}
\end{equation}
where
\[
\bm{1}
=
\begin{pmatrix}
1\\
1
\end{pmatrix}.
\]

Proceeding in the same way, we obtain
\begin{equation}
\mathcal{A}_{P_0}(P_{-2})
=
\frac{1}{2}
\oint_{b_2-a_1-a_2}
\begin{pmatrix}
w_1\,d\lambda\\
w_2\,d\lambda
\end{pmatrix}
=
\frac{B\bm{e}_2-\bm{1}}{2},
\label{eq:A-Pminus2}
\end{equation}
and
\begin{equation}
\mathcal{A}_{P_0}(P_{-1})
=
\frac{1}{2}
\oint_{b_2-a_1}
\begin{pmatrix}
w_1\,d\lambda\\
w_2\,d\lambda
\end{pmatrix}
=
\frac{B\bm{e}_2-\bm{e}_1}{2}.
\label{eq:A-Pminus1}
\end{equation}

Since \(P_0=P_3\), we immediately obtain
\begin{equation}
\Omega_1(P_3)
=
\int_{P_0}^{P_3} d\Omega_1
=
0.
\label{eq:Omega1-P3}
\end{equation}

Let \(\ell\) be a positively oriented small circle surrounding \(\infty^+\). Since
\[
\oint_{a_j} d\Omega_1=0,
\]
the integration path from \(P_0=P_3\) to \(P_{-3}\) is homologous to one half of the cycle \(\ell\). Hence
\begin{equation}
\Omega_1(P_{-3})
=
\frac{1}{2}\oint_{\ell} d\Omega_1
=
\frac{1}{2}\cdot 2\pi\mathrm{i}
=
\pi\mathrm{i}.
\label{eq:Omega1-Pminus3}
\end{equation}

Similarly, the integration paths corresponding to \(P_1\) and \(P_2\) are homologous to one half of the cycle \(\ell+b_1\). Hence
\begin{align}
\Omega_1(P_1)
=
\Omega_1(P_2)
=
\frac{1}{2}
\oint_{\ell+b_1} d\Omega_1
=
\pi\mathrm{i}
+
\frac{1}{2}\oint_{b_1} d\Omega_1
=
\pi\mathrm{i}
-
\pi\mathrm{i}\Delta_1
=
\pi\mathrm{i}(1-\Delta_1).
\label{eq:Omega1-P1}
\end{align}

In the same way,
\begin{align}
\Omega_1(P_{-1})
=
\Omega_1(P_{-2})
=
\frac{1}{2}
\oint_{\ell+b_2} d\Omega_1
=
\pi\mathrm{i}
+
\frac{1}{2}\oint_{b_2} d\Omega_1
=
\pi\mathrm{i}
-
\pi\mathrm{i}\Delta_2
=
\pi\mathrm{i}(1-\Delta_2).
\label{eq:Omega1-Pminus1}
\end{align}

Since
\[
\Omega_1(P_j)\in\mathrm{i}\mathbb{R},
\qquad
\Delta_1+\Delta_2=1,
\]
we obtain
\begin{equation}
\Omega_1(P_j)
-
\Omega_1^*\bigl(\tau_R(P_j)\bigr)
=
\Omega_1(P_j)
+
\Omega_1(P_{-j})
=
\pi\mathrm{i},
\qquad
j=\pm1,\pm2,\pm3.
\label{eq:Omega1-reality}
\end{equation}
Consequently,
\begin{equation}
\frac{\mathrm{i}}{2}
\Bigl(
\Omega_1(P_j)
-
\Omega_1^*\bigl(\tau_R(P_j)\bigr)
\Bigr)
=
\frac{\mathrm{i}}{2}\pi\mathrm{i}
=
-
\frac{\pi}{2}.
\label{eq:Omega1-phase}
\end{equation}

By analogous arguments, one also obtains
\begin{equation}
\Omega_2(P_j)
-
\Omega_2^*\bigl(\tau_R(P_j)\bigr)
=
0,
\qquad
\Omega_3(P_j)
-
\Omega_3^*\bigl(\tau_R(P_j)\bigr)
=
0,
\qquad
j=\pm1,\pm2,\pm3.
\label{eq:Omega23-reality}
\end{equation}

Substituting these identities into \eqref{eq:S1-factorization}, we conclude that
\begin{equation}
\bm{S}_1(P_j)
=
\frac{
\widetilde{\bm{S}}_1(P_j)
}{
|\Theta(\Omega)|^2
},
\qquad
j=\pm1,\pm2,\pm3.
\label{eq:S1-factorization-2}
\end{equation}

Define
\begin{equation}
\widetilde{\theta}_1
=
2\omega_2x+2\omega_3t+\theta-\frac{\pi}{2}.
\label{eq:g1-definition}
\end{equation}
Then \eqref{eq:tilde-S1-expanded} becomes
\begin{equation}
\widetilde{\bm{S}}_1(P_j)
=
\exp\!\bigl(
\mathrm{i}\widetilde{\theta}_1\sigma_3
\bigr)
\begin{pmatrix}
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta^*\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_{-j})
\right)
\\[12pt]
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta^*\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_{-j})
\right)
\end{pmatrix}.
\label{eq:S1-theta-form}
\end{equation}

Using the identities
\[
\Theta^*(\bm{z})
=
\Theta(\bm{z}^*),
\qquad
\Theta(\bm{z})
=
\Theta(\sigma_1\bm{z}),
\qquad
\sigma_1\Omega^*
=
\Omega,
\]
we may rewrite \eqref{eq:S1-theta-form} as
\begin{equation}
\widetilde{\bm{S}}_1(P_j)
=
\exp\!\bigl(
\mathrm{i}\widetilde{\theta}_1\sigma_3
\bigr)
\begin{pmatrix}
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta\!\left(
\Omega-\dfrac{\sigma_1\Delta}{2}
+\sigma_1\mathcal{A}_{P_0}^*(P_{-j})
\right)
\\[12pt]
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta\!\left(
\Omega+\dfrac{\sigma_1\Delta}{2}
+\sigma_1\mathcal{A}_{P_0}^*(P_{-j})
\right)
\end{pmatrix}.
\label{eq:S1-theta-symmetry}
\end{equation}

Since
\[
\sigma_1\Delta
=
-\Delta+\bm{1},
\]
we further obtain
\begin{equation}
\widetilde{\bm{S}}_1(P_j)
=
\exp\!\bigl(
\mathrm{i}\widetilde{\theta}_1\sigma_3
\bigr)
\begin{pmatrix}
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
-\dfrac{\bm{1}}{2}
+\sigma_1\mathcal{A}_{P_0}^*(P_{-j})
\right)
\\[12pt]
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\dfrac{\bm{1}}{2}
+\sigma_1\mathcal{A}_{P_0}^*(P_{-j})
\right)
\end{pmatrix}.
\label{eq:S1-final}
\end{equation}

Finally, using the periodicity relation
\[
\Theta(\bm{z}+\bm{e}_k)
=
\Theta(\bm{z}),
\qquad
k=1,2,
\]
we conclude from \eqref{eq:S1-final} that
\begin{equation}
\widetilde{\bm{S}}_1(P_j)
=
\exp\!\bigl(
\mathrm{i}\widetilde{\theta}_1\sigma_3
\bigr)
\begin{pmatrix}
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta\!\left(
\Omega+\dfrac{\Delta}{2}
+\dfrac{\bm{1}}{2}
+\sigma_1\mathcal{A}_{P_0}^*(P_{-j})
\right)
\\[12pt]
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\mathcal{A}_{P_0}(P_j)
\right)
\Theta\!\left(
\Omega-\dfrac{\Delta}{2}
+\dfrac{\bm{1}}{2}
+\sigma_1\mathcal{A}_{P_0}^*(P_{-j})
\right)
\end{pmatrix}.
\label{eq:S1-shifted}
\end{equation}

Note that
\begin{equation}
\mathcal{A}_{P_0}(P_j)
+
\sigma_1\mathcal{A}_{P_0}^*(P_{-j})
+
\frac{\bm{1}}{2}
=
\bm{0},
\qquad
j=\pm1,\pm2,\pm3.
\label{eq:Abel-identity}
\end{equation}

For example, using \eqref{eq:A-P2} and \eqref{eq:A-Pminus2}, we obtain
\begin{align}
\mathcal{A}_{P_0}(P_2)
+
\sigma_1\mathcal{A}_{P_0}^*(P_{-2})
+
\frac{\bm{1}}{2}
=
\frac{B\bm{e}_1}{2}
+
\sigma_1
\frac{-B\bm{e}_2-\bm{1}}{2}
+
\frac{\bm{1}}{2}
=
\frac{
B\bm{e}_1
-
\sigma_1B\bm{e}_2
}{
2
}
=
\bm{0}.
\label{eq:Abel-identity-P2}
\end{align}
Similarly,
\begin{align}
\mathcal{A}_{P_0}(P_{-2})
+
\sigma_1\mathcal{A}_{P_0}^*(P_2)
+
\frac{\bm{1}}{2}
=
\frac{B\bm{e}_2-\bm{1}}{2}
+
\sigma_1
\frac{-B\bm{e}_1}{2}
+
\frac{\bm{1}}{2}
=
\frac{
B\bm{e}_2
-
\sigma_1B\bm{e}_1
}{
2
}
=
\bm{0}.
\label{eq:Abel-identity-Pminus2}
\end{align}
Here we have used the symmetry relations
\begin{equation}
B\bm{e}_1
=

\HW{Numerical Reproduction of One-Mode Anomalous-Wave Recurrence}{2026-05-20}

In this homework we reproduce the first numerical figure in the paper of
P. G. Grinevich and P. M. Santini, \emph{The finite-gap method and the
periodic NLS Cauchy problem of anomalous waves for a finite number of
unstable modes}, Russian Mathematical Surveys \textbf{74} (2019), 211--263.
The main purpose is not only to draw a similar picture, but also to
explain how the numerical simulation is built from the equation.  I write
the explanation in a beginner-oriented way: first the continuous model,
then the discrete grid, then the time stepping formula, and finally the
MATLAB implementation.

The equation considered in the paper is the focusing nonlinear
Schr\"odinger equation
\begin{equation}
\mathrm{i}u_t+u_{xx}+2|u|^2u=0,
\qquad
u=u(x,t)\in\mathbb{C},
\label{eq:hw3-focusing-nls}
\end{equation}
with periodic boundary conditions in \(x\).  Here \(x\) is the spatial
variable, \(t\) is time, and the unknown \(u(x,t)\) is complex-valued.
The quantity that we plot is the amplitude \(|u(x,t)|\).  The exact
constant-amplitude background is
\begin{equation}
u_0(x,t)=\mathrm{e}^{2\mathrm{i}t},
\label{eq:hw3-background}
\end{equation}
so \(|u_0(x,t)|=1\) for all \(x\) and \(t\).  The numerical experiment
starts from a very small periodic perturbation of the unit background at
\(t=0\), and the interesting question is whether this tiny perturbation
stays tiny or grows into a large visible wave.

\noteheading{Modulation instability.}
Modulation instability is the growth of a small long-wave perturbation of
a plane wave background.  For the focusing NLS equation
\eqref{eq:hw3-focusing-nls}, write the solution near the background in
the form
\begin{equation}
u(x,t)
=
\mathrm{e}^{2\mathrm{i}t}
\left(1+w(x,t)\right),
\qquad
|w|\ll1.
\label{eq:hw3-linearization-ansatz}
\end{equation}
After linearization, each Fourier pair with wave number \(k\) evolves as
a combination of a growing and a decaying mode when
\begin{equation}
0<k<2.
\label{eq:hw3-unstable-band}
\end{equation}
More precisely, the growth rate is
\begin{equation}
\sigma(k)
=
k\sqrt{4-k^2},
\qquad
0<k<2.
\label{eq:hw3-growth-rate}
\end{equation}
If \(k>2\), the corresponding linearized frequency is real and the mode
does not grow exponentially.  This is the first important idea behind the
picture: not every Fourier mode is dangerous.  Only the modes whose wave
numbers lie in the interval \(0<k<2\) are linearly unstable.  Thus the
focusing sign in \eqref{eq:hw3-focusing-nls} is essential: it is the
reason why a perturbation of size \(10^{-4}\) can later become an
\(O(1)\) coherent wave.

On a periodic interval of length \(L\), the admissible Fourier wave
numbers are
\begin{equation}
k_j=\frac{2\pi j}{L},
\qquad
j=1,2,\ldots .
\label{eq:hw3-fourier-modes}
\end{equation}
Therefore \(k_j<2\) is equivalent to \(j<L/\pi\), and the number of
unstable positive Fourier modes is
\begin{equation}
N
=
\left\lfloor \frac{L}{\pi}\right\rfloor .
\label{eq:hw3-number-unstable-modes}
\end{equation}
This is the basic mechanism behind the phrase ``a finite number of
unstable modes'' in the paper.

\noteheading{One unstable mode.}
The initial condition used for Figure~1 of the paper is
\begin{equation}
u(x,0)
=
1
+
\epsilon
\left(
c_1\mathrm{e}^{\mathrm{i}k_1x}
+
c_{-1}\mathrm{e}^{-\mathrm{i}k_1x}
\right),
\qquad
k_1=\frac{2\pi}{L},
\label{eq:hw3-initial-data}
\end{equation}
where
\begin{equation}
L=6,
\qquad
\epsilon=10^{-4},
\qquad
c_1=\frac{1}{2},
\qquad
c_{-1}=\frac{0.3-0.4\mathrm{i}}{2}.
\label{eq:hw3-paper-parameters}
\end{equation}
For these parameters,
\begin{equation}
N
=
\left\lfloor \frac{L}{\pi}\right\rfloor
=1,
\qquad
k_1
=
\frac{\pi}{3}
<2,
\label{eq:hw3-one-mode-condition}
\end{equation}
so only the first Fourier mode belongs to the unstable band
\eqref{eq:hw3-unstable-band}.  The second mode would have wave number
\(2k_1=2\pi/3>2\), so it is outside the unstable band.  Thus this is a
clean one-mode test case.  The perturbation first grows from the almost
constant background, then forms a localized Akhmediev-type peak, and
finally returns close to the background before the next recurrence.

\noteheading{Split-step Fourier discretization.}
The split-step Fourier method is a standard method for equations such as
\eqref{eq:hw3-focusing-nls}, because the equation contains two parts that
are easy in two different representations.  First rewrite the equation as
\begin{equation}
u_t
=
\mathrm{i}u_{xx}
+2\mathrm{i}|u|^2u.
\label{eq:hw3-evolution-form}
\end{equation}
The term \(\mathrm{i}u_{xx}\) is the linear dispersive part.  It is easy
after applying the Fourier transform.  The term
\(2\mathrm{i}|u|^2u\) is the nonlinear part.  It is easy at each fixed
grid point in physical space.  The split-step idea is to solve these two
parts one after the other during each small time interval.

The spatial interval in the code is
\[
[-L/2,L/2]=[-3,3],
\]
and the code uses \(N_x=384\) grid points.  The grid spacing is therefore
\[
\Delta x=\frac{L}{N_x}.
\]
The grid is periodic, so the left endpoint and the right endpoint
represent the same point.  This is why the MATLAB code creates \(N_x+1\)
points by \texttt{linspace} and then removes the last one.  The Fourier
wave numbers used by MATLAB's FFT are ordered as
\begin{equation}
k
=
\frac{2\pi}{L}
\left(
0,1,\ldots,\frac{N_x}{2}-1,
-\frac{N_x}{2},\ldots,-1
\right).
\label{eq:hw3-fft-wavenumbers}
\end{equation}
This ordering is not arbitrary: it is the ordering expected by the
vectors produced by \texttt{fft}.

For the linear subproblem
\[
u_t=\mathrm{i}u_{xx},
\]
write
\[
u(x,t)=\sum_k \widehat{u}_k(t)\mathrm{e}^{\mathrm{i}kx}.
\]
Since differentiating twice gives
\[
u_{xx}
=
\sum_k (-k^2)\widehat{u}_k(t)\mathrm{e}^{\mathrm{i}kx},
\]
each Fourier coefficient satisfies
\[
\frac{\mathrm{d}}{\mathrm{d}t}\widehat{u}_k
=
-\mathrm{i}k^2\widehat{u}_k.
\]
Therefore the exact linear evolution over a time interval \(\tau\) is
\begin{equation}
\widehat{u}_k(t+\tau)
=
\mathrm{e}^{-\mathrm{i}k^2\tau}\widehat{u}_k(t).
\label{eq:hw3-linear-fourier-flow}
\end{equation}
In the MATLAB code, one half linear step has \(\tau=\Delta t/2\), so the
multiplier is
\[
\texttt{Ehalf = exp(-1i*k.\string^2*dt/2)}.
\]

For the nonlinear subproblem
\[
u_t=2\mathrm{i}|u|^2u,
\]
there is no spatial derivative.  Thus each grid value evolves
independently.  During this nonlinear substep the modulus \(|u|\) is
constant, and only the complex phase rotates.  The exact nonlinear
evolution over one full time step \(\Delta t\) is
\begin{equation}
u(t+\Delta t)
=
u(t)\exp\!\left(2\mathrm{i}|u(t)|^2\Delta t\right).
\label{eq:hw3-nonlinear-flow}
\end{equation}
This is exactly the MATLAB command
\[
\texttt{u = u .* exp(2i*abs(u).\string^2*dt)}.
\]

The code uses \(T_{\max}=60\), \(N_t=30000\), and hence
\begin{equation}
\Delta t=\frac{T_{\max}}{N_t}
=
\frac{60}{30000}
=
0.002.
\label{eq:hw3-time-step}
\end{equation}
The number \(0.002\) is small enough for a smooth reproduction while
keeping the runtime short.  One full Strang step is
\begin{equation}
u^{n+1}
=
\mathcal{L}_{\Delta t/2}
\mathcal{N}_{\Delta t}
\mathcal{L}_{\Delta t/2}u^n,
\label{eq:hw3-strang-compact}
\end{equation}
where
\[
\mathcal{L}_{\Delta t/2}u
=
\mathcal{F}^{-1}
\left[
\mathrm{e}^{-\mathrm{i}k^2\Delta t/2}\mathcal{F}u
\right],
\qquad
\mathcal{N}_{\Delta t}u
=
u\exp(2\mathrm{i}|u|^2\Delta t).
\]
Written out in the same order as the MATLAB loop, this is
\begin{equation}
u^{(1)}
=
\mathcal{F}^{-1}
\left[
\mathrm{e}^{-\mathrm{i}k^2\Delta t/2}
\mathcal{F}u^n
\right],
\label{eq:hw3-first-half-step}
\end{equation}
\begin{equation}
u^{(2)}
=
u^{(1)}
\exp\!\left(2\mathrm{i}|u^{(1)}|^2\Delta t\right),
\label{eq:hw3-nonlinear-step-written}
\end{equation}
\begin{equation}
u^{n+1}
=
\mathcal{F}^{-1}
\left[
\mathrm{e}^{-\mathrm{i}k^2\Delta t/2}
\mathcal{F}u^{(2)}
\right].
\label{eq:hw3-second-half-step}
\end{equation}
Equations \eqref{eq:hw3-first-half-step}--\eqref{eq:hw3-second-half-step}
are the mathematical version of the three main operations inside the
\texttt{for} loop: FFT half-step, nonlinear phase rotation, FFT
half-step.

\notes{Linear Stage in the Grinevich--Santini Recurrence Paper}{2026-05-27}

In this note we organize the part of the calculation in
P. G. Grinevich and P. M. Santini,
\emph{The finite gap method and the analytic description of the exact rogue
wave recurrence in the periodic NLS Cauchy problem}, Nonlinearity
\textbf{31} (2018), 5258--5308, which leads to formulas (15) and (16) of
the paper.  The purpose here is only to understand the first linear stage
of modulation instability.  The nonlinear Akhmediev-breather formula
(18)--(19) will be used later, but we do not derive the full finite-gap in this note.

\noteheading{Setting and unstable Fourier modes.}
The equation is the focusing nonlinear Schr\"odinger equation
\begin{equation}
\mathrm{i}u_t+u_{xx}+2|u|^2u=0,
\qquad
u=u(x,t)\in\mathbb{C},
\label{eq:note1-nls}
\end{equation}
on the periodic interval of length \(L\).  The background solution is
\begin{equation}
u_0(x,t)=\mathrm{e}^{2\mathrm{i}t}.
\label{eq:note1-background}
\end{equation}
The initial condition is a small perturbation of this background:
\begin{equation}
u(x,0)
=
1+\epsilon v(x),
\qquad
0<\epsilon\ll1,
\label{eq:note1-initial-small}
\end{equation}
where
\begin{equation}
v(x)
=
\sum_{j\ge1}
\left(
c_j\mathrm{e}^{\mathrm{i}k_jx}
+
c_{-j}\mathrm{e}^{-\mathrm{i}k_jx}
\right),
\qquad
k_j=\frac{2\pi j}{L}.
\label{eq:note1-fourier-expansion}
\end{equation}
The \(j\)-th Fourier mode is the pair of waves
\(\mathrm{e}^{\mathrm{i}k_jx}\) and
\(\mathrm{e}^{-\mathrm{i}k_jx}\).  These two waves must be treated
together, because the linearized NLS equation contains the complex
conjugate of the perturbation.

For one unstable mode \(n\), the paper assumes
\begin{equation}
v(x)
=
c_n\mathrm{e}^{\mathrm{i}k_nx}
+
c_{-n}\mathrm{e}^{-\mathrm{i}k_nx}.
\label{eq:note1-one-mode-data}
\end{equation}
The mode is unstable precisely when
\begin{equation}
0<k_n<2.
\label{eq:note1-unstable-condition}
\end{equation}
It is useful to introduce the angle
\begin{equation}
\phi_n
=
\arccos\frac{k_n}{2}
=
\arccos\frac{\pi n}{L},
\qquad
0<\phi_n<\frac{\pi}{2}.
\label{eq:note1-phi-definition}
\end{equation}
Then
\begin{equation}
k_n=2\cos\phi_n.
\label{eq:note1-k-phi}
\end{equation}

\noteheading{Linearization about the background.}
Write the solution near the background as
\begin{equation}
u(x,t)
=
\mathrm{e}^{2\mathrm{i}t}\bigl(1+w(x,t)\bigr),
\qquad
|w|\ll1.
\label{eq:note1-linearization-ansatz}
\end{equation}
We compute the terms in \eqref{eq:note1-nls}.  First,
\begin{equation}
u_t
=
2\mathrm{i}\mathrm{e}^{2\mathrm{i}t}(1+w)
+
\mathrm{e}^{2\mathrm{i}t}w_t,
\qquad
u_{xx}
=
\mathrm{e}^{2\mathrm{i}t}w_{xx}.
\label{eq:note1-linear-derivatives}
\end{equation}
Next,
\begin{align}
|u|^2u
&=
\mathrm{e}^{2\mathrm{i}t}
|1+w|^2(1+w)
\nonumber\\
&=
\mathrm{e}^{2\mathrm{i}t}
\left(1+2w+\overline{w}\right)
+
O(|w|^2).
\label{eq:note1-nonlinear-expansion}
\end{align}
Substituting these expressions into \eqref{eq:note1-nls} and keeping only
the terms linear in \(w\), the constant background terms cancel and we
obtain
\begin{equation}
\mathrm{i}w_t+w_{xx}+2(w+\overline{w})=0.
\label{eq:note1-linearized-equation}
\end{equation}

\noteheading{Growth rate.}
Because of the term \(\overline{w}\), a single wave
\(\mathrm{e}^{\mathrm{i}kx}\) is coupled to
\(\mathrm{e}^{-\mathrm{i}kx}\).  Therefore, for a fixed wave number \(k\),
we set
\begin{equation}
w(x,t)
=
a\mathrm{e}^{\mathrm{i}kx+\sigma t}
+
b\mathrm{e}^{-\mathrm{i}kx+\sigma t}.
\label{eq:note1-mode-pair}
\end{equation}
Here \(\sigma\) is the growth or decay exponent.  Substituting
\eqref{eq:note1-mode-pair} into \eqref{eq:note1-linearized-equation} and
comparing the coefficients of
\(\mathrm{e}^{\mathrm{i}kx+\sigma t}\) and
\(\mathrm{e}^{-\mathrm{i}kx+\sigma t}\), we get
\begin{equation}
\begin{pmatrix}
2-k^2+\mathrm{i}\sigma & 2
\\
2 & 2-k^2-\mathrm{i}\sigma
\end{pmatrix}
\begin{pmatrix}
a\\
\overline{b}
\end{pmatrix}
=0.
\label{eq:note1-growth-system}
\end{equation}
For a nonzero solution, the determinant must vanish:
\begin{equation}
(2-k^2+\mathrm{i}\sigma)(2-k^2-\mathrm{i}\sigma)-4=0.
\label{eq:note1-determinant-condition}
\end{equation}
Hence
\begin{equation}
\sigma^2
=
4-(2-k^2)^2
=
k^2(4-k^2).
\label{eq:note1-dispersion-relation}
\end{equation}
Thus the unstable growth rate is
\begin{equation}
\sigma(k)=k\sqrt{4-k^2},
\qquad
0<k<2.
\label{eq:note1-growth-rate}
\end{equation}
For the \(n\)-th unstable mode,
\begin{equation}
\sigma_n
=
k_n\sqrt{4-k_n^2}
=
2\sin(2\phi_n),
\label{eq:note1-sigma-n}
\end{equation}
where we used \(k_n=2\cos\phi_n\).

\noteheading{Growing and decaying coordinates.}
The initial Fourier coefficients \(c_n,c_{-n}\) measure how much of the
positive and negative wave numbers is present.  They are not themselves
the growing and decaying amplitudes.  The linearized system diagonalizes
into one direction proportional to \(\mathrm{e}^{\sigma_nt}\) and one
direction proportional to \(\mathrm{e}^{-\sigma_nt}\).

We now compute these two directions explicitly.  First consider a pure
growing solution of \eqref{eq:note1-linearized-equation}:
\begin{equation}
w_{\mathrm{g}}(x,t)
=
\mathrm{e}^{\sigma_nt}
\left(
a\mathrm{e}^{\mathrm{i}k_nx}
+
b\mathrm{e}^{-\mathrm{i}k_nx}
\right).
\label{eq:note1-pure-growing}
\end{equation}
The coefficient of
\(\mathrm{e}^{\mathrm{i}k_nx+\sigma_nt}\) gives
\begin{equation}
\left(2-k_n^2+\mathrm{i}\sigma_n\right)a
+
2\overline{b}
=0.
\label{eq:note1-growing-eigenvector-equation}
\end{equation}
Therefore
\begin{equation}
\overline{b}
=
-\frac{2-k_n^2+\mathrm{i}\sigma_n}{2}a.
\label{eq:note1-bbar-ratio-pre}
\end{equation}
Using
\begin{equation}
k_n=2\cos\phi_n,
\qquad
\sigma_n=2\sin(2\phi_n),
\label{eq:note1-k-sigma-trig}
\end{equation}
we have
\begin{equation}
2-k_n^2
=
-2\cos(2\phi_n).
\label{eq:note1-two-minus-k-square}
\end{equation}
Thus
\begin{equation}
-\frac{2-k_n^2+\mathrm{i}\sigma_n}{2}
=
\cos(2\phi_n)-\mathrm{i}\sin(2\phi_n)
=
\mathrm{e}^{-2\mathrm{i}\phi_n}.
\label{eq:note1-growing-ratio}
\end{equation}
Hence
\begin{equation}
\overline{b}=\mathrm{e}^{-2\mathrm{i}\phi_n}a,
\qquad
b=\mathrm{e}^{2\mathrm{i}\phi_n}\overline{a}.
\label{eq:note1-growing-relation}
\end{equation}
This means that the two Fourier coefficients in a pure growing solution
cannot be chosen independently.

If we call the coefficient of
\(\mathrm{e}^{-\mathrm{i}k_nx}\) in the growing part \(A\), then the
corresponding coefficient of \(\mathrm{e}^{\mathrm{i}k_nx}\) is
\(\mathrm{e}^{2\mathrm{i}\phi_n}\overline{A}\).  Therefore the growing
part may be written as
\begin{equation}
w_{\mathrm{g}}(x,t)
=
\mathrm{e}^{\sigma_nt}
\left(
A\mathrm{e}^{-\mathrm{i}k_nx}
+
\mathrm{e}^{2\mathrm{i}\phi_n}\overline{A}\,
\mathrm{e}^{\mathrm{i}k_nx}
\right).
\label{eq:note1-growing-A-form}
\end{equation}

Similarly, a pure decaying solution has the form
\begin{equation}
w_{\mathrm{d}}(x,t)
=
\mathrm{e}^{-\sigma_nt}
\left(
c\mathrm{e}^{\mathrm{i}k_nx}
+
d\mathrm{e}^{-\mathrm{i}k_nx}
\right).
\label{eq:note1-pure-decaying}
\end{equation}
Now the coefficient of
\(\mathrm{e}^{\mathrm{i}k_nx-\sigma_nt}\) gives
\begin{equation}
\left(2-k_n^2-\mathrm{i}\sigma_n\right)c
+
2\overline{d}
=0.
\label{eq:note1-decaying-eigenvector-equation}
\end{equation}
Using the same trigonometric identities,
\begin{equation}
\overline{d}
=
-\frac{2-k_n^2-\mathrm{i}\sigma_n}{2}c
=
\mathrm{e}^{2\mathrm{i}\phi_n}c,
\qquad
d=\mathrm{e}^{-2\mathrm{i}\phi_n}\overline{c}.
\label{eq:note1-decaying-relation}
\end{equation}
If we call the coefficient of \(\mathrm{e}^{\mathrm{i}k_nx}\) in the
decaying part \(D\), then
\begin{equation}
w_{\mathrm{d}}(x,t)
=
\mathrm{e}^{-\sigma_nt}
\left(
\mathrm{e}^{-2\mathrm{i}\phi_n}\overline{D}\,
\mathrm{e}^{-\mathrm{i}k_nx}
+
D\mathrm{e}^{\mathrm{i}k_nx}
\right).
\label{eq:note1-decaying-D-form}
\end{equation}

The general linear solution in this unstable mode is therefore
\begin{align}
w(x,t)
={}&
\mathrm{e}^{\sigma_nt}
\left(
A\mathrm{e}^{-\mathrm{i}k_nx}
+
\mathrm{e}^{2\mathrm{i}\phi_n}\overline{A}\,
\mathrm{e}^{\mathrm{i}k_nx}
\right)
\nonumber\\
&+
\mathrm{e}^{-\sigma_nt}
\left(
\mathrm{e}^{-2\mathrm{i}\phi_n}\overline{D}\,
\mathrm{e}^{-\mathrm{i}k_nx}
+
D\mathrm{e}^{\mathrm{i}k_nx}
\right).
\label{eq:note1-AD-linear-solution}
\end{align}
At \(t=0\), this must match the initial perturbation
\begin{equation}
w(x,0)
=
\epsilon
\left(
c_n\mathrm{e}^{\mathrm{i}k_nx}
+
c_{-n}\mathrm{e}^{-\mathrm{i}k_nx}
\right).
\label{eq:note1-initial-w}
\end{equation}
Comparing the coefficients of
\(\mathrm{e}^{-\mathrm{i}k_nx}\) and
\(\mathrm{e}^{\mathrm{i}k_nx}\) gives
\begin{align}
A+\mathrm{e}^{-2\mathrm{i}\phi_n}\overline{D}
&=
\epsilon c_{-n},
\label{eq:note1-AD-system-minus}
\\
\mathrm{e}^{2\mathrm{i}\phi_n}\overline{A}+D
&=
\epsilon c_n.
\label{eq:note1-AD-system-plus}
\end{align}

Now solve these two equations.  Taking the complex conjugate of
\eqref{eq:note1-AD-system-plus} gives
\begin{equation}
\mathrm{e}^{-2\mathrm{i}\phi_n}A+\overline{D}
=
\epsilon\overline{c_n}.
\label{eq:note1-conjugate-plus}
\end{equation}
Subtracting \(\mathrm{e}^{-2\mathrm{i}\phi_n}\) times
\eqref{eq:note1-conjugate-plus} from \eqref{eq:note1-AD-system-minus}, we
obtain
\begin{equation}
\left(1-\mathrm{e}^{-4\mathrm{i}\phi_n}\right)A
=
\epsilon
\left(
c_{-n}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}\overline{c_n}
\right).
\label{eq:note1-solve-A-pre}
\end{equation}
Since
\begin{equation}
1-\mathrm{e}^{-4\mathrm{i}\phi_n}
=
\mathrm{e}^{-2\mathrm{i}\phi_n}
\left(
\mathrm{e}^{2\mathrm{i}\phi_n}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}
\right)
=
\mathrm{i}\sigma_n\mathrm{e}^{-2\mathrm{i}\phi_n},
\label{eq:note1-denominator-A}
\end{equation}
and
\begin{equation}
c_{-n}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}\overline{c_n}
=
-\mathrm{e}^{-2\mathrm{i}\phi_n}
\left(
\overline{c_n}
-
\mathrm{e}^{2\mathrm{i}\phi_n}c_{-n}
\right),
\label{eq:note1-numerator-A}
\end{equation}
we get
\begin{equation}
A
=
\frac{\mathrm{i}}{\sigma_n}
\epsilon
\left(
\overline{c_n}
-
\mathrm{e}^{2\mathrm{i}\phi_n}c_{-n}
\right).
\label{eq:note1-A-alpha-pre}
\end{equation}
This is why the paper defines
\begin{equation}
\alpha_n
=
\epsilon
\left(
\overline{c_n}
-
\mathrm{e}^{2\mathrm{i}\phi_n}c_{-n}
\right).
\label{eq:note1-alpha-definition}
\end{equation}
Thus
\begin{equation}
A=\frac{\mathrm{i}\alpha_n}{\sigma_n}.
\label{eq:note1-A-alpha}
\end{equation}

The decaying coefficient is found in the same way.  Taking the complex
conjugate of \eqref{eq:note1-AD-system-minus} gives
\begin{equation}
\overline{A}+\mathrm{e}^{2\mathrm{i}\phi_n}D
=
\epsilon\overline{c_{-n}}.
\label{eq:note1-conjugate-minus}
\end{equation}
Subtracting \(\mathrm{e}^{-2\mathrm{i}\phi_n}\) times
\eqref{eq:note1-AD-system-plus} from \eqref{eq:note1-conjugate-minus}, we
obtain
\begin{equation}
\left(
\mathrm{e}^{2\mathrm{i}\phi_n}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}
\right)D
=
\epsilon
\left(
\overline{c_{-n}}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}c_n
\right).
\label{eq:note1-solve-D-pre}
\end{equation}
Since
\[
\mathrm{e}^{2\mathrm{i}\phi_n}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}
=
\mathrm{i}\sigma_n,
\]
we get
\begin{equation}
D
=
\frac{1}{\mathrm{i}\sigma_n}
\epsilon
\left(
\overline{c_{-n}}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}c_n
\right).
\label{eq:note1-D-beta-pre}
\end{equation}
This is why the paper defines
\begin{equation}
\beta_n
=
\epsilon
\left(
\overline{c_{-n}}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}c_n
\right).
\label{eq:note1-beta-definition}
\end{equation}
Thus
\begin{equation}
D=-\frac{\mathrm{i}\beta_n}{\sigma_n}.
\label{eq:note1-D-beta}
\end{equation}

Combining \eqref{eq:note1-alpha-definition} and
\eqref{eq:note1-beta-definition}, the paper's coefficients are
\begin{equation}
\alpha_n
=
\epsilon
\left(
\overline{c_n}
-
\mathrm{e}^{2\mathrm{i}\phi_n}c_{-n}
\right),
\qquad
\beta_n
=
\epsilon
\left(
\overline{c_{-n}}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}c_n
\right).
\label{eq:note1-alpha-beta}
\end{equation}
They are \(O(\epsilon)\).  The coefficient \(\alpha_n\) is the growing
coordinate of the initial perturbation, while \(\beta_n\) is the decaying
coordinate.

Substituting \(A=\mathrm{i}\alpha_n/\sigma_n\) and
\(D=-\mathrm{i}\beta_n/\sigma_n\) into
\eqref{eq:note1-AD-linear-solution}, we obtain
\begin{align}
w(x,t)
={}&
\frac{\mathrm{i}}{\sigma_n}
\left(
\alpha_n\mathrm{e}^{-\mathrm{i}k_nx}
-
\mathrm{e}^{2\mathrm{i}\phi_n}
\overline{\alpha_n}\,
\mathrm{e}^{\mathrm{i}k_nx}
\right)
\mathrm{e}^{\sigma_nt}
\nonumber\\
&-
\frac{\mathrm{i}}{\sigma_n}
\left(
\beta_n\mathrm{e}^{\mathrm{i}k_nx}
-
\mathrm{e}^{-2\mathrm{i}\phi_n}
\overline{\beta_n}\,
\mathrm{e}^{-\mathrm{i}k_nx}
\right)
\mathrm{e}^{-\sigma_nt}.
\label{eq:note1-linear-complex-form}
\end{align}
The first line is the growing \(\alpha_n\)-wave, and the second line is
the decaying \(\beta_n\)-wave.

\noteheading{The growing wave as a cosine.}
Write
\begin{equation}
\alpha_n=|\alpha_n|\mathrm{e}^{\mathrm{i}\theta_\alpha},
\qquad
\theta_\alpha=\arg(\alpha_n).
\label{eq:note1-alpha-phase}
\end{equation}
The growing part of \eqref{eq:note1-linear-complex-form} is
\begin{equation}
w_{\mathrm{grow}}
=
\frac{\mathrm{i}}{\sigma_n}
\left(
\alpha_n\mathrm{e}^{-\mathrm{i}k_nx}
-
\mathrm{e}^{2\mathrm{i}\phi_n}
\overline{\alpha_n}\,
\mathrm{e}^{\mathrm{i}k_nx}
\right)
\mathrm{e}^{\sigma_nt}.
\label{eq:note1-growing-complex}
\end{equation}
We want to write it as a shifted cosine.  Define
\begin{equation}
X_n^+
=
\frac{\arg(\alpha_n)-\phi_n+\pi/2}{k_n}.
\label{eq:note1-X-plus}
\end{equation}
Then
\begin{equation}
k_nX_n^+
=
\theta_\alpha-\phi_n+\frac{\pi}{2},
\label{eq:note1-X-plus-phase}
\end{equation}
and
\begin{align}
\frac{2|\alpha_n|}{\sigma_n}
\mathrm{e}^{\sigma_nt+\mathrm{i}\phi_n}
\cos\bigl(k_n(x-X_n^+)\bigr)
&=
\frac{|\alpha_n|}{\sigma_n}
\mathrm{e}^{\sigma_nt+\mathrm{i}\phi_n}
\left[
\mathrm{e}^{\mathrm{i}k_n(x-X_n^+)}
+
\mathrm{e}^{-\mathrm{i}k_n(x-X_n^+)}
\right]
\nonumber\\
&=
\frac{\mathrm{i}}{\sigma_n}
\left(
\alpha_n\mathrm{e}^{-\mathrm{i}k_nx}
-
\mathrm{e}^{2\mathrm{i}\phi_n}
\overline{\alpha_n}\,
\mathrm{e}^{\mathrm{i}k_nx}
\right)
\mathrm{e}^{\sigma_nt}.
\label{eq:note1-growing-cosine-check}
\end{align}
Therefore
\begin{equation}
w_{\mathrm{grow}}
=
\frac{2|\alpha_n|}{\sigma_n}
\cos\bigl(k_n(x-X_n^+)\bigr)
\mathrm{e}^{\sigma_nt+\mathrm{i}\phi_n}.
\label{eq:note1-growing-cosine}
\end{equation}
The point \(X_n^+\) is the position of the maximum of the growing
sinusoidal wave.

\noteheading{The decaying wave as a cosine.}
Similarly, write
\begin{equation}
\beta_n=|\beta_n|\mathrm{e}^{\mathrm{i}\theta_\beta},
\qquad
\theta_\beta=\arg(\beta_n).
\label{eq:note1-beta-phase}
\end{equation}
Define
\begin{equation}
X_n^-
=
\frac{-\arg(\beta_n)-\phi_n+\pi/2}{k_n}.
\label{eq:note1-X-minus}
\end{equation}
Then the decaying part of \eqref{eq:note1-linear-complex-form} becomes
\begin{equation}
w_{\mathrm{decay}}
=
\frac{2|\beta_n|}{\sigma_n}
\cos\bigl(k_n(x-X_n^-)\bigr)
\mathrm{e}^{-\sigma_nt-\mathrm{i}\phi_n}.
\label{eq:note1-decaying-cosine}
\end{equation}
The point \(X_n^-\) is the position of the maximum of the decaying
sinusoidal wave.

Equations \eqref{eq:note1-X-plus} and \eqref{eq:note1-X-minus} are precisely
formula (16) in the paper:
\begin{equation}
X_n^+
=
\frac{\arg(\alpha_n)-\phi_n+\pi/2}{k_n},
\qquad
X_n^-
=
\frac{-\arg(\beta_n)-\phi_n+\pi/2}{k_n}.
\label{eq:note1-paper-16}
\end{equation}

\noteheading{Formula (15).}
Combining \eqref{eq:note1-growing-cosine} and
\eqref{eq:note1-decaying-cosine}, we obtain
\begin{equation}
w(x,t)
=
\frac{2}{\sigma_n}
\left[
|\alpha_n|
\cos\bigl(k_n(x-X_n^+)\bigr)
\mathrm{e}^{\sigma_nt+\mathrm{i}\phi_n}
+
|\beta_n|
\cos\bigl(k_n(x-X_n^-)\bigr)
\mathrm{e}^{-\sigma_nt-\mathrm{i}\phi_n}
\right].
\label{eq:note1-w-formula-15}
\end{equation}
Since \(u=\mathrm{e}^{2\mathrm{i}t}(1+w)\), this gives
\begin{align}
u(x,t)
=
\mathrm{e}^{2\mathrm{i}t}
\Bigg\{
1
+
\frac{2}{\sigma_n}
\bigg[
&|\alpha_n|
\cos\bigl(k_n(x-X_n^+)\bigr)
\mathrm{e}^{\sigma_nt+\mathrm{i}\phi_n}
\nonumber\\
&+
|\beta_n|
\cos\bigl(k_n(x-X_n^-)\bigr)
\mathrm{e}^{-\sigma_nt-\mathrm{i}\phi_n}
\bigg]
\Bigg\}
+
O\!\left(\epsilon^2|\log\epsilon|\right).
\label{eq:note1-paper-15}
\end{align}
This is formula (15) of the paper.  The interpretation is simple:
the unstable part of the initial datum splits into a growing
\(\alpha_n\)-wave and a decaying \(\beta_n\)-wave.

The remainder in \eqref{eq:note1-paper-15} should not be confused with the
ordinary error one would get by only linearizing the PDE.  The linearized
calculation explains the main \(O(\epsilon)\) terms.  The stated
\(O(\epsilon^2|\log\epsilon|)\) remainder comes from the finite-gap
construction: the small gaps have size \(O(\epsilon)\), and the asymptotic
expansion of the Riemann matrix and theta functions contains logarithmic
error estimates of the form \(O(\epsilon^2|\log\epsilon|)\).  In the early
time region \(0\le t\le O(1)\), the terms
\(|\alpha_n|\mathrm{e}^{\sigma_nt}\) and
\(|\beta_n|\mathrm{e}^{-\sigma_nt}\) are still \(O(\epsilon)\), so this
linear-stage formula is valid.

\noteheading{Bridge to the first nonlinear stage.}
Formula \eqref{eq:note1-paper-15} is not intended to describe the moment
when the growing wave has become \(O(1)\).  That happens on the logarithmic
time scale
\begin{equation}
t=O\left(\frac{|\log\epsilon|}{\sigma_n}\right).
\label{eq:note1-log-time-scale}
\end{equation}
At that scale the growing \(\alpha_n\)-wave triggers the first nonlinear
stage.  We now show how the appearance time is obtained by matching the
growing wave in \eqref{eq:note1-paper-15} with the left tail of the
Akhmediev breather.

For this matching calculation, abbreviate
\[
\sigma=\sigma_n,\qquad
k=k_n,\qquad
\phi=\phi_n,\qquad
X=X_n^+.
\]
The Akhmediev breather used for the first nonlinear stage has the form
\begin{equation}
A(x,t;\phi,X,T,2\phi)
=
\mathrm{e}^{2\mathrm{i}t+2\mathrm{i}\phi}
\frac{
\cosh\!\left(\sigma(t-T)+2\mathrm{i}\phi\right)
+
\sin\phi\cos\!\left(k(x-X)\right)
}{
\cosh\!\left(\sigma(t-T)\right)
-
\sin\phi\cos\!\left(k(x-X)\right)
}.
\label{eq:note1-akhmediev-special}
\end{equation}
Here \(T\) is still unknown.  The left tail means that the current time is
well before the breather center time \(T\), on the time scale
\(1/\sigma\):
\begin{equation}
\sigma(t-T)\ll -1,
\qquad\text{equivalently}\qquad
T-t\gg \frac{1}{\sigma}.
\label{eq:note1-left-tail-condition}
\end{equation}
This condition does not require \(t<0\).  It only means that \(t\) is far
to the left of the peak time \(T\).

Set
\begin{equation}
y=\sigma(t-T),
\qquad
C=\cos\!\left(k(x-X)\right),
\qquad
s=\sin\phi,
\qquad
q=\mathrm{e}^y.
\label{eq:note1-tail-variables}
\end{equation}
In the left tail, \(q\ll1\).  Formula \eqref{eq:note1-akhmediev-special}
becomes
\begin{equation}
A
=
\mathrm{e}^{2\mathrm{i}t+2\mathrm{i}\phi}
\frac{\cosh(y+2\mathrm{i}\phi)+sC}{\cosh y-sC}.
\label{eq:note1-akhmediev-y-form}
\end{equation}

We expand the denominator first.  Since
\begin{equation}
\cosh y
=
\frac{\mathrm{e}^y+\mathrm{e}^{-y}}{2}
=
\frac{q+q^{-1}}{2}
=
\frac{1}{2q}\left(1+q^2\right),
\label{eq:note1-cosh-y-expand}
\end{equation}
we have
\begin{equation}
\cosh y-sC
=
\frac{1}{2q}
\left(
1-2sCq+q^2
\right).
\label{eq:note1-denominator-tail}
\end{equation}
For the numerator,
\begin{align}
\cosh(y+2\mathrm{i}\phi)
&=
\frac{
\mathrm{e}^{y+2\mathrm{i}\phi}
+
\mathrm{e}^{-y-2\mathrm{i}\phi}
}{2}
\nonumber\\
&=
\frac{1}{2q}
\left(
\mathrm{e}^{-2\mathrm{i}\phi}
+
q^2\mathrm{e}^{2\mathrm{i}\phi}
\right).
\label{eq:note1-cosh-shift-expand}
\end{align}
Therefore
\begin{equation}
\cosh(y+2\mathrm{i}\phi)+sC
=
\frac{1}{2q}
\left(
\mathrm{e}^{-2\mathrm{i}\phi}
+
2sCq
+
q^2\mathrm{e}^{2\mathrm{i}\phi}
\right).
\label{eq:note1-numerator-tail}
\end{equation}
Dividing \eqref{eq:note1-numerator-tail} by
\eqref{eq:note1-denominator-tail}, the common factor \(1/(2q)\) cancels:
\begin{equation}
A
=
\mathrm{e}^{2\mathrm{i}t+2\mathrm{i}\phi}
\frac{
\mathrm{e}^{-2\mathrm{i}\phi}
+
2sCq
+
q^2\mathrm{e}^{2\mathrm{i}\phi}
}{
1-2sCq+q^2
}.
\label{eq:note1-tail-ratio-before}
\end{equation}
Keeping only terms up to first order in \(q\), we use
\[
\frac{1}{1-2sCq+O(q^2)}
=
1+2sCq+O(q^2).
\]
Thus
\begin{align}
A
&=
\mathrm{e}^{2\mathrm{i}t}
\left[
1
+
2sC\left(1+\mathrm{e}^{2\mathrm{i}\phi}\right)q
+
O(q^2)
\right].
\label{eq:note1-tail-pre-coefficient}
\end{align}
The coefficient simplifies because
\begin{equation}
1+\mathrm{e}^{2\mathrm{i}\phi}
=
2\cos\phi\,\mathrm{e}^{\mathrm{i}\phi},
\qquad
2s\left(1+\mathrm{e}^{2\mathrm{i}\phi}\right)
=
2\sin(2\phi)\mathrm{e}^{\mathrm{i}\phi}
=
\sigma\mathrm{e}^{\mathrm{i}\phi}.
\label{eq:note1-tail-coefficient-simplify}
\end{equation}
Since \(q=\mathrm{e}^{\sigma(t-T)}\), the left-tail expansion is
\begin{equation}
A(x,t;\phi,X,T,2\phi)
=
\mathrm{e}^{2\mathrm{i}t}
\left[
1
+
\sigma
\mathrm{e}^{\sigma(t-T)+\mathrm{i}\phi}
\cos\!\left(k(x-X)\right)
+
O\!\left(\mathrm{e}^{2\sigma(t-T)}\right)
\right].
\label{eq:note1-akhmediev-left-tail}
\end{equation}
So, far before its center time, the Akhmediev breather looks like a small
exponentially growing linear wave on top of the background.

Now set \(\phi=\phi_n\), \(k=k_n\), \(X=X_n^+\), and
\(\sigma=\sigma_n\).  The growing part of the linear formula
\eqref{eq:note1-paper-15} is
\begin{equation}
u_{\mathrm{lin}}(x,t)
\sim
\mathrm{e}^{2\mathrm{i}t}
\left[
1
+
\frac{2|\alpha_n|}{\sigma_n}
\mathrm{e}^{\sigma_nt+\mathrm{i}\phi_n}
\cos\!\left(k_n(x-X_n^+)\right)
\right].
\label{eq:note1-linear-growing-for-match}
\end{equation}
The left tail \eqref{eq:note1-akhmediev-left-tail} gives
\begin{equation}
A(x,t;\phi_n,X_n^+,T,2\phi_n)
\sim
\mathrm{e}^{2\mathrm{i}t}
\left[
1
+
\sigma_n
\mathrm{e}^{\sigma_n(t-T)+\mathrm{i}\phi_n}
\cos\!\left(k_n(x-X_n^+)\right)
\right].
\label{eq:note1-breather-tail-for-match}
\end{equation}
Rewrite the coefficient in \eqref{eq:note1-breather-tail-for-match} as
\begin{equation}
\sigma_n\mathrm{e}^{\sigma_n(t-T)+\mathrm{i}\phi_n}
=
\sigma_n\mathrm{e}^{-\sigma_nT}
\mathrm{e}^{\sigma_nt+\mathrm{i}\phi_n}.
\label{eq:note1-tail-coefficient-factor}
\end{equation}
Therefore the breather tail has growing-wave coefficient
\(\sigma_n\mathrm{e}^{-\sigma_nT}\), while the linear solution has
coefficient \(2|\alpha_n|/\sigma_n\).  Matching these coefficients gives
\begin{equation}
\frac{2|\alpha_n|}{\sigma_n}
=
\sigma_n\mathrm{e}^{-\sigma_nT}.
\label{eq:note1-matching-equation}
\end{equation}
Solving this equation,
\begin{equation}
\mathrm{e}^{-\sigma_nT}
=
\frac{2|\alpha_n|}{\sigma_n^2},
\qquad
-\sigma_nT
=
\log\left(\frac{2|\alpha_n|}{\sigma_n^2}\right).
\label{eq:note1-solve-matching}
\end{equation}
Hence
\begin{equation}
T_n(|\alpha_n|)
=
\frac{1}{\sigma_n}
\log\left(\frac{\sigma_n^2}{2|\alpha_n|}\right),
\label{eq:note1-first-appearance-time}
\end{equation}
which is formula (17) in the paper.  The spatial center is \(X_n^+\).
Thus the modulus \(|\alpha_n|\) determines the first appearance time,
while \(\arg(\alpha_n)\) determines the position of the first peak.


\notes{Periodic Spectral Theory and the Monodromy Matrix}{2026-06-03}

This note begins the part of Section 3 of Grinevich--Santini concerned
with the periodic spectral problem.  The purpose is to make precise the
objects that enter the finite-gap description of the periodic NLS Cauchy
problem: the fundamental matrix, the monodromy matrix, Bloch multipliers,
the main spectrum, and the auxiliary spectrum.

\noteheading{The fixed-time spatial spectral problem.}
The focusing NLS equation
\begin{equation}
\mathrm{i}u_t+u_{xx}+2|u|^2u=0
\label{eq:note2-nls}
\end{equation}
is represented by the compatibility of the Lax pair
\begin{equation}
\Psi_x(\lambda,x,t)=U(\lambda,x,t)\Psi(\lambda,x,t),
\qquad
\Psi_t(\lambda,x,t)=V(\lambda,x,t)\Psi(\lambda,x,t),
\label{eq:note2-lax-pair}
\end{equation}
where the spatial matrix is
\begin{equation}
U(\lambda,x,t)
=
\begin{pmatrix}
-\mathrm{i}\lambda & \mathrm{i}u(x,t)
\\
\mathrm{i}\overline{u(x,t)} & \mathrm{i}\lambda
\end{pmatrix}.
\label{eq:note2-U}
\end{equation}
For the construction of the periodic spectrum, first fix a time
\(t=t_0\).  Then \(\lambda\) is a spectral parameter and
\eqref{eq:note2-lax-pair} becomes an ordinary differential equation in
the spatial variable \(x\):
\begin{equation}
\Psi_x(\lambda,x,t_0)
=
U(\lambda,x,t_0)\Psi(\lambda,x,t_0).
\label{eq:note2-spatial-ode}
\end{equation}
If the potential is \(L\)-periodic,
\begin{equation}
u(x+L,t_0)=u(x,t_0),
\label{eq:note2-periodic-potential}
\end{equation}
then \(U(\lambda,x+L,t_0)=U(\lambda,x,t_0)\).  Thus, after fixing
\(\lambda\) and \(t_0\), the spatial problem has the form
\begin{equation}
\Psi_x=A(x)\Psi,
\qquad
A(x+L)=A(x),
\label{eq:note2-periodic-ode}
\end{equation}
with \(A(x)=U(\lambda,x,t_0)\).

Equivalently, the spatial equation can be written as the spectral
problem
\begin{equation}
L\Psi=\lambda\Psi,
\qquad
L=
\begin{pmatrix}
\mathrm{i}\partial_x & u
\\
-\overline{u} & -\mathrm{i}\partial_x
\end{pmatrix}.
\label{eq:note2-L-operator}
\end{equation}
This operator is not self-adjoint in the focusing case.  Consequently
its spectrum is not restricted to the real axis, and the periodic theory
is built from Floquet theory and analytic matrix functions rather than
from the self-adjoint spectral theorem.

\noteheading{Fundamental matrix and propagation.}
For any reference point \(y\), define the fundamental matrix by the
initial-value problem
\begin{equation}
\frac{\partial}{\partial x}\Phi(\lambda;x,y,t_0)
=
U(\lambda,x,t_0)\Phi(\lambda;x,y,t_0),
\qquad
\Phi(\lambda;y,y,t_0)=I.
\label{eq:note2-fundamental-matrix}
\end{equation}
When \(\lambda\) and \(t_0\) are fixed, we write simply
\(\Phi(x,y)\).  The matrix \(\Phi(x,y)\) propagates initial data from
\(y\) to \(x\): if \(\Psi(y)=v\), then
\begin{equation}
\Psi(x)=\Phi(x,y)v.
\label{eq:note2-propagator}
\end{equation}
The two columns of \(\Phi(x,y)\) are the two solutions whose initial
data at \(x=y\) are the standard basis vectors.

The fundamental matrix satisfies the composition law
\begin{equation}
\Phi(x,z)=\Phi(x,y)\Phi(y,z).
\label{eq:note2-composition-law}
\end{equation}
Indeed, for fixed \(y\) and \(z\), both
\(\Phi(x,z)\) and \(\Phi(x,y)\Phi(y,z)\) solve the same matrix ODE in
\(x\), and at \(x=y\) both are equal to \(\Phi(y,z)\).  Uniqueness of
the initial-value problem gives \eqref{eq:note2-composition-law}.

Taking \(z=x\) in \eqref{eq:note2-composition-law} gives the inverse
evolution identity
\begin{equation}
\Phi(x,y)^{-1}=\Phi(y,x).
\label{eq:note2-inverse-evolution}
\end{equation}
The periodicity of \(U\) gives another basic identity:
\begin{equation}
\Phi(x+L,y+L)=\Phi(x,y).
\label{eq:note2-periodic-fundamental-matrix}
\end{equation}
To see this, define \(F(x)=\Phi(x+L,y+L)\).  Then
\[
F_x(x)=U(\lambda,x+L,t_0)F(x)=U(\lambda,x,t_0)F(x),
\]
and \(F(y)=I\).  This is the same initial-value problem as the one for
\(\Phi(x,y)\), hence \eqref{eq:note2-periodic-fundamental-matrix}
follows.

\noteheading{The monodromy matrix.}
Fix a base point \(x_0\).  The monodromy matrix is the fundamental
matrix over one spatial period:
\begin{equation}
T(\lambda;x_0,t_0)
=
\Phi(\lambda;x_0+L,x_0,t_0).
\label{eq:note2-monodromy-definition}
\end{equation}
Equivalently, if \(\Psi(x_0)=v\), then
\begin{equation}
\Psi(x_0+L)=T(\lambda;x_0,t_0)v.
\label{eq:note2-monodromy-action}
\end{equation}
Thus \(T\) records the total effect of the periodic spatial equation
after going once around the period.

The identity
\begin{equation}
\Phi(x+L,x_0)=\Phi(x,x_0)T
\label{eq:note2-monodromy-identity}
\end{equation}
is a direct consequence of the composition law and periodicity.  Namely,
\[
\Phi(x+L,x_0)
=
\Phi(x+L,x_0+L)\Phi(x_0+L,x_0),
\]
and the two factors on the right are, respectively,
\(\Phi(x,x_0)\) and \(T\).

Changing the base point conjugates the monodromy matrix.  If
\(x_1\) is another base point, then
\begin{equation}
T(\lambda;x_1,t_0)
=
\Phi(x_1,x_0)T(\lambda;x_0,t_0)\Phi(x_0,x_1).
\label{eq:note2-basepoint-conjugation}
\end{equation}
Here is the computation.  Starting from the definition at the new base
point,
\[
T(\lambda;x_1,t_0)
=
\Phi(x_1+L,x_1).
\]
Split the propagation from \(x_1\) to \(x_1+L\) through the intermediate
points \(x_0\) and \(x_0+L\):
\[
\Phi(x_1+L,x_1)
=
\Phi(x_1+L,x_0+L)\Phi(x_0+L,x_0)\Phi(x_0,x_1).
\]
By periodicity,
\(\Phi(x_1+L,x_0+L)=\Phi(x_1,x_0)\), and by definition
\(\Phi(x_0+L,x_0)=T(\lambda;x_0,t_0)\).  This gives
\eqref{eq:note2-basepoint-conjugation}.  Since
\(\Phi(x_0,x_1)=\Phi(x_1,x_0)^{-1}\), the two monodromy matrices are
conjugate.
Therefore the eigenvalues of \(T\), and hence the spectral curve below,
do not depend on the chosen base point.

\noteheading{Bloch solutions and Floquet multipliers.}
Let \(v\ne0\) be an eigenvector of the monodromy matrix:
\begin{equation}
Tv=\kappa v.
\label{eq:note2-monodromy-eigenvalue}
\end{equation}
Define
\begin{equation}
\Psi(x)=\Phi(x,x_0)v.
\label{eq:note2-bloch-from-eigenvector}
\end{equation}
Using \eqref{eq:note2-monodromy-identity}, we get
\begin{align}
\Psi(x+L)
&=
\Phi(x+L,x_0)v
\nonumber\\
&=
\Phi(x,x_0)Tv
=
\kappa\Phi(x,x_0)v
=
\kappa\Psi(x).
\label{eq:note2-bloch-relation}
\end{align}
Thus an eigenvector of the monodromy matrix at one base point produces a
solution satisfying
\begin{equation}
\Psi(x+L)=\kappa\Psi(x).
\label{eq:note2-bloch-condition}
\end{equation}
The number \(\kappa\) is a Floquet multiplier, or Bloch multiplier, and
\(\Psi\) is a Bloch solution.  Repeating the same relation gives
\begin{equation}
\Psi(x+mL)=\kappa^m\Psi(x),
\qquad
m\in\mathbb{Z}.
\label{eq:note2-bloch-iterate}
\end{equation}
Hence \(|\kappa|=1\) corresponds to bounded oscillation from period to
period, while \(|\kappa|\ne1\) corresponds to exponential growth in one
spatial direction and decay in the other.

It is convenient to write
\begin{equation}
\kappa=\mathrm{e}^{\mathrm{i}Lp},
\label{eq:note2-quasimomentum}
\end{equation}
where \(p\) is the quasimomentum.  If
\(\Psi(x+L)=\mathrm{e}^{\mathrm{i}Lp}\Psi(x)\), then
\begin{equation}
q(x)=\mathrm{e}^{-\mathrm{i}px}\Psi(x)
\label{eq:note2-periodic-factor}
\end{equation}
satisfies \(q(x+L)=q(x)\).  Indeed,
\[
q(x+L)
=
\mathrm{e}^{-\mathrm{i}p(x+L)}
\mathrm{e}^{\mathrm{i}Lp}\Psi(x)
=
\mathrm{e}^{-\mathrm{i}px}\Psi(x)
=
q(x).
\]
Thus a Bloch solution has the Floquet form
\begin{equation}
\Psi(x)=\mathrm{e}^{\mathrm{i}px}q(x),
\qquad
q(x+L)=q(x).
\label{eq:note2-floquet-form}
\end{equation}
Moreover,
\begin{equation}
|\kappa|
=
\left|\mathrm{e}^{\mathrm{i}Lp}\right|
=
\mathrm{e}^{-L\operatorname{Im}p},
\label{eq:note2-kappa-modulus}
\end{equation}
so \(|\kappa|=1\) is equivalent to
\begin{equation}
\operatorname{Im}p=0.
\label{eq:note2-real-quasimomentum}
\end{equation}

\noteheading{Determinant and reciprocal multipliers.}
The spatial matrix \eqref{eq:note2-U} is traceless:
\begin{equation}
\operatorname{tr}U(\lambda,x,t_0)=0.
\label{eq:note2-traceless-U}
\end{equation}
By Liouville's formula for matrix ODEs,
\begin{equation}
\frac{\partial}{\partial x}\det\Phi(x,y)
=
\operatorname{tr}U(\lambda,x,t_0)\det\Phi(x,y).
\label{eq:note2-liouville}
\end{equation}
Therefore \(\det\Phi(x,y)\) is constant in \(x\).  Since
\(\Phi(y,y)=I\), we obtain
\begin{equation}
\det\Phi(x,y)=1.
\label{eq:note2-det-fundamental}
\end{equation}
In particular,
\begin{equation}
\det T(\lambda;x_0,t_0)=1.
\label{eq:note2-det-monodromy}
\end{equation}
If the two Floquet multipliers are denoted by \(\kappa_+\) and
\(\kappa_-\), then
\begin{equation}
\kappa_+(\lambda)\kappa_-(\lambda)=1.
\label{eq:note2-reciprocal-multipliers}
\end{equation}
Thus the two sheets of the spectral curve correspond to reciprocal
Floquet multipliers.

\noteheading{Floquet discriminant and the spectral curve.}
Define the Floquet discriminant by
\begin{equation}
\Delta(\lambda)=\operatorname{tr}T(\lambda;x_0,t_0).
\label{eq:note2-floquet-discriminant}
\end{equation}
The Floquet multipliers are the eigenvalues of \(T\), so they are the
roots of the characteristic polynomial.  For a \(2\times2\) matrix,
\[
\det(\kappa I-T)
=
\kappa^2-\operatorname{tr}T\,\kappa+\det T
=
\kappa^2-\Delta(\lambda)\kappa+1,
\]
where we used \(\operatorname{tr}T=\Delta(\lambda)\) and \(\det T=1\).
Thus the characteristic equation is
\begin{equation}
\kappa^2-\Delta(\lambda)\kappa+1=0.
\label{eq:note2-characteristic-equation}
\end{equation}
Therefore
\begin{equation}
\kappa_\pm(\lambda)
=
\frac{
\Delta(\lambda)
\pm
\sqrt{\Delta(\lambda)^2-4}
}{2}.
\label{eq:note2-kappa-pm}
\end{equation}
The square root becomes single-valued after passing to the two-sheeted
spectral curve
\begin{equation}
\nu^2=\Delta(\lambda)^2-4.
\label{eq:note2-spectral-curve}
\end{equation}
A point on the spectral curve is written as
\(\gamma=(\lambda,\nu)\), and the corresponding multiplier is
\begin{equation}
\kappa(\gamma)
=
\frac{\Delta(\lambda)+\nu}{2}.
\label{eq:note2-kappa-on-curve}
\end{equation}
The two points above the same value of \(\lambda\) give the two
reciprocal multipliers.

\noteheading{Periodic and antiperiodic points.}
A periodic eigenfunction satisfies
\begin{equation}
\Psi(x+L)=\Psi(x),
\label{eq:note2-periodic-eigenfunction}
\end{equation}
so its multiplier is \(\kappa=1\).  An antiperiodic eigenfunction
satisfies
\begin{equation}
\Psi(x+L)=-\Psi(x),
\label{eq:note2-antiperiodic-eigenfunction}
\end{equation}
so its multiplier is \(\kappa=-1\).  Since
\(\kappa+\kappa^{-1}=\Delta(\lambda)\), the periodic and antiperiodic
points are characterized by
\begin{equation}
\Delta(\lambda)=2
\qquad\text{and}\qquad
\Delta(\lambda)=-2,
\label{eq:note2-periodic-antiperiodic-discriminant}
\end{equation}
or, equivalently,
\begin{equation}
\Delta(\lambda)^2-4=0.
\label{eq:note2-branch-double-condition}
\end{equation}
These are the points where the two Floquet multipliers coincide:
\(\kappa_+=\kappa_-=\pm1\).  If the zero of
\(\Delta^2-4\) is simple, the point is a branch point of the spectral
curve.  Even-order zeros may instead be double or resonant points.

\noteheading{Main spectrum.}
The main spectrum is the spectrum of the spatial operator on the line.
In Floquet terms, it is the projection to the \(\lambda\)-plane of the
points on the spectral curve whose Bloch multiplier has unit modulus:
\begin{equation}
\lambda\in\sigma_{\mathrm{main}}
\quad\Longleftrightarrow\quad
|\kappa(\gamma)|=1
\quad
\text{for some }\gamma\text{ above }\lambda.
\label{eq:note2-main-spectrum}
\end{equation}
Equivalently, using \(\kappa=\mathrm{e}^{\mathrm{i}Lp}\),
\begin{equation}
\operatorname{Im}p(\gamma)=0.
\label{eq:note2-main-spectrum-quasimomentum}
\end{equation}
The corresponding Bloch functions are generalized eigenfunctions; they
need not be in \(L^2(\mathbb{R})\), just as plane waves for a constant
coefficient operator are not square integrable on the whole line.  Since
the focusing NLS spectral operator is not self-adjoint, the main
spectrum may contain complex arcs, not only real intervals.

\noteheading{Auxiliary spectrum and divisor.}
The main spectrum determines the spectral curve, but not the point of
the solution inside the isospectral family.  For this additional data one
uses the auxiliary spectrum, also called the divisor.  In the convention
used in the paper, an auxiliary point is a point
\(\gamma\in\Gamma\) such that the first component of the Bloch
eigenfunction vanishes at the base point:
\begin{equation}
\Psi_1(\gamma,x_0,t)=0.
\label{eq:note2-auxiliary-condition}
\end{equation}
Because the same eigenfunction satisfies the Bloch relation, this also
implies
\begin{equation}
\Psi_1(\gamma,x_0+L,t)
=
\kappa(\gamma)\Psi_1(\gamma,x_0,t)
=
0.
\label{eq:note2-auxiliary-endpoint}
\end{equation}
Thus the auxiliary spectrum is equivalently a Dirichlet-type spectrum on
\([x_0,x_0+L]\):
\begin{equation}
\Psi_1(\lambda,x_0,t)=0,
\qquad
\Psi_1(\lambda,x_0+L,t)=0.
\label{eq:note2-dirichlet-type}
\end{equation}

In terms of the monodromy matrix, write
\begin{equation}
T(\lambda)
=
\begin{pmatrix}
T_{11}(\lambda) & T_{12}(\lambda)
\\
T_{21}(\lambda) & T_{22}(\lambda)
\end{pmatrix}.
\label{eq:note2-monodromy-entries}
\end{equation}
To see why the off-diagonal entry appears, choose the solution whose
initial value at \(x_0\) is
\begin{equation}
\Psi(x_0)
=
e_2
=
\begin{pmatrix}
0\\
1
\end{pmatrix}.
\label{eq:note2-auxiliary-initial-vector}
\end{equation}
Then the first component at \(x_0\) is zero.  After one period,
\begin{equation}
\Psi(x_0+L)
=
T(\lambda)
\begin{pmatrix}
0\\
1
\end{pmatrix}.
\label{eq:note2-auxiliary-after-period}
\end{equation}
Using \eqref{eq:note2-monodromy-entries}, this is
\begin{equation}
T(\lambda)e_2
=
\begin{pmatrix}
T_{12}(\lambda)
\\
T_{22}(\lambda)
\end{pmatrix}.
\label{eq:note2-Te2}
\end{equation}
The endpoint Dirichlet condition is therefore
\begin{equation}
T_{12}(\lambda)=0.
\label{eq:note2-T12-condition}
\end{equation}
When this holds, \(e_2\) is also an eigenvector of \(T\), with multiplier
\(\kappa=T_{22}(\lambda)\).  Hence an auxiliary spectral point is not
only a value of \(\lambda\); one must also specify which sheet of the
spectral curve is chosen, or equivalently which multiplier is attached
to the zero.  This is why the auxiliary spectrum is naturally a divisor
on the spectral curve.

\noteheading{Bridge to the finite-gap construction.}
The periodic spectral theory gives two kinds of data.  The main spectrum
determines the spectral curve through the Floquet discriminant and the
branch or double points.  The auxiliary divisor determines the phase
position on that curve.  In finite-gap theory, the \(x\)- and
\(t\)-dependence of the divisor becomes linear after the Abel transform,
and the potential is reconstructed by theta functions.  Those theta
function formulas are not derived here; the goal of this note is only to
set up the spectral objects used later in Section 3.
