\Week{Algebro-Geometric Solutions for the mKdV Equation (1)}{2025-09-16}

\obj{Derive the mKdV equation from the associated spectral problem and construct the AKNS hierarchy by explicit recursion. } 

% \summary{Starting from the Lax pair, we expand the formal solution, obtain recursive relations for the coefficients, and use them to derive the evolution equations. This yields the mKdV equation as a member of the AKNS hierarchy, with all steps carried out explicitly.} 


The focusing modified Korteweg--de Vries (mKdV) equation is given by
\begin{equation}\label{eq:mkdv}
u_t + 6u^2 u_x + u_{xxx} = 0,
\qquad u(x,t) \in \mathbb{R}, \quad (x,t) \in \mathbb{R}^2.
\end{equation}

A standard method for establishing integrability is via a Lax pair. We introduce a matrix function $\Phi(x,t)$ satisfying
\begin{equation*}
\Phi_x = U \Phi, 
\qquad 
\Phi_t = V \Phi,
\end{equation*}
where $U$ and $V$ are $2 \times 2$ matrices depending on $u$ and its derivatives. The compatibility condition
\begin{equation*}
\Phi_{xt} = \Phi_{tx}
\quad \Longleftrightarrow \quad
U_t - V_x + [U,V] = 0,
\end{equation*}
with $[U,V] = UV - VU$, is equivalent to the zero-curvature equation.

Following the AKNS formalism, the spatial part of the Lax pair is taken as
\begin{equation}\label{eq:AKNS-x}
\Phi_x = U \Phi,
\qquad
U = -\mathrm{i} \lambda \sigma_3 + Q,
\end{equation}
where
\begin{equation}\label{eq:Q-sigma3}
Q = \begin{pmatrix} 0 & u \\ -u & 0 \end{pmatrix},
\qquad
\sigma_3 = \operatorname{diag}(1,-1).
\end{equation}

We introduce the transformation
\begin{equation}\label{eq:Phi-gauge}
\Phi(x;\lambda) = m(x;\lambda) e^{-\mathrm{i}\lambda x \sigma_3}, 
\qquad 
m(x;\lambda) \to \mathbb{I} \quad \text{as } \lambda \to \infty.
\end{equation}
Substituting into the spectral problem \eqref{eq:AKNS-x} gives
\begin{equation}\label{eq:lax}
\Phi_x \Phi^{-1} 
= m_x m^{-1} - \mathrm{i}\lambda m\sigma_3 m^{-1}
= -\mathrm{i}\lambda \sigma_3 + Q.
\end{equation}
Define
\begin{equation}\label{eq:Psi-def}
\Psi = m \sigma_3 m^{-1}.
\end{equation}
Differentiating $\Psi$ and using \eqref{eq:lax}, we compute
\begin{align*}
\Psi_x 
&= m_x \sigma_3 m^{-1} - m \sigma_3 m^{-1} m_x m^{-1} \\
&= [m_x m^{-1}, m \sigma_3 m^{-1}] \\
&= \big[m_x m^{-1} - \mathrm{i}\lambda m\sigma_3 m^{-1}, m \sigma_3 m^{-1}\big] \\
&= [U,\Psi].
\end{align*}
Thus $\Psi$ satisfies
\begin{equation}\label{eq:zero-curvature}
\Psi_x = [U,\Psi],
\end{equation}
which is the stationary zero-curvature equation.

The matrix $m(x;\lambda)$ admits the asymptotic expansion
\begin{equation}\label{eq:m-expansion}
m(x;\lambda) = \mathbb{I} + \lambda^{-1} m_1 + \lambda^{-2} m_2 + \cdots.
\end{equation}
From \eqref{eq:Psi-def} and \eqref{eq:m-expansion} we obtain the formal expansion
\begin{equation}\label{eq:Psi-expansion}
\Psi = \Psi_0 + \Psi_1 \lambda^{-1} + \Psi_2 \lambda^{-2} + \cdots,
\qquad 
\Psi_0 = \sigma_3.
\end{equation}
Substituting this into \eqref{eq:zero-curvature} yields
\begin{align*}
\sum_{j=1}^{\infty} \Psi_{j,x} \lambda^{-j} 
&= - \mathrm{i}\lambda \Big[ \sigma_3, \sum_{j=1}^{\infty} \Psi_j \lambda^{-j} \Big]
+ \Big[ Q, \sum_{j=0}^\infty \Psi_j \lambda^{-j} \Big] \\
&= - \mathrm{i}\sum_{j=0}^{\infty} \Big[ \sigma_3,  \Psi_{j+1}  \Big] \lambda^{-j}
+ \sum_{j=0}^\infty \Big[ Q,  \Psi_j \Big] \lambda^{-j}.
\end{align*}

At order $O(1)$ we find
\begin{equation*}
- \mathrm{i} [\sigma_3, \Psi_1] + [Q, \sigma_3] = 0 
\quad \Longleftrightarrow \quad [\mathrm{i}\Psi_1 + Q, \sigma_3] = 0,
\end{equation*}
which implies
\begin{equation*}
\Psi_1^{\mathrm{off}} = \mathrm{i} Q.
\end{equation*}

For general $j \geq 1$, the coefficient of $\lambda^{-j}$ gives
\begin{equation}\label{eq:Psi-j-recursion}
\Psi_{j,x} 
= - \mathrm{i}[\sigma_3,\Psi_{j+1}] + [Q,\Psi_j]
= -2\mathrm{i}\sigma_3 \Psi_{j+1}^{\mathrm{off}} + [Q,\Psi_j].
\end{equation}
Hence
\begin{equation}\label{eq:Psi-j-off-recursion}
\Psi_{j,x}^{\mathrm{off}} 
= -2\mathrm{i} \sigma_3 \Psi_{j+1}^{\mathrm{off}} + [Q,\Psi_j^{\mathrm{diag}}]
\quad \Longleftrightarrow \quad
\Psi_{j+1}^{\mathrm{off}} 
= \tfrac{\mathrm{i}}{2}\sigma_3\big(\Psi_{j,x}^{\mathrm{off}} - [Q,\Psi_j^{\mathrm{diag}}]\big),
\end{equation}
and
\begin{equation}\label{eq:Psi-j-diag-recursion}
\Psi_{j,x}^{\mathrm{diag}} = [Q,\Psi_j^{\mathrm{off}}]
\quad \Longleftrightarrow \quad
\Psi_j^{\mathrm{diag}} = \partial_x^{-1}\big([Q,\Psi_j^{\mathrm{off}}]\big).
\end{equation}

Thus, the recursion relation \eqref{eq:Psi-j-off-recursion} provides a direct way to compute the
off-diagonal coefficients algebraically in terms of lower-order terms, while the diagonal recursion
\eqref{eq:Psi-j-diag-recursion} requires performing an integration. In practice,
this makes the off-diagonal part more convenient to obtain explicitly, 
whereas the diagonal part is generally harder to compute in closed form.

By \eqref{eq:Psi-def}, the additional constraint $\Psi^2 = \mathbb{I}$ provides algebraic relations among the coefficients. Substituting the expansion \eqref{eq:Psi-expansion} yields
\begin{equation*}
\mathbb{I} = \left( \sum_{j=0}^\infty \Psi_j \lambda^{-j} \right)^2
= \sum_{j=0}^\infty \left( \sum_{k=0}^j \Psi_k \Psi_{j-k} \right) \lambda^{-j},
\end{equation*}
so that
\begin{equation}\label{eq:Psi-sq-coeff}
\sum_{k=0}^j \Psi_k \Psi_{j-k} = 0, 
\qquad j \geq 1.
\end{equation}
Separating diagonal and off-diagonal parts, for $j \geq 1$ we obtain
\begin{equation}\label{eq:Psi-sq-diag}
    \sum_{k=0}^j \big( \Psi_k^{\mathrm{diag}} \Psi_{j-k}^{\mathrm{diag}} + \Psi_k^{\mathrm{off}} \Psi_{j-k}^{\mathrm{off}} \big) = 0,
\end{equation}
\begin{equation}\label{eq:Psi-sq-off-diag}
\sum_{k=0}^j \big( \Psi_k^{\mathrm{off}} \Psi_{j-k}^{\mathrm{diag}} + \Psi_k^{\mathrm{diag}} \Psi_{j-k}^{\mathrm{off}} \big) = 0.
\end{equation}

Since $\Psi_0 = \sigma_3$, the off-diagonal constraint \eqref{eq:Psi-sq-off-diag} for $j \geq 2$ takes the form
\begin{equation*}
\sigma_3 \Psi_j^{\mathrm{off}} + \Psi_j^{\mathrm{off}} \sigma_3 
+ \sum_{k=1}^{j-1} \left( \Psi_k^{\mathrm{off}} \Psi_{j-k}^{\mathrm{diag}} + \Psi_k^{\mathrm{diag}} \Psi_{j-k}^{\mathrm{off}} \right) = 0.
\end{equation*}
Noting that $\sigma_3 \Psi_j^{\mathrm{off}} + \Psi_j^{\mathrm{off}} \sigma_3 = 0$, this simplifies to
\begin{equation}\label{eq:Psi-sq-off-diag-reduced}
\sum_{k=1}^{j-1} \left( \Psi_k^{\mathrm{off}} \Psi_{j-k}^{\mathrm{diag}} + \Psi_k^{\mathrm{diag}} \Psi_{j-k}^{\mathrm{off}} \right) = 0.
\end{equation}

It should be noted that from the off-diagonal constraint \eqref{eq:Psi-sq-off-diag} one can only deduce 
the reduced form \eqref{eq:Psi-sq-off-diag-reduced}, which merely expresses a compatibility condition 
between lower-order coefficients. In particular, this relation does not provide a practical means 
to compute the diagonal parts explicitly.

For the diagonal part \eqref{eq:Psi-sq-diag} we have
\begin{equation*}
\sigma_3 \Psi_j^{\mathrm{diag}} + \Psi_j^{\mathrm{diag}} \sigma_3 
+ \sum_{k=1}^{j-1} \big( \Psi_k^{\mathrm{diag}} \Psi_{j-k}^{\mathrm{diag}} + \Psi_k^{\mathrm{off}} \Psi_{j-k}^{\mathrm{off}} \big) = 0,
\qquad j \geq 2.
\end{equation*}
Since $\sigma_3 \Psi_j^{\mathrm{diag}} = \Psi_j^{\mathrm{diag}} \sigma_3$, this simplifies to
\begin{equation}\label{eq:Psi-sq-diag-reduced}
\Psi_j^{\mathrm{diag}} 
= -\tfrac{1}{2} \sigma_3 \sum_{k=1}^{j-1} 
\big( \Psi_k^{\mathrm{diag}} \Psi_{j-k}^{\mathrm{diag}} + \Psi_k^{\mathrm{off}} \Psi_{j-k}^{\mathrm{off}} \big),
\qquad j \geq 2.
\end{equation}
We will use this reduced relation \eqref{eq:Psi-sq-diag-reduced} to compute the diagonal part $\Psi_j^{\mathrm{diag}}$ recursively at each order. This provides an explicit algebraic formula for the diagonal coefficients in terms of lower-order diagonal and off-diagonal terms.

For $j=1$, the condition \eqref{eq:Psi-sq-coeff} reads
\begin{equation*}
\sigma_3 \Psi_1 + \Psi_1 \sigma_3 = 2\sigma_3 \Psi_1^{\mathrm{diag}} = 0,
\end{equation*}
so that $\Psi_1 = \Psi_1^{\mathrm{off}} = \mathrm{i}Q$.

Using \eqref{eq:Psi-j-off-recursion} and \eqref{eq:Psi-sq-diag-reduced} recursively, we find
\begin{equation*}
\Psi_2^{\mathrm{off}} = -\tfrac{1}{2} \sigma_3 Q_x,
\qquad
\Psi_2^{\mathrm{diag}} = \tfrac{1}{2} \sigma_3 Q^2,
\end{equation*}
and hence
\begin{equation*}
\Psi_2 = \tfrac{1}{2} \sigma_3 (Q^2 - Q_x).
\end{equation*}

For the next coefficient, we compute
\begin{align*}
\Psi_3^{\mathrm{off}} 
&= \tfrac{\mathrm{i}}{2} \sigma_3 \big( \Psi_{2,x}^{\mathrm{off}} - [Q,\Psi_2^{\mathrm{diag}}] \big) \\
&= \tfrac{\mathrm{i}}{2} \sigma_3 \left( -\tfrac{1}{2} \sigma_3 Q_{xx} - [Q, \tfrac{1}{2} \sigma_3 Q^2] \right) \\
&= \tfrac{\mathrm{i}}{2} \sigma_3 \left( -\tfrac{1}{2} \sigma_3 Q_{xx} - \tfrac{1}{2}(Q\sigma_3 Q^2 - \sigma_3 Q^2 Q) \right) \\
&= -\tfrac{\mathrm{i}}{4} Q_{xx} + \tfrac{\mathrm{i}}{2} Q^3.
\end{align*}
For the diagonal part,
\begin{align*}
\Psi_3^{\mathrm{diag}} 
&= -\tfrac{1}{2} \sigma_3 \Big( \Psi_1^{\mathrm{off}}\Psi_2^{\mathrm{off}} + \Psi_2^{\mathrm{off}}\Psi_1^{\mathrm{off}} 
+ \Psi_1^{\mathrm{diag}}\Psi_2^{\mathrm{diag}} + \Psi_2^{\mathrm{diag}}\Psi_1^{\mathrm{diag}} \Big) \\
&= -\tfrac{1}{2} \sigma_3 \Big( (\mathrm{i}Q)(-\tfrac{1}{2}\sigma_3 Q_x) + (-\tfrac{1}{2}\sigma_3 Q_x)(\mathrm{i}Q) \Big) \\
&= -\tfrac{1}{2} \sigma_3 \left( -\tfrac{\mathrm{i}}{2}(Q\sigma_3 Q_x + \sigma_3 Q_x Q) \right) \\
&= \tfrac{\mathrm{i}}{4}\big(Q_x Q - Q Q_x\big) = 0.
\end{align*}

Proceeding to $\Psi_4$, the off-diagonal recursion \eqref{eq:Psi-j-off-recursion} with $\Psi_3^{\mathrm{diag}}=0$ gives
\begin{align*}
\Psi_4^{\mathrm{off}}
&= \tfrac{\mathrm{i}}{2}\sigma_3\big(\Psi_{3,x}^{\mathrm{off}} - [Q,\Psi_3^{\mathrm{diag}}]\big)
= \tfrac{\mathrm{i}}{2}\sigma_3 \Psi_{3,x}^{\mathrm{off}} \\
&= \tfrac{\mathrm{i}}{2}\sigma_3 \partial_x\!\big(-\tfrac{\mathrm{i}}{4} Q_{xx} + \tfrac{\mathrm{i}}{2} Q^3\big) \\
&= \tfrac{\mathrm{i}}{2}\sigma_3 \big(-\tfrac{\mathrm{i}}{4} Q_{xxx} + \tfrac{\mathrm{i}}{2} (Q^3)_x\big) \\
&= \tfrac{1}{8}\sigma_3 Q_{xxx} - \tfrac{1}{4}\sigma_3 (Q^3)_x.
\end{align*}
Since $Q_x Q = Q Q_x$ for the present $Q$ given in \eqref{eq:Q-sigma3}, we have
\[
(Q^3)_x = Q_x Q^2 + Q Q_x Q + Q^2 Q_x = 3 Q_x Q^2,
\]
and therefore
\begin{equation*}
\Psi_4^{\mathrm{off}} = \tfrac{1}{8}\sigma_3 Q_{xxx} - \tfrac{3}{4}\sigma_3 Q_x Q^2.
\end{equation*}
For the diagonal part, only the nonvanishing combinations contribute; terms involving $\Psi_1^{\mathrm{diag}}$ or $\Psi_3^{\mathrm{diag}}$ are zero and omitted. Hence
\begin{align*}
\Psi_4^{\mathrm{diag}}
&= -\tfrac{1}{2}\sigma_3\Big(\Psi_1^{\mathrm{off}}\Psi_3^{\mathrm{off}} + \Psi_3^{\mathrm{off}}\Psi_1^{\mathrm{off}}
+ \Psi_2^{\mathrm{off}}\Psi_2^{\mathrm{off}} + \Psi_2^{\mathrm{diag}}\Psi_2^{\mathrm{diag}}\Big) \\
&= -\tfrac{1}{2}\sigma_3\Big((\mathrm{i}Q)\big(-\tfrac{\mathrm{i}}{4}Q_{xx}+\tfrac{\mathrm{i}}{2}Q^3\big)
+ \big(-\tfrac{\mathrm{i}}{4}Q_{xx}+\tfrac{\mathrm{i}}{2}Q^3\big)(\mathrm{i}Q) \\
&\qquad\qquad\quad + \big(-\tfrac{1}{2}\sigma_3 Q_x\big)^2 + \big(\tfrac{1}{2}\sigma_3 Q^2\big)^2\Big) \\
&= -\tfrac{1}{8}\sigma_3\big(Q Q_{xx} + Q_{xx} Q - 3Q^4 - Q_x^2\big).
\end{align*}

Since $(-\mathrm{i}\lambda \Psi)_{+} = -\mathrm{i}\lambda \sigma_3 + Q$, we obtain
\begin{equation*}
\Phi_x \Phi^{-1} = U = (-\mathrm{i}\lambda \Psi)_{+}.
\end{equation*}
Here the subscript $(\cdot)_{+}$ denotes the projection onto the nonnegative powers of $\lambda$ in the Laurent expansion.  
This motivates the definition of the AKNS hierarchy by prescribing the time evolution
\begin{equation}\label{eq:AKNS-hierarchy}
\Phi_{t_n} = \big(-\mathrm{i}\lambda^n \Psi\big)_{+}\Phi \equiv V_n \Phi.
\end{equation}

The first few members of the hierarchy are given by
\begin{align*}
n=2 &\quad \Rightarrow \quad \text{nonlinear Schrödinger (NLS) flow}, \\
n=3 &\quad \Rightarrow \quad \text{complex mKdV flow}, \quad \Rightarrow \quad \text{mKdV flow}.
\end{align*}
In terms of the expansion coefficients $\Psi_j$, the operators $V_n$ take the form
\begin{align*}
V_2 &= -\mathrm{i}\big(\lambda^2 \Psi_0 + \lambda \Psi_1 + \Psi_2\big), \\
V_3 &= -\mathrm{i}\big(\lambda^3 \Psi_0 + \lambda^2 \Psi_1 + \lambda \Psi_2 + \Psi_3\big).
\end{align*}

The zero-curvature equation \(U_t - V_x + [U,V] = 0\) with \(V_{\mathrm{mKdV}} = 4V_3\) gives, by collecting the constant terms in \(\lambda\) (i.e., at order \(O(1)\)),
\begin{equation*}
Q_t + 4\mathrm{i} \Psi_{3,x} + [Q,-4\mathrm{i}\Psi_3] = 0.
\end{equation*}
Taking the off-diagonal part gives \(Q_t + 4\mathrm{i}\Psi_{3,x}^{\mathrm{off}} = 0\), 
that is,
\begin{align}\label{eq:matrix-mkdv}    
Q_t + 4\mathrm{i}\Big(-\tfrac{\mathrm{i}}{4}Q_{xx}+\tfrac{\mathrm{i}}{2}Q^3\Big)_x
&= Q_t + Q_{xxx} - 2(Q^3)_x = 0.
\end{align}
Since
\begin{equation*}
Q = \begin{pmatrix} 0 & u \\ -u & 0 \end{pmatrix}, 
\qquad Q^2 = -u^2 \mathbb{I},
\end{equation*}
this reduces to the scalar equation
\begin{equation*}
u_t + u_{xxx} + 2(u^3)_x = 0,
\end{equation*}
which is precisely the mKdV equation \eqref{eq:mkdv}.

\extra

Using the definitions \eqref{eq:Phi-gauge} and  \eqref{eq:Psi-def} we have
\begin{equation}\label{eq:Psi-def-new}
\Psi = \Phi \sigma_3 \Phi^{-1}.
\end{equation}
Differentiating $\Psi$ with respect to $t_n$ and using \eqref{eq:AKNS-hierarchy} gives
\begin{align*}
\Psi_{t_n}
&= \Phi_{t_n}\sigma_3 \Phi^{-1} - \Phi\sigma_3 \Phi^{-1} \Phi_{t_n} \Phi^{-1} \\
&= [\Phi_{t_n} \Phi^{-1},\, \Psi] = [V_n,\Psi].
\end{align*}
Thus
\begin{equation}\label{eq:Psi-t-comm}
\Psi_{t_n} = [V_n,\Psi].
\end{equation}

Next we simplify the zero-curvature equation for the pair $(U,V_n)$.
The zero-curvature equation is
\[
U_{t_n} - (V_n)_x + [U,V_n]=0,
\]
and since $U=-\mathrm{i}\lambda\sigma_3+Q$ we have $U_{t_n}=Q_{t_n}$.  
With the normalization
\[
V_n=\alpha_n\big(-\mathrm{i}\lambda^n\Psi\big)_{+}
\]
we obtain
\begin{equation*}
Q_{t_n}=(V_n)_x - [U,V_n]
= \alpha_n\big(-\mathrm{i}\lambda^n\Psi_x\big)_{+} - [U,\alpha_n\big(-\mathrm{i}\lambda^n\Psi\big)_{+}].
\end{equation*}
Pulling out constants and using $\Psi_x=[U,\Psi]$ gives
\begin{equation}\label{eq:interm-refined}
Q_{t_n} = -\mathrm{i}\alpha_n\Big(\big(\lambda^n[U,\Psi]\big)_{+} - [U,\big(\lambda^n\Psi\big)_{+}]\Big).
\end{equation}

Write $U=-\mathrm{i}\lambda\sigma_3+Q$ and split the first term in \eqref{eq:interm-refined}:
\[
\lambda^n[U,\Psi] = -\mathrm{i}\lambda^{n+1}[\sigma_3,\Psi] + \lambda^n[Q,\Psi].
\]
Because $Q$ does not depend on $\lambda$ we have
\[
\big(\lambda^n[Q,\Psi]\big)_{+} = [Q,\big(\lambda^n\Psi\big)_{+}],
\]
so the $\!Q$--commutator contributions cancel in \eqref{eq:interm-refined}. The remaining terms therefore come only from the $\sigma_3$--commutator:
\begin{align*}
Q_{t_n}
&= -\mathrm{i}\alpha_n\Big(-\mathrm{i}\big(\lambda^{n+1}[\sigma_3,\Psi]\big)_{+} + \mathrm{i}\,\lambda[\sigma_3,\big(\lambda^n\Psi\big)_{+}]\Big)\\
&= \alpha_n\Big(\lambda[\sigma_3,\big(\lambda^n\Psi\big)_{+}] - \big(\lambda^{n+1}[\sigma_3,\Psi]\big)_{+}\Big).
\end{align*}

To simplify the difference of finite sums, use the expansions
\[
\Psi=\sum_{j\ge0}\Psi_j\lambda^{-j},\qquad
(\lambda^n\Psi)_{+}=\sum_{j=0}^n\lambda^{\,n-j}\Psi_j.
\]
A direct comparison of coefficients shows
\begin{align*}
\lambda[\sigma_3,(\lambda^n\Psi)_{+}] - \big(\lambda^{n+1}[\sigma_3,\Psi]\big)_{+}
&= \sum_{j=0}^n \lambda^{\,n+1-j}[\sigma_3,\Psi_j] - \sum_{j=0}^{n+1} \lambda^{\,n+1-j}[\sigma_3,\Psi_j] \\
&= -\lambda^{\,n+1-(n+1)}[\sigma_3,\Psi_{n+1}] \\
&= -[\sigma_3,\Psi_{n+1}].
\end{align*}
Hence the evolution of $Q$ is
\begin{equation}\label{eq:Q-t-final-refined}
Q_{t_n} = -\alpha_n\,[\sigma_3,\Psi_{n+1}].
\end{equation}
Since diagonal parts commute with $\sigma_3$ and the commutator doubles the off-diagonal part, we may equivalently write
\begin{equation}\label{eq:Q-t-off}
Q_{t_n} = -2\alpha_n\,\sigma_3\,\Psi_{n+1}^{\mathrm{off}}.
\end{equation}


\Week{Algebro-Geometric Solutions for the mKdV Equation (2)}{2025-09-23}

\obj{Construct the \(L\) matrix for mKdV, establish its symmetries and conserved determinant, and relate the eigen-data to the associated hyperelliptic curve and Riemann–Theta functions. } 


Recall the AKNS spectral problem (1974)
\begin{equation}\label{eq:AKNS}
\Phi_x=U(\lambda;Q)\,\Phi,\qquad 
U(\lambda;Q)=-\mathrm{i}\lambda\sigma_3+Q,
\end{equation}
where $\sigma_3=\begin{pmatrix}1&0\\[2pt]0&-1\end{pmatrix}$ and $Q$ is off-diagonal.  In what follows we set
\begin{equation}\label{eq:Q}
Q=\begin{pmatrix}0 & u\\[2pt]-u & 0\end{pmatrix}.
\end{equation}

Introduce the gauge
\begin{gather}
\Phi=m(x;\lambda)\,e^{-\mathrm{i}\lambda x\sigma_3},\qquad
m=\mathbb{I}+\lambda^{-1}m_1+\lambda^{-2}m_2+O(\lambda^{-3}),\\
\Psi:=m\,\sigma_3\,m^{-1}=\Psi_0+\Psi_1\lambda^{-1}+\Psi_2\lambda^{-2}+\cdots,
\end{gather}
so that $\Psi^2=\mathbb{I}$ and $\Psi$ satisfies the stationary zero-curvature equation
\begin{equation}\label{eq:Psi_x}
\Psi_x=[U,\Psi].
\end{equation}

Decompose a matrix $A$ into diagonal and off-diagonal parts $A=A^{\mathrm{diag}}+A^{\mathrm{off}}$.
From \eqref{eq:Psi_x} and $\Psi^2=\mathbb{I}$ one obtains the recursion
\begin{align}
\Psi_{j+1}^{\mathrm{off}}
&=\tfrac{\mathrm{i}}{2}\sigma_3\Big(\Psi_{j,x}^{\mathrm{off}}-[Q,\Psi_j^{\mathrm{diag}}]\Big), \qquad j\ge0, \label{eq:rec_off}\\
\Psi_{j+1}^{\mathrm{diag}}
&=-\tfrac{1}{2}\sigma_3\sum_{k=1}^{j}\Big(\Psi_k^{\mathrm{diag}}\Psi_{\,j-k+1}^{\mathrm{diag}}
+\Psi_k^{\mathrm{off}}\Psi_{\,j-k+1}^{\mathrm{off}}\Big), \qquad j\ge1.\label{eq:rec_diag}
\end{align}
Hence each $\Psi_j$ is a differential polynomial in $Q$ and its $x$-derivatives.

One checks that
\begin{equation}\label{eq:Phi_xPhiinv}
\Phi_x\Phi^{-1}=m_x m^{-1}-\mathrm{i}\lambda m\sigma_3 m^{-1}=-\mathrm{i}(\lambda\Psi)_{+}.
\end{equation}
Let $\Phi(x,t_n;\lambda)=m(x,t_n;\lambda)\,e^{-\mathrm{i}(\lambda x + \lambda^n t_n)\sigma_3}$, then
\begin{equation}\label{eq:Phi_t}
\Phi_{t_n}\Phi^{-1}=m_{t_n}m^{-1}-\mathrm{i}\lambda^n m\sigma_3 m^{-1}=-\mathrm{i}(\lambda^n\Psi)_{+}\equiv V_n,
\end{equation}
so that $V_n=-\mathrm{i}(\lambda^n\Psi)_{+}$ and
\begin{equation}\label{eq:Psi_t}
\Psi_{t_n}=[V_n,\Psi].
\end{equation}
The compatibility $\Phi_{x t_n}=\Phi_{t_n x}$ is equivalent to the zero curvature equation
\begin{equation}\label{eq:zero_curv}
U_{t_n}- V_{n,x}+[U,V_n]=0.
\end{equation}

\begin{definition}
Let $g \geq 0$ be a fixed integer, and let $\{\alpha_j\}_{j=0}^{g+1}$ be constants. Define
\begin{equation}\label{eq:L_def}
L := -\mathrm{i} \sum_{j=0}^{g+1} \alpha_j \big(\lambda^j \Psi(x, t, \lambda)\big)_{+},
\end{equation}
where $(\cdot)_{+}$ denotes the projection onto nonnegative powers of $\lambda$.
\end{definition}

It is convenient to rewrite \(L\) by expanding the nonnegative part explicitly. Using \(\Psi=\sum_{k\ge0}\Psi_k\lambda^{-k}\) we have
\begin{align*}
\big(\lambda^j\Psi\big)_{+}
&=\sum_{k=0}^{j}\Psi_k\lambda^{\,j-k},
\end{align*}
hence
\begin{align}\label{eq:L_expanded}
L(\lambda)
&=-\mathrm{i}\sum_{j=0}^{g+1}\alpha_j\sum_{k=0}^{j}\Psi_k\lambda^{\,j-k}
=-\mathrm{i}\sum_{l=0}^{g+1}\left(\sum_{j=l}^{g+1}\alpha_j\Psi_{j-l}\right)\lambda^{\,l},
\end{align}
where we set \(l=j-k\) to obtain the last form.

\begin{prop}\label{prop:constraint}
Suppose $L$ as defined in \eqref{eq:L_def} satisfies the stationary zero-curvature equations
\begin{equation}\label{eq:stationaryZC}
L_x = [U, L], \qquad L_{t_n} = [V_n, L].
\end{equation}
Then the following additional constraint holds:
\begin{equation}\label{eq:constraint}
\sum_{j=1}^{g+1} \alpha_j\, \Psi_{j+1}^{\mathrm{off}}(x, t_n) + \mathrm{i}\, \alpha_0\, Q = 0.
\end{equation}
\end{prop}

\begin{proof}
Set
\begin{equation}\label{eq:Lhat}
\widehat{L}(\lambda):=-\mathrm{i}\Big(\sum_{j=0}^{g+1}\alpha_j\lambda^j\Big)\Psi(\lambda).
\end{equation}
By construction \(\widehat{L}_x=[U,\widehat{L}]\). Expand \(\Psi=\sum_{k\ge0}\Psi_k\lambda^{-k}\) and substitute into \(\widehat{L}\):
\begin{align}\label{eq:Lhat_series}
\widehat{L}(\lambda)
&=-\mathrm{i}\sum_{j=0}^{g+1}\sum_{k\ge0}\alpha_j\Psi_k\lambda^{\,j-k}
=-\mathrm{i}\sum_{l=-\infty}^{g+1}\Big(\sum_{j=0}^{g+1}\alpha_j\Psi_{\,j-l}\Big)\lambda^{\,l},
\end{align}
where we set \(l=j-k\). Decompose \(\widehat{L}=\widehat{L}_{+}+\widehat{L}_{-}\) into nonnegative and negative powers of \(\lambda\). From \(\widehat{L}_x=[U,\widehat{L}]\) we obtain
$$ \left(\widehat{L}_+ +\widehat{L}_{-}\right)_x=\left[U, \widehat{L}_+ +\widehat{L}_{-}\right]. $$ 

After extracting nonnegative powers of \(\lambda\), we have
\[
(\widehat{L}_{+})_x - [U, \widehat{L}_{+}]
= \left( [-\mathrm{i} \lambda \sigma_3, \widehat{L}_{-}] \right)_{+}
= -\mathrm{i} \left[ \sigma_3, \sum_{j=0}^{g+1} \alpha_j \Psi_{j+1} \right]
= -2\mathrm{i} \sigma_3 \sum_{j=0}^{g+1} \alpha_j \Psi_{j+1}^{\mathrm{off}}  = 0,
\]
which yields the constraint \eqref{eq:constraint}.

Next, to ensure compatibility, we require to prove $L_{x t_n} = L_{t_n x}$. Observe that
\[
\begin{aligned}
L_{x t_n} &= [U_{t_n}, L] + [U, [V_n, L]], \\
L_{t_n x} &= [V_{n, x}, L] + [V_n, [U, L]].
\end{aligned}
\]
Applying the Jacobi identity,
\[
[[A, B], C] + [[C, A], B] + [[B, C], A] = 0,
\]
we find
\[
[U, [V_n, L]] - [V_n, [U, L]] = [[L, V_n],U] + [[U,L], V_n] = [[U, V_n], L].
\]
Therefore, it follows from the zero-curvature condition \eqref{eq:zero_curv} that
\[
L_{x t_n} - L_{t_n x} = [U_{t_n} - V_{n, x} + [U, V_n], L] = 0.
\]
\end{proof}

Specializing to genus \(g=2\), define
\begin{align}
L(x,t;\lambda)&=\tfrac{1}{4}V(\lambda;Q)+\alpha_1 U(\lambda;Q), \label{eq:def-L}\\
V(\lambda;Q)&=4\lambda^2 U(\lambda;Q)
+2\mathrm{i}\lambda\sigma_3\big(Q_x-Q^2\big)
-\big(Q_{xx}-2Q^3\big). \label{eq:def-V}
\end{align}

The zero curvature condition for the mKdV equation is
\[
U_t-V_x+[U,V]=0,
\]
which yields the scalar modified KdV equation
\begin{equation}\label{eq:mKdV}
u_t+u_{xxx}+6u^2 u_x=0,\qquad u(x,t)\in\mathbb{R}.
\end{equation}

The matrices \(U\) and \(V\) satisfy the following symmetries:
\begin{equation}\label{eq:sym-UV}
\begin{aligned}
U^{\dagger}(\lambda^*;Q) &= -U(\lambda;Q), &\qquad U^{\top}(-\lambda;Q)&=-U(\lambda;Q),\\
V^{\dagger}(\lambda^*;Q) &= -V(\lambda;Q), &\qquad V^{\top}(-\lambda;Q)&=-V(\lambda;Q).
\end{aligned}
\end{equation}
As a consequence the matrix \(L\) inherits analogous symmetries:
\begin{equation}\label{eq:sym-L}
L^{\dagger}(x,t;\lambda^*)=-L(x,t;\lambda),\qquad 
L^{\top}(x,t;-\lambda)=-L(x,t;\lambda).
\end{equation}

\begin{prop}\label{prop:det-invariants}
The determinant of \(L\) is independent of \(x\) and \(t\):
\[
\partial_x\det L=\partial_t\det L=0.
\]
\end{prop}
\begin{proof}
By Jacobi's formula (or Abel's theorem applying to \(L_x = (L_x L^{-1}) L\)), for any differentiable matrix \(L(x,t;\lambda)\)
\[
\partial_x\det L=\det L\;\operatorname{tr}\!\big( L_x L^{-1}\big).
\]
Using the stationary zero-curvature equation \(L_x =[U,L]=U L-L U\) we obtain
\[
\partial_x\det L
=\det L\;\operatorname{tr}\!\big((U L-L U)L^{-1}\big)
=\det L\big(\operatorname{tr}U - \operatorname{tr}(L^{-1}L U)  \big)=0,
\]
since \(\operatorname{tr}(L U L^{-1})=\operatorname{tr}(L^{-1}L U)\).
The identical argument with \(\partial_t L=[V,L]\) yields \(\partial_t\det L=0\).
Hence \(\det L\) does not depend on \(x\) or \(t\).
\end{proof}

Therefore the determinant of \(L\) is constant in \(x\) and \(t\). Write its factorization as
\begin{equation}\label{eq:det-symmetry}
\det L(\lambda)=\prod_{j=1}^{6}\big(\lambda-\hat\lambda_j\big).
\end{equation}
By the symmetry relations \eqref{eq:sym-L} each root \(\hat\lambda_j\) forces \(\hat\lambda_j^*\) and \(-\hat\lambda_j\) to be roots as well. Two possible root structures follow.

\begin{enumerate}
  \item[\textbf{Case 1.}] Three purely imaginary pairs:
  \begin{equation}\label{eq:det-case1}
    \det L(\lambda)=
    \bigl(\lambda^{2}-\lambda_{1}^{2}\bigr)
    \bigl(\lambda^{2}-\lambda_{2}^{2}\bigr)
    \bigl(\lambda^{2}-\lambda_{3}^{2}\bigr),
    \qquad \lambda_{j}\in\mathrm{i}\mathbb{R}.
  \end{equation}

  \item[\textbf{Case 2.}] One nonreal quadruple together with one purely imaginary pair:
  \begin{equation}\label{eq:det-case2}
    \det L(\lambda)=
    (\lambda-\lambda_{1})(\lambda-\lambda_{1}^{*})
    (\lambda+\lambda_{1})(\lambda+\lambda_{1}^{*})
    (\lambda-\lambda_{2})(\lambda+\lambda_{2}),
  \end{equation}
  with \(\lambda_{1}\in\mathbb{C}\setminus(\mathbb{R}\cup\mathrm{i}\mathbb{R})\) and 
  \(\lambda_{2}\in\mathrm{i}\mathbb{R}\).
\end{enumerate}

The matrix elements of \(L\) take the explict form
\begin{equation}\label{eq:L-elements}
\begin{aligned}
L_{11} &= -\mathrm{i}\lambda^3-\mathrm{i}\lambda\Big(\alpha_1-\tfrac{1}{2}u^2\Big), \quad L_{22} = -L_{11},\\[4pt]
L_{12} &= u\Big(\lambda^2 + \mathrm{i}\frac{u_x}{2u}\,\lambda -\frac{u_{xx}}{4u}+\alpha_1-\tfrac{1}{2}u^2\Big)
= u(\lambda-\mu_1)(\lambda-\mu_2),\\[6pt]
L_{21} &= -u\Big(\lambda^2 - \mathrm{i}\frac{u_x}{2u}\,\lambda -\frac{u_{xx}}{4u}+\alpha_1-\tfrac{1}{2}u^2\Big)
= -u(\lambda-\mu_1^*)(\lambda-\mu_2^*),
\end{aligned}
\end{equation}
where \(\mu_1,\mu_2\) denote the (in general complex) roots of the quadratic factor in \(L_{12}/u\). The conjugation in \(L_{21}\) is consistent with the reality of \(u\) and the symmetry relations \eqref{eq:sym-L}.


Recall that
\[
\begin{cases}
\Phi_x = U\Phi,\\[4pt]
\Phi_t = V\Phi,
\end{cases}
\qquad
\begin{cases}
L_x = [U,L],\\[4pt]
L_t = [V,L],
\end{cases}
\]
It follows that \(L\Phi\) also satisfies the same Lax pair, since
\begin{align*}
(L\Phi)_x &= L_x\Phi + L\Phi_x = (UL-LU)\Phi + LU\Phi = U(L\Phi),\\
(L\Phi)_t &= V(L\Phi).
\end{align*}
This observation is also a direct consequence of the identity \(L=\Psi=\Phi\,\sigma_3\,\Phi^{-1}\) which implies \(L\Phi=\Phi\sigma_3\).

Assume \(\Phi(x,t;\lambda)\) is normalized by \(\Phi(0,0;\lambda)=\mathbb{I}\). By uniqueness of solutions of the linear ODEs we obtain
\begin{equation}\label{eq:conjugation}
L(x,t;\lambda)\,\Phi(x,t;\lambda)=\Phi(x,t;\lambda)\,L(0,0;\lambda).
\end{equation}
In particular, evaluation at \((x,t)=(0,0)\) yields \(L(0,0;\lambda)\). So knowledge of \(L(0,0;\lambda)\) determines the action of \(L\) on the columns of \(\Phi\) for all \((x,t)\).

Set
\begin{equation}\label{eq:det-L0}
\det\bigl(L(x,t;\lambda)\bigr)=\det\bigl(L(0,0;\lambda)\bigr)=\prod_{j=1}^{3}\bigl(\lambda^2-\lambda_j^2\bigr)=y^2,
\qquad
y=\pm\sqrt{\prod_{j=1}^{3}(\lambda^2-\lambda_j^2)}.
\end{equation}
A short calculation shows that \(\pm \mathrm{i}y\) are eigenvalues of \(L(0,0;\lambda)\):
\[
\det\!\bigl(\pm \mathrm{i}y\,\mathbb{I}-L(0,0;\lambda)\bigr)
= -y^2 - L_{11}^2 - L_{12}L_{21}
= -y^2 + \det\bigl(L(0,0;\lambda)\bigr)=0,
\]
where \(L_{ij}\) denotes the \((i,j)\)-entry of \(L(0,0;\lambda)\).

Hence there exists an invertible matrix \(M\), independent of \(x\) and \(t\), such that
\begin{equation}\label{eq:L0-diag}
L(0,0;\lambda)=
M
\begin{pmatrix}
\mathrm{i}y & 0\\[4pt]
0 & -\mathrm{i}y
\end{pmatrix}
M^{-1}.
\end{equation}
Combining \eqref{eq:conjugation} and \eqref{eq:L0-diag} yields
\[
L(x,t;\lambda)\,\Phi(x,t;\lambda)M
=\Phi(x,t;\lambda)M
\begin{pmatrix}
\mathrm{i}y & 0\\[4pt]
0 & -\mathrm{i}y
\end{pmatrix}.
\]
Write the two columns of \(\Phi(x,t;\lambda)M\) as
\[
\Phi(x,t;\lambda)M=\bigl(\Phi_1,\Phi_2\bigr),
\]
so that
\[
L(x,t;\lambda)\Phi_1=\mathrm{i}y\,\Phi_1,\qquad
L(x,t;\lambda)\Phi_2=-\mathrm{i}y\,\Phi_2.
\]
Equivalently,
\[
\bigl(L-\mathrm{i}y\bigr)\Phi_1=0,\qquad
\bigl(L+\mathrm{i}y\bigr)\Phi_2=0,
\]
and therefore
\[
\ker\bigl(L\mp \mathrm{i}y\bigr)=\operatorname{span}\{\Phi_{1,2}\}.
\]

Since each eigenvector may be rescaled, we may write
\[
\Phi_1=\begin{pmatrix}\Phi_{11}\\[4pt]\Phi_{21}\end{pmatrix}
=\Phi_{11}\begin{pmatrix}1\\[4pt] r_1\end{pmatrix},
\qquad
\Phi_2=\Phi_{12}\begin{pmatrix}1\\[4pt] r_2\end{pmatrix},
\]
so
\[
\ker\bigl(L\mp \mathrm{i}y\bigr)=\operatorname{span}\!\begin{pmatrix}1\\[4pt] r_{1,2}\end{pmatrix}.
\]
Writing the eigenvalue equation explicitly gives
\[
\begin{pmatrix}
L_{11}\mp \mathrm{i}y & L_{12}\\[4pt]
L_{21} & -L_{11}\mp \mathrm{i}y
\end{pmatrix}
\begin{pmatrix}1\\[4pt] r_{1,2}\end{pmatrix}=0,
\]
from which one obtains the two equivalent expressions for the ratio \(r_{1,2}\):
\begin{equation}\label{eq:r-expressions}
r_{1,2}=\frac{\pm \mathrm{i}y-L_{11}}{L_{12}}
=\frac{L_{21}}{L_{11}\pm \mathrm{i}y}.
\end{equation}

The point \(P = (\lambda,y)\) lies on the hyperelliptic curve \eqref{eq:det-L0}, which is a compact Riemann surface of genus \(g=2\) with punctures \(\infty^{\pm}\). Here \(P\to \infty^\pm \;\Leftrightarrow\; \lambda\to\infty,\; y\to\pm\lambda^3\). 
Denote
\[
\mu=\{\hat{\mu}_1,\hat{\mu}_2\},\qquad \hat{\mu}_j = (\mu_j, \mathrm{i} L_{11}(\mu_j)), \qquad j=1,2 ,
\]
\[
\mu^*=\{\hat{\mu}_1^*,\hat{\mu}_2^*\},\qquad \hat{\mu}_j^* = (\mu_j^*, -\mathrm{i} L_{11}(\mu_j^*)), \qquad j=1,2.
\]
The divisor of \(r_1\) is then given by
\[
\operatorname{Div}\big(r_1(x,t;P)\big)
= D_{\infty^-,\mu^*}(\lambda)-D_{\infty^+,\mu}(\lambda).
\]
That is, \(r_1(x,t;P)\) has zeros at the points of \(\mu^*\) and \(\infty^-\), and poles at the points of \(\mu\) and \(\infty^+\). This description of the divisor is central in constructing Baker--Akhiezer functions on the hyperelliptic Riemann surface defined by
\[
y^2=\prod_{j=1}^3(\lambda^2-\lambda_j^2).
\]

\begin{definition}[Riemann--Theta function]
Let \(R_g\) be a compact Riemann surface of genus \(g\). Fix a canonical homology basis and let \(B\) denote the associated Riemann period matrix, which is symmetric with negative definite real part. The \emph{Riemann--Theta function} is the entire function
\[
\Theta(B;z)=\sum_{n\in\mathbb{Z}^g}
\exp\!\left\{\tfrac{1}{2}\,n^\top B\,n+n^\top z\right\},
\]
where
\[
n=\begin{pmatrix} n_1\\ \vdots \\ n_g\end{pmatrix}, \quad n_j\in\mathbb{Z},\qquad 
z=\begin{pmatrix} z_1\\ \vdots \\ z_g\end{pmatrix}, \quad z_j\in\mathbb{C}.
\]
\end{definition}

\begin{prop}
The Riemann--Theta function satisfies:
\begin{enumerate}
\item Periodicity in the standard lattice directions:
\[
\Theta(z+2\pi i e_k)=\Theta(z), \qquad k=1,\dots,g.
\]
\item Quasi-periodicity in the period matrix directions:
\[
\Theta(z+B e_k)=\Theta(z)\,e^{-z_k-B_{kk}/2}, \qquad k=1,\dots,g.
\]
\item Evenness:
\[
\Theta(-z)=\Theta(z).
\]
\end{enumerate}
\end{prop}

\begin{proof}
(1) For \(z\mapsto z+2\pi i e_k\), the exponential in the definition acquires a factor
\[
\exp\!\bigl(n^\top (2\pi i e_k)\bigr)=\exp(2\pi i n_k)=1,
\]
since \(n_k\in\mathbb{Z}\). Thus \(\Theta\) is periodic in this direction.

(2) For \(z\mapsto z+B e_k\), we compute
\begin{align*}
\Theta(z+B e_k)
&=\sum_{n\in\mathbb{Z}^g}
\exp\!\left\{\tfrac{1}{2}n^\top B n + n^\top (z+B e_k)\right\}\\
&=\sum_{n\in\mathbb{Z}^g}
\exp\!\left\{\tfrac{1}{2}n^\top B n + n^\top z + n^\top B e_k\right\}.
\end{align*}
Write \(m=n+e_k\). Then \(n=m-e_k\), and the exponent becomes
\begin{align*}
&\phantom{=} \tfrac{1}{2}(m-e_k)^\top B(m-e_k)
+ (m-e_k)^\top z + (m-e_k)^\top B e_k \\
& = \tfrac{1}{2}\big(m^\top B m - m^\top B e_k - e_k^\top B m + e_k^\top B e_k\big) + m^\top z - e_k^\top z + m^\top B e_k - e_k^\top B e_k \\
&= \tfrac{1}{2}m^\top B m + m^\top z 
- \tfrac{1}{2}\big(m^\top B e_k + e_k^\top B m\big)
+ m^\top B e_k - \tfrac{1}{2}e_k^\top B e_k - e_k^\top z.
\end{align*}
Since \(B=B^\top\), we have \(m^\top B e_k=e_k^\top B m\); moreover,
\(e_k^\top z=z_k\) and \(e_k^\top B e_k=B_{kk}\). Hence
\[
\tfrac{1}{2}(m-e_k)^\top B(m-e_k)
+ (m-e_k)^\top z + (m-e_k)^\top B e_k
= \tfrac{1}{2}m^\top B m + m^\top z - z_k - \tfrac{1}{2}B_{kk}.
\]
Therefore,
\[
\Theta(z+B e_k)=e^{-z_k - B_{kk}/2}\,\Theta(z).
\]

(3) Finally, since the defining series is invariant under \(n\mapsto -n\), it follows that
\[
\Theta(-z)=\Theta(z).
\]
\end{proof}

\Week{Algebro-Geometric Solutions for the mKdV Equation (3)}{2025-09-28}

% \obj{} 
\obj{Construct algebro–geometric mKdV solutions on a genus-two curve by expressing $r_1$ and u in Riemann theta functions and matching their asymptotics at \(\infty^\pm\), poles and zeroes.}

The Riemann–Roch theorem provides a fundamental relation between the divisor of a meromorphic function and the corresponding Abel map data on a Riemann surface.  
In the present setting, for the meromorphic function \(r_1(x,t;P)\) we have
\begin{equation}\label{div-r1}
\operatorname{Div}\bigl(r_1(x,t;P)\bigr)
= D_{\infty^-,\mu^*}(\lambda)-D_{\infty^+,\mu}(\lambda),
\end{equation}
which identifies the divisor of \(r_1\) as the difference between two effective divisors supported at the points \(\infty^{\pm}\) and at the divisor of spectral data \(\mu,\mu^*\).

A concrete representation of the function \(r_1\) is given in terms of Riemann theta functions:
\begin{equation}\label{def-r1}
r_1(x,t;P)
=\hat{r}_1(x,t)\,
\frac{\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(P)-\alpha_{P_0}(D_{\mu^*})\right)}
     {\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(P)-\alpha_{P_0}(D_\mu)\right)}
\, \exp\!\bigl(\Omega_1(P)-\ln(\omega_1)\bigr),
\quad P=(\lambda,y).
\end{equation}
Here \(\Theta\) denotes the Riemann theta function, \(\mathcal{C}\) is the vector of Riemann constants, \(\mathcal{A}_{P_0}\) is the Abel map based at \(P_0\), and \(\alpha_{P_0}\) is its linear extension to divisors.  
The exponential factor is determined by an Abelian integral \(\Omega_1\), while \(\hat{r}_1(x,t)\) is a normalizing prefactor independent of \(\lambda\).

The underlying algebraic curve \(R_2\) associated with this construction is the genus-two hyperelliptic Riemann surface
\begin{equation}\label{def-R2}
R_2 := \{(\lambda,y)\mid y^2=\prod_{j=1}^3(\lambda^2-\lambda_j^2)\},
\qquad
P\to \infty^\pm \;\Leftrightarrow\; \lambda\to\infty,\; y\to\pm\lambda^3 + O(\lambda).
\end{equation}
Thus each point \(P\in R_2\) is represented by \((\lambda,y)\), where
\[
y=\pm \sqrt{(\lambda^2-\lambda_1^2)(\lambda^2-\lambda_2^2)(\lambda^2-\lambda_3^2)}.
\]

To construct the Jacobi variety one chooses a canonical homology basis \(\{a_1,a_2;b_1,b_2\}\) on \(R_2\), satisfying
\begin{equation}\label{homology}
a_i\circ a_j=0,\quad
b_i\circ b_j=0,\quad
a_i\circ b_j=-\,b_j\circ a_i=\delta_{ij}=
\begin{cases}
1,& i=j,\\
0,& i\neq j,
\end{cases}
\quad i,j=1,2.
\end{equation}

A basis of holomorphic differentials is given by
\begin{equation}\label{diff-forms}
\oint_{a_j} w_i\,d\lambda=2\pi \mathrm{i}\,\delta_{ij},\quad
\oint_{b_j} w_i\,d\lambda=B_{ij},\quad
w_i(\lambda)=\sum_{k=0}^1 d_{ik}\lambda^k y^{-1},\quad d_{ik} \in \mathbb{C},
\end{equation}
where the normalization guarantees the \(a\)-periods form the identity matrix and the \(b\)-periods form the Riemann period matrix \(B=(B_{ij})\).

The Abel map is defined using these normalized differentials. First, introduce the period lattice
\begin{equation}\label{def-L2}
L_2=\{z\in\mathbb{C}^2 \mid z=2\pi \mathrm{i}(n+B m),\; n,m\in\mathbb{Z}^2\},
\end{equation}
where \(B\) is the period matrix and \(n,m,z\) are column vectors in \(\mathbb{C}^2\).  
The Jacobian of \(R_2\) is then
\begin{equation}
J(R_2)=\mathbb{C}^2/L_2.
\end{equation}

For a base point \(P_0\in R_2\), the Abel map sends a point \(P\in R_2\) to
\begin{align}
\mathcal{A}_{P_0}: R_2 &\to J(R_2), \\
P &\mapsto \mathcal{A}_{P_0}(P)
=\bigl(\mathcal{A}_{P_0,1}(P),\mathcal{A}_{P_0,2}(P)\bigr)
=\Bigl(\int_{P_0}^P w_1\,d\lambda,\; \int_{P_0}^P w_2\,d\lambda\Bigr).
\end{align}
This definition extends linearly to divisors.  
A divisor \(D=\sum_{P\in R_2} D(P)\cdot P\) is a finite formal sum of points \(P\) on \(R_2\) with integer coefficients \(D(P)\in\mathbb{Z}\), and we set
\begin{equation}
\alpha_{P_0}(D)=\sum_{P\in R_2} D(P)\,\mathcal{A}_{P_0}(P).
\end{equation}


Finally, we introduce the Abelian integral
\begin{equation}\label{def-Omega1}
\Omega_1(P)=\int_{P_0}^P d\Omega_1,\qquad
\oint_{a_i} d\Omega_1=0,
\end{equation}
with the prescribed asymptotics at the points at infinity,
\begin{equation}\label{Omega1-asymp}
\Omega_1(P)=\pm\Bigl(\ln(\lambda)+\ln(\omega_1)+o(1)\Bigr),\quad P\to \infty^\pm.
\end{equation}
This integral provides the exponential term appearing in \eqref{def-r1} and ensures the correct divisor structure of \(r_1\).


Recall that the function \(r_1(x,t;P)\) admits the representation
\begin{equation}
r_1(x,t;P) = \frac{L_{21}}{\mathrm{i}y + L_{11}},
\end{equation}
where
\begin{equation}
L_{21} = -u\left(\lambda^2 - \frac{\mathrm{i}u_x}{2u}\lambda - \frac{u_{xx}}{4u} + \alpha_1 - \frac{u^2}{2}\right), 
\qquad
L_{11} = -\mathrm{i}\lambda^3 - \mathrm{i}\lambda\left(\alpha_1 - \frac{u^2}{2}\right).
\end{equation}

To analyze its asymptotic behavior, note first that on the hyperelliptic curve \(R_2\),
\begin{equation}
y=\pm\left(\lambda^3-\tfrac{1}{2}(\lambda_1^2+\lambda_2^2+\lambda_3^2)\lambda\right)+O(\lambda^{-1}),
\qquad P\to \infty^\pm.
\end{equation}

As \(P\to \infty^-\), the expansion of the denominator yields
\begin{equation}
L_{11}+\mathrm{i}y=-2 \mathrm{i} \lambda^3 - \mathrm{i}\lambda\left(\alpha_1-\tfrac{u^2}{2}\right)
+\tfrac{\mathrm{i}}{2}(\lambda_1^2+\lambda_2^2+\lambda_3^2)\lambda
+O(\lambda^{-1}),
\end{equation}
so that
\begin{equation}
r_1(x,t;P)=\frac{L_{21}}{\mathrm{i}y+L_{11}}
=-\frac{\mathrm{i}u}{2\lambda}+O(\lambda^{-2}), \qquad P\to \infty^-.
\end{equation}
Similarly, near \(\infty^+\) one obtains
\begin{equation}
r_1(x,t;P)=\frac{\mathrm{i}y-L_{11}}{L_{12}}
=\frac{2\mathrm{i}}{u}\lambda+O(1),\qquad P\to \infty^+.
\end{equation}

On the other hand, from the Riemann theta function representation \eqref{def-r1}, the asymptotics at the two points at infinity are
\begin{equation}\label{eq:r1-asym}
r_1(x,t;P)=
\begin{cases}
\hat{r}_1(x,t)\,
\dfrac{\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^+)-\alpha_{P_0}(D_{\mu^*})\right)}
      {\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^+)-\alpha_{P_0}(D_{\mu})\right)}\,\lambda
+O(1), & P\to \infty^+, \\[2ex]
\hat{r}_1(x,t)\,
\dfrac{\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^-)-\alpha_{P_0}(D_{\mu^*})\right)}
      {\omega_1^2\,\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^-)-\alpha_{P_0}(D_\mu)\right)}\,\lambda^{-1}
+O(\lambda^{-2}), & P\to \infty^-.
\end{cases}
\end{equation}

By comparing these expansions with the direct computation, one obtains the following theta-function formula for the potential \(u(x,t)\):
\begin{equation}\label{eq:u-by-hatR1-theta}
u(x,t) =2\,\mathrm{i}\,
\frac{\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^+)-\alpha_{P_0}(D_\mu)\right)}
     {\hat{r}_1(x,t)\,\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^+)-\alpha_{P_0}(D_{\mu^*})\right)}
=2\,\mathrm{i}\,\hat{r}_1(x,t)\,
\frac{\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^-)-\alpha_{P_0}(D_{\mu^*})\right)}
     {\omega_1^2\,\Theta\!\left(\mathcal{C}+\mathcal{A}_{P_0}(\infty^-)-\alpha_{P_0}(D_\mu)\right)}.
\end{equation}
This identity expresses the potential \(u\) in terms of theta functions associated with the Jacobian of the spectral curve, thereby linking the algebro–geometric spectral data with the physical field variable.


Recall that the column vector 
\[
\Phi_1=\begin{pmatrix}\Phi_{11}\\[4pt]\Phi_{21}\end{pmatrix}=\begin{pmatrix}\Phi_{11}\\[4pt]\Phi_{11}r_1\end{pmatrix}
\]
satisfies both parts of the Lax pair:
\begin{equation}\label{eq:phi_lax}
\begin{split}
\partial_x\begin{pmatrix}\Phi_{11}\\[4pt]\Phi_{11}r_1\end{pmatrix}
=
\begin{pmatrix}
-\mathrm{i}\lambda & u \\[4pt]
-u & \mathrm{i}\lambda
\end{pmatrix}
\begin{pmatrix}\Phi_{11}\\[4pt]\Phi_{11}r_1\end{pmatrix}, \qquad
\partial_t\begin{pmatrix}\Phi_{11}\\[4pt]\Phi_{11}r_1\end{pmatrix}
=
\begin{pmatrix}
V_{11} & V_{12} \\[4pt]
V_{21} & -V_{11}
\end{pmatrix}
\begin{pmatrix}\Phi_{11}\\[4pt]\Phi_{11}r_1\end{pmatrix}.
\end{split}
\end{equation}
Hence the scalar component \(\Phi_{11}\) satisfies
\begin{equation}\label{eq:phi_firstorder}
\Phi_{11,x}=\bigl(-\mathrm{i}\lambda+u\,r_1\bigr)\Phi_{11},\qquad
\Phi_{11,t}=\bigl(V_{11}+V_{12}r_1\bigr)\Phi_{11}.
\end{equation}

Integrating the differential \(1\)-form in \eqref{eq:phi_firstorder} from the base point \((x_0,t_0)\) to \((x,t)\) yields
\begin{equation}\label{eq:phi_integral}
\Phi_{11}(x,t;\lambda)
=\Phi_{11}(x_0,t_0;\lambda)\,
\exp\!\Bigg(\int_{(x_0,t_0)}^{(x,t)}
\Big[(-\mathrm{i}\lambda+u\,r_1)\,dx+(V_{11}+V_{12}r_1)\,dt\Big]\Bigg).
\end{equation}
Hence \(\Phi_{11}\in\mathcal{M}\bigl(R_2\setminus\{\infty^\pm\}\bigr)\), the space of functions meromorphic on \(R_2\) away from \(\infty^\pm\). The representation \eqref{eq:phi_integral} also shows that \(\Phi_{11}\) has an essential singularity at each of \(\infty^\pm\). For convenience, we set the initial point as $(x_0,t_0)=(0,0)$. 

The large-\(\lambda\) asymptotics differ on the two sheets. On the sheet with \(P\to\infty^-\) we fix the normalization
\[
\Phi_{1,0}^- := \lim_{P\to\infty^-}\Phi_{11}(x_0,t_0;\lambda)=1,
\]
and obtain
\begin{align}
\Phi_{11}(x,t;\lambda)&=\exp\bigl(-\mathrm{i}\lambda(x+4\lambda^2 t)\bigr)+O(1),
\qquad P\to\infty^-,\label{eq:phi_asym_minus}\\
\Phi_{21}(x,t;\lambda)=r_1\Phi_{11}&=O(\lambda^{-1})\exp\bigl(-\mathrm{i}\lambda(x+4\lambda^2 t)\bigr),
\qquad P\to\infty^-.
\notag
\end{align}

On the other sheet, \(P\to\infty^+\), the expansion of \(r_1\) is
\begin{equation}\label{eq:r1_expansion}
r_1(x,t;P)=\frac{\mathrm{i}y-L_{11}}{L_{12}}
=\frac{2\,\mathrm{i}}{u}\,\lambda+\frac{u_x}{u^2}+O(\lambda^{-1}),
\qquad P\to\infty^+.
\end{equation}
Consequently
\begin{equation}\label{eq:coeff_expansion}
-\mathrm{i}\lambda+u\,r_1=\mathrm{i}\lambda+(\ln u)_x+O(\lambda^{-1}),
\qquad P\to\infty^+.
\end{equation}
Substituting \eqref{eq:coeff_expansion} into \eqref{eq:phi_integral} gives, with the normalization constant
\begin{equation}\label{eq:phi_normalisation}
\Phi_{1,0}^+:=\lim_{P\to\infty^+}\Phi_{11}(x_0,t_0;\lambda),
\end{equation}
the asymptotic formulas
\begin{align}
\Phi_{11}(x,t;\lambda)&=\Phi_{1,0}^+\,u(x,t)\,\exp\!\bigl(\mathrm{i}\lambda(x+4\lambda^2 t)\bigr)\bigl(1+O(\lambda^{-1})\bigr),
\qquad P\to\infty^+,\label{eq:phi_asym_plus}\\
\Phi_{21}(x,t;\lambda)=r_1\Phi_{11}&=2\,\mathrm{i}\lambda\,\Phi_{1,0}^+\,\exp\!\bigl(\mathrm{i}\lambda(x+4\lambda^2 t)\bigr)\bigl(1+O(\lambda^{-1})\bigr),
\qquad P\to\infty^+.
\notag
\end{align}

Recall
\begin{equation}\label{r1-def-L12}
r_1(\lambda)=\frac{\mathrm{i}\,y(\lambda)-L_{11}(\lambda)}{L_{12}(\lambda)}
=\frac{\mathrm{i}\,y-L_{11}}{u(\lambda-\mu_1)(\lambda-\mu_2)},
\end{equation}
\begin{equation}\label{r1-def-L21}
r_1(\lambda)=\frac{L_{21}(\lambda)}{\mathrm{i}\,y(\lambda)+L_{11}(\lambda)}
=\frac{-u(\lambda-\mu_1^*)(\lambda-\mu_2^*)}{\mathrm{i}\,y+L_{11}}.
\end{equation}
We write
\[
L(\lambda)=
\begin{pmatrix}
L_{11}(\lambda)&L_{12}(\lambda)\\[4pt]
L_{21}(\lambda)&-L_{11}(\lambda)
\end{pmatrix},\qquad
U(\lambda)=
\begin{pmatrix}
-\mathrm{i}\lambda & u\\[4pt]
-u & \mathrm{i}\lambda
\end{pmatrix},
\]
and recall the stationary zero-curvature equation
\begin{equation}\label{L-comm}
L_x=[U,L].
\end{equation}

With \(\sigma_3=\operatorname{diag}(1,-1)\) and off–diagonal part \(L^{\mathrm{off}}=\begin{pmatrix}0&L_{12}\\[4pt]L_{21}&0\end{pmatrix}\), we compute
\begin{align}
[U,L]
&=-2\mathrm{i}\lambda\,\sigma_3 L^{\mathrm{off}}+[Q,L]\nonumber\\[4pt]
&=-2\mathrm{i}\lambda\,\sigma_3 L^{\mathrm{off}}+\begin{pmatrix} 0 & u \\ -u & 0 \end{pmatrix}\begin{pmatrix}
L_{11}&L_{12}\\[4pt]
L_{21}&-L_{11}
\end{pmatrix} -  \begin{pmatrix}
L_{11}&L_{12}\\[4pt]
L_{21}&-L_{11}
\end{pmatrix} \begin{pmatrix} 0 & u \\ -u & 0 \end{pmatrix}\nonumber\\[4pt]
&=-2\mathrm{i}\lambda
\begin{pmatrix}0&L_{12}\\[4pt]-L_{21}&0\end{pmatrix}
+\begin{pmatrix}
u(L_{21}+L_{12}) & -2uL_{11}\\[4pt]
-2uL_{11} & -u(L_{21}+L_{12})
\end{pmatrix}\nonumber\\[6pt]
&=\begin{pmatrix}
u(L_{21}+L_{12}) & -2\mathrm{i}\lambda L_{12}-2uL_{11}\\[6pt]
2\mathrm{i}\lambda L_{21}-2uL_{11} & -u(L_{21}+L_{12})
\end{pmatrix}.
\end{align}
Thus from \eqref{L-comm} we obtain
\begin{equation}\label{L-entries-x}
\begin{split}
\partial_x L_{11}&=u(L_{21}+L_{12}),\\[4pt]
\partial_x L_{12}&=-2\mathrm{i}\lambda\,L_{12}-2uL_{11},\\[4pt]
\partial_x L_{21}&=2\mathrm{i}\lambda\,L_{21}-2uL_{11}.
\end{split}
\end{equation}

By assumption
\begin{equation}\label{L12-factor}
L_{12}(\lambda)=u(\lambda-\mu_1)(\lambda-\mu_2).
\end{equation}
Differentiating with respect to \(x\) gives
\begin{equation}\label{L12-x}
\partial_x L_{12}(\lambda)
=u_x(\lambda-\mu_1)(\lambda-\mu_2)
-u\,\mu_{1,x}(\lambda-\mu_2)
-u\,\mu_{2,x}(\lambda-\mu_1).
\end{equation}

Evaluating \eqref{L12-x} at \(\lambda=\mu_1\) leaves only the middle term:
\begin{equation}
\partial_x L_{12}(\mu_1)=-u\,\mu_{1,x}(\mu_1-\mu_2).
\end{equation}
On the other hand, from \eqref{L-entries-x} evaluated at \(\lambda=\mu_1\) we have
\begin{equation}
\partial_x L_{12}(\mu_1)=-2u\,L_{11}(\mu_1).
\end{equation}
Equating the two expressions yields
\begin{equation}\label{mu1-x}
\mu_{1,x}=\frac{2L_{11}(\mu_1)}{\mu_1-\mu_2}.
\end{equation}

Next, the determinant relation \(y^2=\det L=-L_{11}^2-L_{12}L_{21}\) at \(\lambda=\mu_1\) reduces to
\begin{equation}\label{y-squared}
y(\mu_1)^2=-L_{11}(\mu_1)^2,
\end{equation}
since \(L_{12}(\mu_1)=0\). Hence \(-\mathrm{i}\,y(\mu_1)=\pm L_{11}(\mu_1)\).
We fix the sign by setting
\begin{equation}\label{choice-sign}
-\mathrm{i}\,y(\mu_1)=L_{11}(\mu_1),
\end{equation}
so that \(P=(\mu_1,\mathrm{i}L_{11}(\mu_1))\) represents a pole of the meromorphic function \(r_1(P)=r_1(\lambda)\) on the Riemann surface \(R_2\).  
% This convention determines the branch of \(y(\mu_1)\) and is chosen because zeros and poles of \(\Phi_{11}\) will be identified through the local behaviour of \(r_1\).

Consider now the principal part of \(u\,r_1\) near \(\lambda=\mu_1\). From \eqref{r1-def-L12} and \eqref{L12-factor},
\begin{equation}\label{ur1-pole}
u\,r_1(\lambda)
=\frac{\mathrm{i}y(\lambda)-L_{11}(\lambda)}{(\lambda-\mu_1)(\lambda-\mu_2)}
=\frac{\mathrm{i}y(\mu_1)-L_{11}(\mu_1)}{(\mu_1-\mu_2)}\cdot\frac{1}{\lambda-\mu_1}+O(1),
\qquad \lambda\to\mu_1.
\end{equation}
Using \eqref{choice-sign} and \eqref{mu1-x},
\[
\mathrm{i}y(\mu_1)-L_{11}(\mu_1)=-2L_{11}(\mu_1)=-(\mu_1-\mu_2)\,\mu_{1,x}.
\]
Thus \eqref{ur1-pole} becomes
\begin{equation}\label{ur1-final}
u\,r_1(\lambda)
=-\frac{\mu_{1,x}}{\lambda-\mu_1}+O(1)
=\partial_x\!\bigl(\ln(\lambda-\mu_1)\bigr)+O(1),
\qquad \lambda\to\mu_1.
\end{equation}
Then by \eqref{eq:phi_integral} and \eqref{ur1-final} we conclude that \(\lambda=\mu_1\) is a simple zero of \(\Phi_{11}(x,t;\lambda)\).

{u(\lambda-\mu_1)(\lambda-\mu_2)}.
\end{equation}
Expanding the numerator and evaluating at \(\lambda=\mu_1\) yields the simple pole
\begin{equation}\label{v12r1-pole-res}
V_{12}(\lambda)\,r_1(\lambda)
=\frac{-2L_{11}(\mu_1)\,V_{12}(\mu_1)}{u(\mu_1-\mu_2)}\cdot\frac{1}{\lambda-\mu_1}
+O(1),
\qquad \lambda\to\mu_1.
\end{equation}

Using the evolution law \eqref{mu1-t} for \(\mu_1\) this principal part becomes
\begin{equation}\label{v12r1-final}
V_{12}(\lambda)\,r_1(\lambda)
=-\frac{\mu_{1,t}}{\lambda-\mu_1}+O(1)
=\partial_t\!\bigl(\ln(\lambda-\mu_1)\bigr)+O(1),
\qquad \lambda\to\mu_1.
\end{equation}

\end{equation}

\end{aligned}

\textbf{Abel’s theorem.}
Let $M(R_2)$ denote the meromorphic functions on $R_2$, and $M'(R_2)$ the meromorphic $1$-forms.  
For $f\in M(R_2)\setminus\{0\}$ and $P\in R_2$, the order $\operatorname{ord}_P(f)$ is the integer $k$ such that, in a local coordinate $z$ with $z(P)=0$,
\[
f(z)=z^{\,k}g(z),\qquad g(0)\neq0.
\]
The divisor of $f$ is
\[
(f)=\sum_{P\in R_2}\operatorname{ord}_P(f)\,P,
\]
which is a finite sum because $f$ has only finitely many zeros and poles.  
The divisor $(w)$ of a meromorphic $1$-form $w\in M'(R_2)\setminus\{0\}$ is defined analogously by using its local expression $w(z)=z^{\,k}h(z)\,\dd z$ with $h(0)\neq0$.

A meromorphic function is holomorphic if and only if $(f)\ge0$.  
A meromorphic $1$-form is holomorphic (of the first kind) if and only if $(w)\ge0$.  
A divisor of the form $(f)$ is called principal; a divisor of the form $(w)$ is canonical.

\begin{thm}[Abel’s theorem]
A divisor $D$ is principal if and only if $\deg D=0$ and $\alpha_{P_0}(D)=0$.
\end{thm}

\medskip

\textbf{Case 1.} \(\lambda_1,\lambda_2,\lambda_3\in \mathrm{i}\mathbb{R}\).

Let \(\tau\) denote complex conjugation on the curve,
\[
\tau\!:\;(\lambda,y)\mapsto(\lambda^*,y^*).
\]
The action of \(\tau\) on the homology basis may be chosen (up to orientation) as
\begin{equation}\label{eq:tau-action}
\tau(a_1)=-a_2,\qquad \tau(a_2)=-a_1,\qquad
\tau(b_1)=b_2,\qquad \tau(b_2)=b_1.
\end{equation}

With the choice of cycles described above, the \(a\)-periods reduce to integrals along paths on the first sheet:
\begin{equation}\label{eq:a1-periods}
\oint_{a_1}\frac{d\lambda}{y}
=2\int_{\lambda_2}^{\lambda_1}\frac{d\lambda}{y},\qquad
\oint_{a_1}\frac{\lambda\,d\lambda}{y}
=2\int_{\lambda_2}^{\lambda_1}\frac{\lambda\,d\lambda}{y},
\end{equation}
and
\begin{equation}\label{eq:a2-periods}
\oint_{a_2}\frac{d\lambda}{y}
=2\int_{\lambda_1^*}^{\lambda_2^*}\frac{d\lambda}{y},\qquad
\oint_{a_2}\frac{\lambda\,d\lambda}{y}
=2\int_{\lambda_1^*}^{\lambda_2^*}\frac{\lambda\,d\lambda}{y}.
\end{equation}

Recall the definition
\begin{equation}\label{eq:Lambda-subst}
y(\lambda)=\sqrt{(\lambda^2-\lambda_1^2)(\lambda^2-\lambda_2^2)(\lambda^2-\lambda_3^2)},
\qquad \Lambda=\lambda^2,\qquad d\Lambda=2\lambda\,d\lambda.
\end{equation}
Applying the substitution \(\Lambda=\lambda^2\) transforms the basic integral into a rational function of \(\Lambda\):
\begin{equation}\label{eq:basic-int-lambda}
\int \frac{d\lambda}{y(\lambda)}
= \int \frac{d\lambda}{\sqrt{\prod_{i=1}^3(\lambda^2-\lambda_i^2)}}
= \int \frac{d\Lambda}{2\sqrt{\Lambda\prod_{i=1}^3(\Lambda-\lambda_i^2)}}.
\end{equation}
Thus the problem reduces to integrals of the general form
\begin{equation}\label{eq:general-elliptic}
\int \frac{d\Lambda}{\sqrt{\prod_{i=1}^4(\Lambda-\beta_i)}},
\qquad \beta_1<\beta_2<\beta_3<\beta_4.
\end{equation}

To reduce \eqref{eq:general-elliptic} to the Legendre normal form we use the change of variable
\begin{equation}\label{eq:z-change}
z^2 \;=\; \frac{\beta_3-\beta_1}{\beta_4-\beta_1}\,\frac{\Lambda-\beta_4}{\Lambda-\beta_3}.
\end{equation}
A direct computation yields the standard reduction
\begin{equation}\label{eq:elliptic-reduction}
\int \frac{d\Lambda}{\sqrt{\prod_{i=1}^4(\Lambda-\beta_i)}}
\;=\;
\frac{2}{\sqrt{(\beta_3-\beta_1)(\beta_4-\beta_2)}}
\int \frac{dz}{\sqrt{(1-z^2)(1-k^2 z^2)}},
\end{equation}
with the elliptic modulus given by
\begin{equation}\label{eq:k-general}
k^2 \;=\; \frac{(\beta_3-\beta_2)(\beta_4-\beta_1)}{(\beta_3-\beta_1)(\beta_4-\beta_2)}.
\end{equation}

For reference, the incomplete elliptic integrals in Jacobi form are
\[
F(z,k)=\int_0^z\frac{dt}{\sqrt{(1-t^2)(1-k^2 t^2)}},\qquad
E(z,k)=\int_0^z\sqrt{\frac{1-k^2 t^2}{1-t^2}}\,dt,\qquad 0<k<1,
\]
and the elliptic integral of the third kind is
\[
\Pi(z,\alpha^2,k)=\int_0^z\frac{dt}{(1-\alpha^2 t^2)\sqrt{(1-t^2)(1-k^2 t^2)}}.
\]
These standard forms allow the evaluation of the \(a\)-period integrals \eqref{eq:a1-periods}--\eqref{eq:a2-periods} once the correspondence between the branch points \(\{\beta_i\}\) and the original \(\lambda\)-branch points is fixed.

\medskip

Applying \eqref{eq:elliptic-reduction}--\eqref{eq:k-general} to our concrete problem:
we may take
\[
\beta_1=\lambda_3^2,\qquad \beta_2=\lambda_2^2,\qquad \beta_3=\lambda_1^2,\qquad \beta_4=0,
\]
and (using \eqref{eq:z-change}) obtain
\begin{equation}\label{eq:z-explicit}
z^2 \;=\; \frac{\lambda_1^2-\lambda_3^2}{-\lambda_3^2}\,\frac{\Lambda}{\Lambda-\lambda_1^2}
\;=\; \frac{(\lambda_1^2-\lambda_3^2)\,\lambda^2}{\lambda_3^2(\lambda_1^2-\lambda^2)}.
\end{equation}
For these choices the modulus becomes
\begin{equation}\label{eq:k-specific}
k_1^{(1)\,2}
\;=\;
\frac{(\lambda_1^2-\lambda_2^2)(0-\lambda_3^2)}{(\lambda_1^2-\lambda_3^2)(0-\lambda_2^2)}
\;=\;
\frac{\lambda_3^2}{\lambda_2^2}\,\frac{\lambda_1^2-\lambda_2^2}{\lambda_1^2-\lambda_3^2},
\end{equation}
so that
\begin{equation}\label{eq:int-lambda-to-F}
\int \frac{d\lambda}{y(\lambda)}
\;=\;
\frac{1}{\sqrt{-\lambda_2^2(\lambda_1^2-\lambda_3^2)}}
\int \frac{dz}{\sqrt{(1-z^2)(1-k_1^{(1)\,2} z^2)}} ,
\qquad k_1^{(1)} \;=\; \frac{\lambda_3}{\lambda_2}\, \sqrt{\frac{\lambda_1^2-\lambda_2^2}{\lambda_1^2-\lambda_3^2}}
\end{equation}

Defining the incomplete elliptic integral
\begin{equation}\label{eq:nu-def}
\nu \;=\; F\bigl(z,k_1^{(1)}\bigr)
\;=\; \int_0^z \frac{dt}{\sqrt{(1-t^2)(1-k_1^{(1)\,2}t^2)}},
\qquad z=\operatorname{sn}\bigl(\nu,k_1^{(1)}\bigr),
\end{equation}
we obtain the standard elliptic parametrisation of \(\lambda\) via \eqref{eq:z-explicit} and \eqref{eq:nu-def}.

\medskip

Recall the relations:
\[
\mathrm{sn}^2(u,k)+\mathrm{cn}^2(u,k)=1,\qquad
\mathrm{dn}^2(u,k)+k^2\mathrm{sn}^2(u,k)=1,
\]
and the complete elliptic integrals
\[
K(k)=F(1,k),\qquad K'(k)=F(1,k'),\qquad k'^2=1-k^2.
\]

\medskip

From \eqref{eq:z-explicit} one checks the mapping of special \(\Lambda\)-values to \(z\)-values:
\[
\Lambda=\lambda^2=0\; \leftrightarrow\;  z=0 \; \leftrightarrow\;  \nu=0,\qquad
\Lambda=\lambda_1^2\; \leftrightarrow\;  z^2\to\infty \; \leftrightarrow\;  \nu=\pm \mathrm{i}K_1^{(1)\prime} \,
\]
and
\[
\Lambda=\lambda^2=\lambda_2^2\; \leftrightarrow\;  z^2 = \frac{\lambda_2^2}{\lambda_3^2}\,\frac{\lambda_1^2-\lambda_3^2}{\lambda_1^2-\lambda_2^2} = \frac{1}{k_1^{(1)\,2}} \; \leftrightarrow\; \nu = \pm (K_1^{(1)}+ \mathrm{i}K_1^{(1)\prime} ) .
\]

\medskip


With this choice, the \(a_1\)-period can be expressed in the \(\nu\)-variable as an integral between two points in the complex \(\nu\)-plane. In particular, we have
\begin{equation}\label{eq:a1-period-dlambda_over_y}
2\int_{\lambda_2}^{\lambda_1} \frac{d\lambda}{y(\lambda)}
\;=\;
\frac{2}{\sqrt{-\lambda_2^2(\lambda_1^2-\lambda_3^2)}}
\int_{K_1^{(1)} + \mathrm{i}K_1^{(1)\prime}}^{\mathrm{i}K_1^{(1)\prime}} \dd \nu
\;=\;
\frac{-2\,K_1^{(1)}}{\sqrt{-\lambda_2^2(\lambda_1^2-\lambda_3^2)}}
\;=\;
\frac{-2\mathrm{i}\,K_1^{(1)}}{\lambda_2 \sqrt{\lambda_1^2-\lambda_3^2}} \in \mathbb{R}^- .
\end{equation}

The symmetry under complex conjugation, \(\tau:(\lambda,y)\mapsto (\lambda^*,y^*)\), implies the relations
\begin{equation}\label{eq:a2-conj-rel}
\left(\oint_{a_2} \frac{d\lambda}{y}\right)^* = \oint_{a_2} \tau^* \frac{d\lambda}{y}
= \oint_{\tau(a_2)} \frac{d\lambda}{y} = \oint_{-a_1} \frac{d\lambda}{y} 
= -\oint_{a_1} \frac{d\lambda}{y} \in \mathbb{R}^+.
\end{equation}

Similarly, one finds
\begin{equation}\label{eq:a2-lambda-dlambda}
\oint_{a_2} \frac{\lambda\,d\lambda}{y} = \oint_{a_1} \frac{\lambda\,d\lambda}{y}.
\end{equation}

Using the substitution \(\Lambda = \lambda^2\), one rewrites
\begin{equation}\label{eq:lambda-dlambda-to-Lambda}
\int \frac{\lambda\,d\lambda}{y(\lambda)} = \int \frac{d\Lambda}{2\sqrt{\prod_{i=1}^3 (\Lambda - \lambda_i^2)}}.
\end{equation}
Introducing the change of variables
\begin{equation}\label{eq:z-change-2}
z^2 = \frac{\beta_1 - \beta_3}{\Lambda - \beta_3},
\end{equation}
the integral becomes
\begin{equation}\label{eq:int-lambda-dlambda-y-to-F}
\int \frac{\lambda\,d\lambda}{y(\lambda)} 
= \frac{1}{-\mathrm{i}\, \sqrt{\lambda_1^2 - \lambda_3^2}}
\int \frac{dz}{\sqrt{(1-z^2)(1-k_2^{(1)2} z^2)}}, 
\qquad k_2^{(1)} = \sqrt{\frac{\lambda_1^2 - \lambda_2^2}{\lambda_1^2 - \lambda_3^2}}.
\end{equation}
Hence,
\begin{equation}\label{eq:a1-period-lambda_over_y-result}
\oint_{a_1} \frac{\lambda\,d\lambda}{y} = \frac{-2\mathrm{i}\,K_2^{(1)}}{\sqrt{\lambda_1^2 - \lambda_3^2}} 
= \oint_{a_2} \frac{\lambda\,d\lambda}{y}.
\end{equation}

Since
\begin{equation}
\begin{pmatrix}
\displaystyle \oint_{a_1} \frac{d\lambda}{y} & \displaystyle \oint_{a_2} \frac{d\lambda}{y} \\[2ex]
\displaystyle \oint_{a_1} \frac{\lambda\,d\lambda}{y} & \displaystyle \oint_{a_2} \frac{\lambda\,d\lambda}{y}
\end{pmatrix}
=
\begin{pmatrix}
\displaystyle \oint_{a_1} \frac{d\lambda}{y} & 0 \\[2ex]
0 & \displaystyle \oint_{a_1} \frac{\lambda\,d\lambda}{y}
\end{pmatrix}
\begin{pmatrix}
1 & -1 \\ 
1 & 1
\end{pmatrix},
\end{equation}
the relation \eqref{eq:D-via-Ainv} becomes
\begin{equation*}
\begin{pmatrix} d_{10} & d_{11} \\[2pt] d_{20} & d_{21} \end{pmatrix} 
= \pi \mathrm{i} \,
\begin{pmatrix}
1 & 1 \\ 
-1 & 1
\end{pmatrix}
\begin{pmatrix}
\displaystyle \oint_{a_1} \frac{d\lambda}{y} & 0 \\[2pt]
0 & \displaystyle \oint_{a_1} \frac{\lambda\,d\lambda}{y}
\end{pmatrix}^{-1}.
\end{equation*}
This yields the coefficients
\begin{equation}\label{eq:d-coeffs-final}
\begin{aligned}
d_{10} &= -\frac{\pi \lambda_2 \sqrt{\lambda_1^2 - \lambda_3^2}}{2\, K_1^{(1)}}, &
\qquad d_{11} &= -\frac{\pi \sqrt{\lambda_1^2 - \lambda_3^2}}{2\, K_2^{(1)}},\\[2mm]
d_{20} &= -\,d_{10}, &
\qquad d_{21} &= d_{11}.
\end{aligned}
\end{equation}

\extra{}

Define
\begin{equation}\label{eq:F-def}
F(z,k)=\int_0^z \frac{dt}{f(t,k)}, \qquad
f(t,k)=\sqrt{(1-t^2)(1-k^2 t^2)}, \qquad 0<k<1,
\end{equation}
where the principal branch of the square root is chosen and we fix $f(0,k)=1$.

Set
\begin{equation}\label{eq:K-def}
K(k)=F(1,k), \qquad K'(k)=K(k'), \qquad k'=\sqrt{1-k^2}.
\end{equation}

We take the segments $[-k^{-1},-1]\cup[1,k^{-1}]$ as branch cuts for $f(t,k)$ on the extended real axis. Starting at $t=0$ and moving right along the real axis, each time one crosses a branch point an extra phase factor $e^{-\pi\mathrm{i}/2}=-\mathrm{i}$ is accumulated; moving left from $0$ produces the conjugate factor $e^{\pi\mathrm{i}/2}=\mathrm{i}$. With these conventions, $f(t,k)$ on the first sheet admits the explicit real-axis description
\begin{equation}\label{eq:f-real}
f(t,k)=
\begin{cases}
\sqrt{(1-t^2)(1-k^2 t^2)}\in\mathbb{R}^+,
& t\in(-1,1),\\[0.3em]
-\mathrm{i}\sqrt{(t^2-1)(1-k^2 t^2)}\in\mathrm{i}\mathbb{R}^-, 
& t\in(1,k^{-1}),\\[0.3em]
\mathrm{i}\sqrt{(t^2-1)(1-k^2 t^2)}\in\mathrm{i}\mathbb{R}^+,
& t\in(-k^{-1},-1),\\[0.3em]
-\sqrt{(t^2-1)(k^2 t^2-1)}\in\mathbb{R}^-,
& t\in(-\infty,-k^{-1})\cup(k^{-1},\infty).
\end{cases}
\end{equation}

We now evaluate several definite integrals that will be used below.

For the interval $(k^{-1},\infty)$ perform the substitution
\begin{equation}\label{eq:t-sub-1}
t=k^{-1}s^{-1}, \qquad dt=-k^{-1}s^{-2}\,ds ,
\end{equation}
which maps $t\in[k^{-1},\infty)$ to $s\in(0,1]$. Direct computation gives
\begin{equation}\label{eq:int-infty}
\begin{aligned}
\int_{k^{-1}}^{\infty}\frac{dt}{\sqrt{(t^2-1)(k^2 t^2-1)}}
&=\int_1^0
\frac{-k^{-1}s^{-2}\,ds}
{\sqrt{(k^{-2} s^{-2}-1)(s^{-2}-1)}}\\
&=\int_0^1\frac{ds}{\sqrt{(1-s^2)(1-k^2 s^2)}}\\
&=F(1,k)=K(k).
\end{aligned}
\end{equation}

For the finite interval $(1,k^{-1})$ use the substitution
\begin{equation}\label{eq:t-sub-2}
t=\bigl(1-(k')^2 s^2\bigr)^{-1/2}, \qquad
dt=(k')^2 s\bigl(1-(k')^2 s^2\bigr)^{-3/2}ds,
\end{equation}
which sends $s\in[0,1]$ to $t\in[1,k^{-1}]$. Under this change of variables
\begin{equation}\label{eq:sqrt-expand}
\begin{aligned}
\sqrt{(t^2-1)(1-k^2 t^2)}
&=\sqrt{\frac{(k's)^2}{1-(k's)^2}\cdot (k')^2\frac{1-s^2}{1-(k's)^2}}\\
&=(k')^2 s\,\frac{\sqrt{1-s^2}}{1-(k's)^2}.
\end{aligned}
\end{equation}
Hence
\begin{equation}\label{eq:int-1-k}
\begin{aligned}
\int_1^{k^{-1}}
\frac{dt}{\sqrt{(t^2-1)(1-k^2 t^2)}}
&=\int_0^1\frac{ds}{\sqrt{(1-s^2)\bigl(1-(k')^2 s^2\bigr)}}\\
&=F(1,k')=K(k')=K'(k).
\end{aligned}
\end{equation}

Combining the evaluations on $[0,1]$ and $(1,k^{-1})$, and recalling the phase $-\mathrm{i}$ on $(1,k^{-1})$ from \eqref{eq:f-real}, we obtain
\begin{equation}\label{eq:F-kinv}
\begin{aligned}
F(k^{-1},k)
&=\int_0^1\frac{dt}{\sqrt{(1-t^2)(1-k^2 t^2)}}
+\mathrm{i}\int_1^{k^{-1}}
\frac{dt}{\sqrt{(t^2-1)(1-k^2 t^2)}}\\
&=K(k)+\mathrm{i}K'(k).
\end{aligned}
\end{equation}

Taking the limit $z\to+\infty$ along the positive real axis (passing through $k^{-1}$) yields
\begin{equation}\label{eq:F-infty}
F(+\infty,k)=K(k)+\mathrm{i}K'(k)-K(k)=\mathrm{i}K'(k),
\end{equation}
while symmetry across the origin gives
\begin{equation}\label{eq:F-minus1}
F(-1,k)=-F(1,k)=-K(k).
\end{equation}

Similarly, evaluating $F$ at $-k^{-1}$ (follow the path from $0$ to $-1$ to $-k^{-1}$ and use the phase convention on $(-k^{-1},-1)$ in \eqref{eq:f-real}) yields
\begin{equation}\label{eq:F-kminus}
\begin{aligned}
F(-k^{-1},k)
&=-K(k)+\int_{-1}^{-k^{-1}}
\frac{dt}{\mathrm{i}\sqrt{(t^2-1)(1-k^2 t^2)}}\\
&=-K(k)+\mathrm{i}\int_{1}^{k^{-1}}
\frac{dt}{\sqrt{(t^2-1)(1-k^2 t^2)}}\\
&=-K(k)+\mathrm{i}K'(k).
\end{aligned}
\end{equation}
Finally, taking $z\to-\infty$ along the negative real axis (passing through $-k^{-1}$) gives
\begin{equation}\label{eq:F-minus-infty}
F(-\infty,k)=-K(k)+\mathrm{i}K'(k)+K(k)=\mathrm{i}K'(k).
\end{equation}


Since
\begin{equation}\label{eq:type1} \tag{type~1}
\int \frac{\lambda^{2n}}{y(\lambda)}\,d\lambda
=\int \frac{\Lambda^{n}\,d\Lambda}
{2\bigl(\Lambda(\Lambda-\lambda_1^{2})(\Lambda-\lambda_2^{2})(\Lambda-\lambda_3^{2})\bigr)^{1/2}},
\qquad \Lambda=\lambda^2,
\end{equation}
and
\begin{equation}\label{eq:type2} \tag{type~2}
\int \frac{\lambda^{2n+1}}{y(\lambda)}\,d\lambda
=\int \frac{\Lambda^{n}\,d\Lambda}
{2\bigl((\Lambda-\lambda_1^{2})(\Lambda-\lambda_2^{2})(\Lambda-\lambda_3^{2})\bigr)^{1/2}},
\qquad \Lambda=\lambda^2,
\end{equation}
we treat the two types of integrals separately.  

\medskip

\textbf{Computation of Type~1 Integrals (\(\mathcal{Y}_{a_i}^{[0]}, \mathcal{Y}_{b_i}^{[0]}\)). }
Introduce a change of variable from $\Lambda$ to $z$ by
\begin{equation}\label{eq:z-def}
z^{2}=\frac{\beta_{3}-\beta_{1}}{\beta_{4}-\beta_{1}}\,
\frac{\Lambda-\beta_{4}}{\Lambda-\beta_{3}},
\qquad \beta_{1}<\beta_{2}<\beta_{3}<\beta_{4}.
\end{equation}
Solving \eqref{eq:z-def} for $\Lambda$ yields
\begin{equation}\label{eq:Lambda-z}
\Lambda
=\frac{z^{2}\beta_{3}(\beta_{4}-\beta_{1})-(\beta_{3}-\beta_{1})\beta_{4}}
{z^{2}(\beta_{4}-\beta_{1})-(\beta_{3}-\beta_{1})}
=\beta_{3}+\frac{(\beta_{3}-\beta_{1})(\beta_{3}-\beta_{4})}
{(\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})}.
\end{equation}
Differentiation gives
\begin{equation}\label{eq:dLambda}
d\Lambda
=\frac{2(\beta_{3}-\beta_{1})(\beta_{4}-\beta_{3})(\beta_{4}-\beta_{1})\,z}
{\bigl((\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})\bigr)^{2}}\,dz.
\end{equation}

The shifted factors $\Lambda-\beta_{i}$ become rational functions of $z$.  In particular
\begin{align}
\Lambda-\beta_{1}
&=\frac{(\beta_{3}-\beta_{1})(\beta_{4}-\beta_{1})(z^{2}-1)}
{(\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})},\label{eq:Lambda-beta1}\\[4pt]
\Lambda-\beta_{2}
&=\frac{(\beta_{3}-\beta_{2})(\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})(\beta_{4}-\beta_{2})}
{(\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})},\label{eq:Lambda-beta2}\\[4pt]
\Lambda-\beta_{4}
&=\frac{(\beta_{3}-\beta_{4})(\beta_{4}-\beta_{1})z^{2}}
{(\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})}.\label{eq:Lambda-beta4}
\end{align}

Hence the full quartic factor factors as
\begin{equation}\label{eq:prod-Lambda-beta}
\begin{aligned}
\prod_{i=1}^{4}(\Lambda-\beta_{i})
&=\frac{(\beta_{3}-\beta_{1})^{2}(\beta_{4}-\beta_{1})^{2}(\beta_{4}-\beta_{3})^{2}\;z^{2}\;(z^{2}-1)}
{\bigl((\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})\bigr)^{4}}\\
&\qquad\times\Bigl((\beta_{3}-\beta_{2})(\beta_{4}-\beta_{1}) z^{2}
-(\beta_{3}-\beta_{1})(\beta_{4}-\beta_{2})\Bigr).
\end{aligned}
\end{equation}

Define the modulus
\begin{equation}\label{eq:k1-def}
k_{1}^{2}=\frac{(\beta_{3}-\beta_{2})(\beta_{4}-\beta_{1})}
{(\beta_{3}-\beta_{1})(\beta_{4}-\beta_{2})}.
\end{equation}
Taking a principal square root of \eqref{eq:prod-Lambda-beta} (and grouping algebraic constants) yields
\begin{equation}\label{eq:sqrt-prod}
\Bigl(\prod_{i=1}^{4}(\Lambda-\beta_{i})\Bigr)^{1/2}
=\frac{(\beta_{3}-\beta_{1})(\beta_{4}-\beta_{1})(\beta_{4}-\beta_{3})}
{\bigl((\beta_{4}-\beta_{1})z^{2}-(\beta_{3}-\beta_{1})\bigr)^{2}}\;
\sqrt{(\beta_{3}-\beta_{1})(\beta_{4}-\beta_{2})}\; z\;\sqrt{(1-z^{2})(1-k_{1}^{2} z^{2})}.
\end{equation}

Combining \eqref{eq:dLambda} and \eqref{eq:sqrt-prod} produces the standard elliptic reduction
\begin{equation}\label{eq:integral-reduction}
\int\frac{d\Lambda}{\sqrt{\prod_{i=1}^{4}(\Lambda-\beta_{i})}}
=\frac{2}{\sqrt{(\beta_{4}-\beta_{2})(\beta_{3}-\beta_{1})}}\,
\int\frac{dz}{\sqrt{(1-z^{2})(1-k_{1}^{2} z^{2})}},
\end{equation}
valid on the appropriate choice of branches for the square roots.

Specializing the parameters by
\begin{equation}\label{eq:beta-lambda}
\beta_{1}=\lambda_{3}^{2},\qquad \beta_{2}=\lambda_{2}^{2},
\qquad \beta_{3}=\lambda_{1}^{2},\qquad \beta_{4}=0,
\end{equation}
the change of variable \eqref{eq:z-def} becomes
\begin{equation}\label{eq:z-special}
z^{2}=\frac{\lambda_{1}^{2}-\lambda_{3}^{2}}{-\lambda_{3}^{2}}\,
\frac{\lambda^{2}}{\lambda^{2}-\lambda_{1}^{2}},
\qquad
z=\sqrt{\frac{\lambda_{1}^{2}-\lambda_{3}^{2}}{-\lambda_{3}^{2}}}\;
\frac{\lambda}{\sqrt{\lambda^{2}-\lambda_{1}^{2}}},
\end{equation}
where the positive square root is taken for the principal branch.  The modulus \eqref{eq:k1-def} simplifies to
\begin{equation}\label{eq:k1-lambda}
k_{1}^{2}
=\frac{(\lambda_{1}^{2}-\lambda_{2}^{2})(\lambda_{2}^{2}-\lambda_{3}^{2})}
{(\lambda_{1}^{2}-\lambda_{3}^{2})(0-\lambda_{2}^{2})}
=\frac{(\lambda_{1}^{2}-\lambda_{2}^{2})\,\lambda_{3}^{2}}
{(\lambda_{1}^{2}-\lambda_{3}^{2})\,\lambda_{2}^{2}}.
\end{equation}

Choose the cut $[-\lambda_1,\lambda_1]$ for the function $\sqrt{\lambda^2-\lambda_1^2}$ and fix the sheet by requiring
\[
\sqrt{\lambda^2-\lambda_1^2}\sim\lambda\qquad(\lambda\to\infty).
\]
For $\operatorname{Re}\lambda\ge0$ and \(\lambda \in \mathrm{i}\mathbb{R}\) write $\lambda=\mathrm{i}t$ with $t\in\mathbb{R}$. If $\lambda_j=\mathrm{i}t_j$ then
\begin{equation}\label{eq:sqrt-imag-axis}
\sqrt{\lambda^2-\lambda_1^2}=
\begin{cases}
\mathrm{i}\sqrt{t^2-t_1^2}, & t>t_1,\\[4pt]
\sqrt{t_1^2-t^2}, & -t_1<t<t_1,\\[4pt]
-\mathrm{i}\sqrt{t^2-t_1^2}, & t<-t_1.
\end{cases}
\end{equation}

Hence for \(\lambda \in \mathrm{i}\mathbb{R}\) on the imaginary axis with $\operatorname{Re}\lambda\ge0$,
\begin{equation}\label{eq:z-values}
z=
\begin{cases}
\displaystyle \sqrt{\frac{\lambda_1^2-\lambda_3^2}{-\lambda_3^2}}\;\frac{\lambda}{\sqrt{\lambda^2-\lambda_1^2}}\in\mathrm{i}\mathbb{R}, 
& \lambda\in(-\lambda_1,\lambda_1),\\[12pt]
\displaystyle \sqrt{\frac{\lambda_1^2-\lambda_3^2}{-\lambda_3^2}}\;\frac{|\lambda|}{\sqrt{\lambda_1^2-\lambda^2}}\in\mathbb{R}^{+}, 
& \lambda\in(-\mathrm{i}\infty,-\lambda_1)\cup(\lambda_1,+\mathrm{i}\infty).
\end{cases}
\end{equation}

The oriented real/imaginary segments map under $\lambda\mapsto z$ as follows (the images are indicated in the $z$-plane):
\begin{equation}\label{eq:segment-mapping}
\begin{aligned}
(\lambda_3,\lambda_2)&\longrightarrow(1,k_1^{-1}), &\qquad (\lambda_2,\lambda_1)&\longrightarrow(k_1^{-1},+\infty),\\[4pt]

\begin{equation}\label{eq:y-factorised}
\begin{aligned}
y(\lambda)
&=(\lambda-\lambda_3)\sqrt{\frac{\lambda-\lambda_2}{\lambda-\lambda_3}}
   (\lambda-\lambda_1)\sqrt{\frac{\lambda+\lambda_1}{\lambda-\lambda_1}}
   (\lambda+\lambda_2)\sqrt{\frac{\lambda+\lambda_3}{\lambda+\lambda_2}}\\[4pt]

   \sqrt{\frac{t-t_2}{t-t_3}}
   \sqrt{\frac{t+t_1}{t-t_1}}
   \sqrt{\frac{t+t_3}{t+t_2}}.
\end{aligned}
\end{equation}

Here the square root is taken as the principal branch:
\begin{equation}\label{eq:sqrt-branch}
\sqrt{z}=\sqrt{|z|}\,{\mathrm e}^{\tfrac{1}{2}\mathrm{i}\arg z},\qquad \arg z\in(-\pi,\pi].
\end{equation}

To determine the permutation of real/imaginary values and the signs of \(y(\mathrm{i}t)\) as \(t\) varies along the imaginary axis, it is convenient to count
\begin{itemize}
  \item the sign \(s=\operatorname{sign}\bigl((t-t_3)(t-t_1)(t+t_2)\bigr)\) of the real linear prefactor, and
  \item the number \(k\in\{0,1,2,3\}\) of the three real ratios
    \[
    \frac{t-t_2}{t-t_3},\qquad
    \frac{t+t_1}{t-t_1},\qquad
    \frac{t+t_3}{t+t_2}
    \]
    that are negative (each negative ratio contributes a factor \(\mathrm{i}\) from the principal square root).
\end{itemize}
With these conventions, from \eqref{eq:y-factorised} one has
\[
y(\mathrm{i}t)=(-\mathrm{i})\;s\;(\mathrm{i})^{\,k}\;\bigl(\text{positive real}\bigr)
= -s\,\mathrm{i}^{\,k+1}\,\bigl(\text{positive real}\bigr),
\]
so that \(y(\mathrm{i}t)\) is imaginary when \(k\equiv 0,2\pmod 2\) and real when \(k\equiv 1,3\pmod 2\), with the sign determined by the combination above.

Evaluating \(s\) and \(k\) on the natural partition of the real axis by \(\pm t_1,\pm t_2,\pm t_3\) gives the following classification:

\begin{equation}\label{eq:yt-signs}

\begin{cases}
\mathrm{i}\,\mathbb{R}^{-}, & t>t_3,\\[4pt]
\mathbb{R}^{-},          & t_2<t<t_3,\\[4pt]
\mathrm{i}\,\mathbb{R}^{+}, & t_1<t<t_2,\\[4pt]
\mathbb{R}^{+},          & -t_1<t<t_1,\\[4pt]
\mathrm{i}\,\mathbb{R}^{-}, & -t_2<t<-t_1,\\[4pt]
\mathbb{R}^{-},          & -t_3<t<-t_2,\\[4pt]
\mathrm{i}\,\mathbb{R}^{+}, & t<-t_3.
\end{cases}
\end{equation}

For clarity, a short justification for each interval is as follows (sign of linear prefactor \(s\) and number \(k\) of negative ratios are given in parentheses):

\begin{itemize}
  \item \(t>t_3\): \(s=+1,\ k=0\) \(\Rightarrow\) \(y=-\mathrm{i}\cdot(+1)\cdot 1\cdot(\text{pos})\in \mathrm{i}\mathbb{R}^{-}.\)
  \item \(t_2<t<t_3\): \(s=-1,\ k=1\) \(\Rightarrow\) \(y=(-1)\cdot(\text{pos})\in\mathbb{R}^{-}.\)
  \item \(t_1<t<t_2\): \(s=-1,\ k=0\) \(\Rightarrow\) \(y=+ \mathrm{i}\cdot(\text{pos})\in\mathrm{i}\mathbb{R}^{+}.\)
  \item \(-t_1<t<t_1\): \(s=+1,\ k=1\) \(\Rightarrow\) \(y=+(\text{pos})\in\mathbb{R}^{+}.\)
  \item \(-t_2<t<-t_1\): \(s=+1,\ k=0\) \(\Rightarrow\) \(y=-\mathrm{i}\cdot(\text{pos})\in\mathrm{i}\mathbb{R}^{-}.\)
  \item \(-t_3<t<-t_2\): \(s=-1,\ k=1\) \(\Rightarrow\) \(y=(-1)\cdot(\text{pos})\in\mathbb{R}^{-}.\)
  \item \(t<-t_3\): \(s=-1,\ k=0\) \(\Rightarrow\) \(y=+\mathrm{i}\cdot(\text{pos})\in\mathrm{i}\mathbb{R}^{+}.\)
\end{itemize}

All statements above are obtained without algebraically combining the fractional square-root factors; each square root is treated as the principal branch \eqref{eq:sqrt-branch} applied to the linear fractional arguments appearing in \eqref{eq:y-factorised}.

\begin{equation}\label{eq:y-expansion}
y=\lambda^{3}\Bigl(1+\hat{v}_{1}\lambda^{-2}+\hat{v}_{2}\lambda^{-4}+\cdots\Bigr),
\qquad
y^{-1}=\lambda^{-3}\Bigl(1-\hat{v}_{1}\lambda^{-2}
        +(\hat{v}_{1}^{2}-\hat{v}_{2})\lambda^{-4}+\cdots\Bigr).
\end{equation}
Decompose $y$ as
\begin{equation}\label{eq:y-split}
y=y^{[n_{+}]}+y^{[n_{-}]}
=\lambda^{3}\Bigl(1+\hat{v}_{1}\lambda^{-2}+\cdots+\hat{v}_{n}\lambda^{-2n}\Bigr)
  +y^{[n_{-}]},
\end{equation}
where $y^{[n_{-}]}=\hat{v}_{n+1}\lambda^{-2(n+1)+3}+\cdots$.  
Multiplying \eqref{eq:y-split} by $y^{-1}$ and using \eqref{eq:y-expansion} gives
\begin{align}
y^{[n_{+}]}y^{-1}
&=(y-y^{[n_{-}]})y^{-1}
 =1-y^{[n_{-}]}y^{-1} \\
&=1-\bigl(\hat{v}_{n+1}\lambda^{-2(n+1)}+\cdots\bigr)
      \lambda^{3}\cdot\lambda^{-3}
      \Bigl(1-\hat{v}_{1}\lambda^{-2}+\cdots\Bigr) \\
&=1-\hat{v}_{n+1}\lambda^{-2(n+1)}+\cdots.
\end{align}

In particular,
\begin{equation}\label{eq:lambda3y-1}
\lambda^{3}y^{-1}=1-\hat{v}_{1}\lambda^{-2}+\cdots,
\end{equation}
and
\begin{equation}\label{eq:lambda3y-1-next}
\lambda^{3}\bigl(1+\hat{v}_{1}\lambda^{-2}\bigr)y^{-1}
=1-\hat{v}_{2}\lambda^{-4}+\cdots,
\end{equation}
with analogous identities holding for higher-order terms.

Since
\begin{equation}\label{eq:Omega2-asymp}
\Omega_{2}(P)=\pm\bigl(\lambda+\omega_{2}+o(1)\bigr),
\qquad p\to\infty^{\pm},
\end{equation}
differentiation shows that
\begin{equation}\label{eq:dOmega2-asymp}
d\Omega_{2}(P)=\pm\bigl(1+O(\lambda^{-2})\bigr)\,d\lambda.
\end{equation}
Thus $d\Omega_{2}$ can be written in the form
\begin{equation*}
d\Omega_{2}(P)=\frac{1}{y}\bigl(\lambda^{3}+c_{21}\lambda+c_{20}\bigr)\,d\lambda.
\end{equation*}

The period normalization for the holomorphic differentials
\[

\]
is expressed by
\[

=\oint_{a_j} w_i\,d\lambda
=\sum_{k=0}^1 d_{ik}\,\mathcal{Y}_{a_j}^{[k]},
\qquad
B_{ij}
=\oint_{b_j} w_i\,d\lambda
=\sum_{k=0}^1 d_{ik}\,\mathcal{Y}_{b_j}^{[k]} .
\]
The symmetry properties of the period integrals,
\[
\mathcal{Y}_{a_1}^{[0]}=-\mathcal{Y}_{a_2}^{[0]},\qquad
\mathcal{Y}_{b_1}^{[0]}=-\mathcal{Y}_{b_2}^{[0]},\qquad
\mathcal{Y}_{a_1}^{[1]}=\mathcal{Y}_{a_2}^{[1]},\qquad
\mathcal{Y}_{b_1}^{[1]}=\mathcal{Y}_{b_2}^{[1]},
\]
allow the normalization conditions to be written as
\begin{equation}\label{eq:dY-rel}
\begin{pmatrix}
d_{10} & d_{11} \\
d_{20} & d_{21}
\end{pmatrix}
\begin{pmatrix}
\mathcal{Y}_{a_1}^{[0]} & 0 \\
0 & \mathcal{Y}_{a_2}^{[1]}
\end{pmatrix}
\begin{pmatrix}
1 & -1 \\
1 & 1
\end{pmatrix}
=2\pi\mathrm{i}\,\mathbb{I}_2 ,
\end{equation}
and
\begin{equation}\label{eq:dy-rel}
\begin{pmatrix}
d_{10} & d_{11} \\
d_{20} & d_{21}
\end{pmatrix}
\begin{pmatrix}
\mathcal{Y}_{b_1}^{[0]} & 0 \\
0 & \mathcal{Y}_{b_2}^{[1]}
\end{pmatrix}
\begin{pmatrix}
1 & -1 \\
1 & 1
\end{pmatrix}
= B^{(1)} .
\end{equation}

Eliminating the coefficients \( d_{ik} \) between \eqref{eq:dY-rel} and \eqref{eq:dy-rel} gives
\begin{equation}\label{eq:B1-def}
\begin{aligned}
B^{(1)}
&=\pi\mathrm{i}
\begin{pmatrix}
1 & 1 \\
-1 & 1
\end{pmatrix}
\begin{pmatrix}
\left(\mathcal{Y}_{a_1}^{[0]}\right)^{-1}\mathcal{Y}_{b_1}^{[0]} & 0 \\
0 & \left(\mathcal{Y}_{a_2}^{[1]}\right)^{-1}\mathcal{Y}_{b_2}^{[1]}

\begin{pmatrix}
1 & -1 \\
1 & 1
\end{pmatrix} \\
&=\pi\mathrm{i}
\begin{pmatrix}
1 & 1 \\
-1 & 1
\end{pmatrix}
\begin{pmatrix}
\tau_1^{(1)} & 0 \\
0 & \tau_2^{(1)}
\end{pmatrix}
\begin{pmatrix}
1 & -1 \\
1 & 1
\end{pmatrix} \\

\begin{pmatrix}
\tau_1^{(1)} & \tau_2^{(1)} \\
-\tau_1^{(1)} & \tau_2^{(1)}
\end{pmatrix}
\begin{pmatrix}
1 & -1 \\
1 & 1
\end{pmatrix} 
=\pi\mathrm{i}

\end{pmatrix}.
\end{aligned}
\end{equation}

0 & \tau_2^{(1)}
\end{pmatrix}

1 & -1 \\
1 & 1
\end{pmatrix}
n

\right\},
\end{aligned}
\end{equation}
where

-1 & 1

-1 & 1

\end{equation}

\end{equation}

\textbf{Computation of Type~1 Integrals (\(\mathcal{Y}_{a_i}^{[2]}, \mathcal{Y}_{b_i}^{[2]}\)). }

We introduce the auxiliary function \(\Lambda\) (cf.\ \eqref{eq:Lambda-z}) by
\begin{equation}\label{eq:Lambda-def-z-rev}
\Lambda(z)
=\frac{\lambda_1^2 z^2}{\,z^2-\dfrac{\lambda_1^2-\lambda_3^2}{-\lambda_3^2}\,}
=\frac{\lambda_1^2\lambda_3^2 z^2}{\lambda_3^2 z^2+\bigl(\lambda_1^2-\lambda_3^2\bigr)}
=\frac{\lambda_1^2\lambda_3^2 z^2}{\lambda_1^2-\lambda_3^2\bigl(1-z^2\bigr)} .
\end{equation}

Recall the relation between \(z\) and the elliptic argument \(\nu\),
\begin{equation}\label{eq:nu-z-rev}
\nu = F\!\bigl(z,k_1^{(1)}\bigr),\qquad
z=\operatorname{sn}\!\bigl(\nu,k_1^{(1)}\bigr).
\end{equation}
We will use the imaginary-period shift identities for Jacobi elliptic functions,
\begin{equation}\label{eq:Jacobi-shift-rev}
\operatorname{sn}\!\bigl(\nu+\mathrm{i}K_1',k_1^{(1)}\bigr)
=\frac{1}{k_1^{(1)}\,\operatorname{sn}(\nu,k_1^{(1)})},\qquad
\operatorname{cn}\!\bigl(\nu+\mathrm{i}K_1',k_1^{(1)}\bigr)
=-\mathrm{i}\,\frac{\operatorname{dn}(\nu,k_1^{(1)})}{k_1^{(1)}\,\operatorname{sn}(\nu,k_1^{(1)})}.
\end{equation}

Set
\begin{equation}\label{eq:nu-shift-rev}
\nu=\nu_1+\mathrm{i}K_1',
\end{equation}
so that \(\operatorname{sn}(\nu,k_1^{(1)})\) and \(\operatorname{cn}(\nu,k_1^{(1)})\)
in \eqref{eq:Lambda-def-z-rev} may be expressed in terms of \(\nu_1\).
Substituting \(\,z=\operatorname{sn}(\nu,k_1^{(1)})\,\) into \eqref{eq:Lambda-def-z-rev}
and using \eqref{eq:Jacobi-shift-rev} yields
\begin{align}
\Lambda
=\frac{\lambda_1^2\lambda_3^2\,\operatorname{sn}^2\!\bigl(\nu,k_1^{(1)}\bigr)}
{\lambda_1^2-\lambda_3^2\,\operatorname{cn}^2\!\bigl(\nu,k_1^{(1)}\bigr)} 
&=\frac{\lambda_1^2\lambda_3^2}
{\lambda_1^2\bigl(k_1^{(1)}\bigr)^2\operatorname{sn}^2\!\bigl(\nu_1,k_1^{(1)}\bigr)
+\lambda_3^2\operatorname{dn}^2\!\bigl(\nu_1,k_1^{(1)}\bigr)} \nonumber\\
&=\frac{\lambda_1^2\lambda_3^2}
{\lambda_3^2+\bigl(k_1^{(1)}\bigr)^2\bigl(\lambda_1^2-\lambda_3^2\bigr)
\operatorname{sn}^2\!\bigl(\nu_1,k_1^{(1)}\bigr)} .
\label{eq:Lambda-final-rev}
\end{align}

Using the identity
\[
\bigl(k_1^{(1)}\bigr)^2
=\frac{(\lambda_1^2-\lambda_2^2)\lambda_3^2}{(\lambda_1^2-\lambda_3^2)\lambda_2^2},
\]
the expression \eqref{eq:Lambda-final-rev} can be rewritten in the equivalent forms
\begin{equation}\label{eq:Lambda-alt-rev}
\Lambda
=\frac{\lambda_1^2\lambda_3^2}
{\lambda_3^2+\dfrac{\lambda_3^2}{\lambda_2^2}\bigl(\lambda_1^2-\lambda_2^2\bigr)
\operatorname{sn}^2\!\bigl(\nu_1,k_1^{(1)}\bigr)}
=\frac{\lambda_1^2}
{1-\dfrac{\lambda_1^2-\lambda_2^2}{-\lambda_2^2}
\operatorname{sn}^2\!\bigl(\nu_1,k_1^{(1)}\bigr)} .
\end{equation}
The last form highlights that \(\Lambda\) becomes a simple rational function
of \(\operatorname{sn}^2(\nu_1,k_1^{(1)})\), which is convenient for integrating.

We now compute \(\mathcal{Y}_{a_1}^{[2]}\). Starting from the integral in \eqref{eq:type1}
and changing variables to the elliptic parameter \(\nu\) (with the shift \(\nu=\nu_1+\mathrm{i}K_1'\))
one obtains
\begin{equation}\label{eq:Ya1-change}
\mathcal{Y}_{a_1}^{[2]}
=2\int_{\lambda_2}^{\lambda_1}\frac{\lambda^2\,d\lambda}{y(\lambda)}
=\frac{2\lambda_1^2}{\sqrt{-\lambda_2^2(\lambda_1^2-\lambda_3^2)}}
\int_{K_1+\mathrm{i}K_1'}^{\mathrm{i}K_1'}
\frac{d\nu}{1-\dfrac{\lambda_1^2-\lambda_2^2}{-\lambda_2^2}
\operatorname{sn}^2\!\bigl(\nu-\mathrm{i}K_1',k_1^{(1)}\bigr)} .
\end{equation}
The integrand is an elementary function of \(\operatorname{sn}^2\), hence the integral
is an incomplete elliptic integral of the third kind. Writing the integral in terms of
\(\Pi(\alpha^2,u,k)\), with characteristic \(\alpha_1^2=\dfrac{\lambda_1^2-\lambda_2^2}{-\lambda_2^2}\),
modulus \(k=k_1^{(1)}\), and upper limit \(u\), we have
\begin{equation}\label{eq:Ya1-incomplete}
\mathcal{Y}_{a_1}^{[2]}
=\frac{2\lambda_1^2}{-\mathrm{i}\lambda_2\sqrt{\lambda_1^2-\lambda_3^2}}\,
\left.\Pi\!\biggl(\alpha_1^2,\,u,\,k_1^{(1)}\biggr)\right|_{u=K_1}^{u=0},
\qquad \alpha_1^2=\frac{\lambda_1^2-\lambda_2^2}{-\lambda_2^2}.
\end{equation}
Evaluating the difference of the incomplete elliptic integrals between \(u=K_1\) and \(u=0\)
gives the complete elliptic integral of the third kind \(\Pi(\alpha_1^2,k)\). Therefore
\begin{equation}\label{eq:Ya1-final}
\mathcal{Y}_{a_1}^{[2]}
=\frac{-2\mathrm{i}\lambda_1^2}{\lambda_2\sqrt{\lambda_1^2-\lambda_3^2}}\,
\Pi\!\bigl(\alpha_1^2,\,k_1^{(1)}\bigr),
\qquad \alpha_1^2=\frac{\lambda_1^2-\lambda_2^2}{-\lambda_2^2}.
\end{equation}

By symmetry of the contour (or by the corresponding sign change in the integration limits),
one obtains the companion relation
\begin{equation}\label{eq:Ya2-final}
\mathcal{Y}_{a_2}^{[2]}=-\mathcal{Y}_{a_1}^{[2]}.
\end{equation}


Similarly,

\operatorname{sn}(z+K,k)=\frac{\operatorname{cn}(z,k)}{\operatorname{dn}(z,k)},\qquad
\operatorname{sn}(\mathrm{i}z,k)=\mathrm{i}\,

\begin{equation}\label{eq:Jacobi-identities-2}

\textbf{Computation of Type~2 Integrals (\(\mathcal{Y}_{a_i}^{[3]}, \mathcal{Y}_{b_i}^{[3]}\)).}

For integrals of type~2, we use the parametrization
\begin{equation}\label{eq:Lambda-type2}
\Lambda
=\lambda_1^2+\frac{\lambda_3^2-\lambda_1^2}{z^2},
\qquad
z=\operatorname{sn}\!\bigl(\nu,k_2^{(1)}\bigr),
\qquad
\bigl(k_2^{(1)}\bigr)^2
=\frac{\lambda_1^2-\lambda_2^2}{\lambda_1^2-\lambda_3^2}.
\end{equation}
Writing \(\nu=\nu_1+\mathrm{i}K^{\prime}\), this expression admits several equivalent forms. Indeed,
\[
\Lambda
=\lambda_1^2
+\bigl(\lambda_3^2-\lambda_1^2\bigr)\bigl(k_2^{(1)}\bigr)^2
\operatorname{sn}^2\!\bigl(\nu_1,k_2^{(1)}\bigr)
=\lambda_1^2
+\bigl(\lambda_3^2-\lambda_1^2\bigr)
\bigl(1-\operatorname{dn}^2(\nu_1,k_2^{(1)})\bigr),
\]
and hence
\[
\Lambda
=\lambda_3^2
+\bigl(\lambda_1^2-\lambda_3^2\bigr)
\operatorname{dn}^2\!\bigl(\nu_1,k_2^{(1)}\bigr).
\]

Using this representation, we compute
\begin{equation}\label{eq:Y-a1-3}
\begin{aligned}
\mathcal{Y}_{a_1}^{[3]}
&=2\int_{\lambda_2}^{\lambda_1}\frac{\lambda^{3}\,d\lambda}{y(\lambda)} 
=\frac{2\mathrm{i}}{\sqrt{\lambda_1^{2}-\lambda_3^{2}}}
\int_{K_2+\mathrm{i}K_2'}^{\mathrm{i}K_2'}
\Bigl(
\lambda_3^{2}
+\bigl(\lambda_1^{2}-\lambda_3^{2}\bigr)
\operatorname{dn}^{2}\!\bigl(\nu-\mathrm{i}K_2',k_2^{(1)}\bigr)
\Bigr)\,d\nu .
\end{aligned}
\end{equation}
The integral separates into a constant term and a term involving \(\operatorname{dn}^2\). Evaluating each contribution yields
\[
\mathcal{Y}_{a_1}^{[3]}
=-\frac{2\mathrm{i}\lambda_3^{2}K_2^{(1)}}{\sqrt{\lambda_1^{2}-\lambda_3^{2}}}
+\left.
\frac{2\mathrm{i}\bigl(\lambda_1^{2}-\lambda_3^{2}\bigr)}
{\sqrt{\lambda_1^{2}-\lambda_3^{2}}}
E(u,k_2^{(1)})\right|_{u=K_2}^{u=0}=-\frac{2\mathrm{i}\bigl(
\lambda_3^{2}K_2^{(1)}
+\bigl(\lambda_1^{2}-\lambda_3^{2}\bigr)E_2^{(1)}
\bigr)}
{\sqrt{\lambda_1^{2}-\lambda_3^{2}}}.
\]

\begin{equation}\label{eq:Y-a2-3}
\mathcal{Y}_{a_2}^{[3]}=\mathcal{Y}_{a_1}^{[3]}.
\end{equation}

\begin{equation}\label{eq:Y-b2-3-start}
\begin{aligned}

&=2\int_{-\lambda_2}^{-\lambda_3}\frac{\lambda^{3}\,d\lambda}{y(\lambda)}

\operatorname{dn}^{2}\!\bigl(\nu-\mathrm{i}K_2',k_2^{(1)}\bigr)
\Bigr)\,d\nu ,
\end{aligned}
\end{equation}
we employ the standard shift identities
\begin{equation}\label{eq:dn-cn-shift}
\operatorname{dn}(z-K,k)=\frac{k'}{\operatorname{dn}(z,k)},
\qquad
\operatorname{cn}(z-K',k')
=\frac{k\,\operatorname{sn}(z,k')}{\operatorname{dn}(z,k')}.
\end{equation}
From these formulas one obtains
\begin{equation}\label{eq:dn-transform}
\begin{aligned}
\operatorname{dn}\!\left(\mathrm{i}u-K_2-\mathrm{i}K_2',k_2^{(1)}\right)
&=\frac{k_2^{(1)\prime}}
{\operatorname{dn}\!\left(\mathrm{i}(u-K_2'),k_2^{(1)}\right)} \\
&=k_2^{(1)\prime}
\frac{\operatorname{cn}(u-K_2',k_2^{(1)\prime})}
{\operatorname{dn}(u-K_2',k_2^{(1)\prime})}
=k_2^{(1)\prime}\operatorname{sn}(u,k_2^{(1)\prime}).
\end{aligned}
\end{equation}

Setting \(\nu=\mathrm{i}u-K_2\), the integral becomes
\begin{equation}\label{eq:Y-b2-3}
\begin{aligned}
\mathcal{Y}_{b_2}^{[3]}
&=\frac{2\mathrm{i}}{\sqrt{\lambda_1^{2}-\lambda_3^{2}}}
\int_{K_2'}^{0}
\Bigl(
\lambda_3^{2}
+\bigl(\lambda_1^{2}-\lambda_3^{2}\bigr)
\bigl(k_2^{(1)\prime}\bigr)^{2}
\operatorname{sn}^{2}(u,k_2^{(1)\prime})
\Bigr)\,\mathrm{i}\,du \\
&=\frac{2}{\sqrt{\lambda_1^{2}-\lambda_3^{2}}}
\int_{0}^{K_2'}
\Bigl(
\lambda_1^{2}
-\bigl(\lambda_1^{2}-\lambda_3^{2}\bigr)
\operatorname{dn}^{2}(u,k_2^{(1)\prime})
\Bigr)\,du 
=\frac{
2\lambda_1^{2}K_2'
-2\bigl(\lambda_1^{2}-\lambda_3^{2}\bigr)E_2'
}
{\sqrt{\lambda_1^{2}-\lambda_3^{2}}}.
\end{aligned}
\end{equation}
Again by symmetry of the \(b\)-cycles,
\begin{equation}\label{eq:Y-b1-3}
\mathcal{Y}_{b_1}^{[3]}=\mathcal{Y}_{b_2}^{[3]}.
\end{equation}

where we have fixed \((x_0,t_0)=(0,0)\) and used the relation
\[
\widehat{V}=v\,\widehat{U},
\qquad
v=2(\lambda_1^2+\lambda_2^2+\lambda_3^2).
\]

Combining these identities with the definition of \(D\) yields the following shifted forms:
\begin{equation}\label{eq:D-Delta-shift1}
D+\Delta
=\mathcal{C}
+\mathcal{A}_{P_0}(\infty^+)
-\alpha_{P_0}(D_{\mu_0}),
\end{equation}
\begin{equation}\label{eq:D-Delta-shift2}
D+\Delta+\mathrm{i}\widehat{U}(x+v t)
=\mathcal{C}
+\mathcal{A}_{P_0}(\infty^+)
-\alpha_{P_0}(D_\mu),
\end{equation}
\begin{equation}\label{eq:D-shift}
D+\mathrm{i}\widehat{U}(x+v t)
=\mathcal{C}
+\mathcal{A}_{P_0}(\infty^-)
-\alpha_{P_0}(D_\mu)
=\mathcal{C}
+\mathcal{A}_{P_0}(\infty^+)
-\alpha_{P_0}(D_{\mu^*}),
\end{equation}
and
\begin{equation}\label{eq:D-minus-Delta}
D-\Delta+\mathrm{i}\widehat{U}(x+v t)
=\mathcal{C}
+\mathcal{A}_{P_0}(\infty^-)
-\alpha_{P_0}(D_{\mu^*}).
\end{equation}

We now introduce the constant
\begin{equation}\label{eq:Cu0-def}
C_{u_0}
=\frac{\Theta(D)}
{\Phi_{1,0}^{+}\,\Theta(D+\Delta)} .
\end{equation}
With this notation, the representation \eqref{eq:u-by-phi10+-theta-mu} can be rewritten as
\begin{equation}\label{eq:u-theta-ratio-1}
\begin{aligned}
u(x,t)
&=\frac{\Theta(D)\,
\Theta\!\bigl(D+\Delta+\mathrm{i}\widehat{U}(x+v t)\bigr)}
{\Phi_{1,0}^{+}\,
\Theta(D+\Delta)\,
\Theta\!\bigl(D+\mathrm{i}\widehat{U}(x+v t)\bigr)}
\,\mathrm{e}^{2\mathrm{i}\omega_2 x+2\mathrm{i}\omega_3 t} \\
&=C_{u_0}\,
\frac{\Theta\!\bigl(D+\Delta+\mathrm{i}\widehat{U}(x+v t)\bigr)}
{\Theta\!\bigl(D+\mathrm{i}\widehat{U}(x+v t)\bigr)}
\,\mathrm{e}^{2\mathrm{i}\omega_2 x+2\mathrm{i}\omega_3 t}.
\end{aligned}
\end{equation}
Equivalently, starting from \eqref{eq:u-by-phi10+-theta-mu*}, one obtains
\begin{equation}\label{eq:u-theta-ratio-2}
u(x,t)
=\frac{-4\,\Theta\!\bigl(D-\Delta+\mathrm{i}\widehat{U}(x+v t)\bigr)}
{\omega_1^{2}\,C_{u_0}\,
\Theta\!\bigl(D+\mathrm{i}\widehat{U}(x+v t)\bigr)}
\,\mathrm{e}^{-2\mathrm{i}\omega_2 x-2\mathrm{i}\omega_3 t}.
\end{equation}



\textbf{For Case~1.}

Since
\begin{equation}\label{eq:U-def}
\widehat{U}=(U_1,U_2)=(2d_{11},2d_{21}),
\qquad
d_{21}=d_{11}
=-\frac{\pi\sqrt{\lambda_1^2-\lambda_3^2}}{2K_2^{(1)}}\in\mathbb{R}^{-},
\end{equation}
it follows that \(U_1=U_2\). Consequently,
\begin{equation}\label{eq:omega2-zero}
\omega_2
=-\frac{\mathrm{i}}{4\pi}\sum_{j=1}^2 U_j\,\mathcal{Y}_{a_j}^{[2]}
=-\frac{\mathrm{i}}{4\pi}
\bigl(U_1\mathcal{Y}_{a_1}^{[2]}+U_2\mathcal{Y}_{a_2}^{[2]}\bigr)
=0,
\end{equation}
where we have used \(\mathcal{Y}_{a_1}^{[2]}=-\mathcal{Y}_{a_2}^{[2]}\). Hence
\begin{equation}\label{eq:omega3-zero}
\omega_3=v\,\omega_2=0.
\end{equation}

Moreover,
\begin{equation}\label{eq:Delta1-def}
\Delta=\Delta^{(1)}
=\binom{\pi\mathrm{i}+\nu^{(1)}}{\pi\mathrm{i}-\nu^{(1)}},
\qquad
\nu^{(1)}
=\frac{\pi\mathrm{i}}{K_1^{(1)}}\,
F\!\left(\frac{\lambda_2}{\lambda_3},k_1^{(1)}\right)
\in \mathrm{i}\mathbb{R}.
\end{equation}
From this explicit form one readily finds
\begin{equation}\label{eq:Delta-sym}
\Delta^*=-\Delta,
\qquad
\sigma_1\Delta
=\binom{\pi\mathrm{i}-\nu^{(1)}}{\pi\mathrm{i}+\nu^{(1)}}
=\binom{2\pi\mathrm{i}}{2\pi\mathrm{i}}-\Delta,
\qquad
\sigma_1=
\begin{pmatrix}
0&1\\[2pt]
1&0
\end{pmatrix}.
\end{equation}

Using \eqref{eq:B1-def} and \eqref{eq:c1c2-def}, the period matrix satisfies
\begin{equation}\label{eq:sigma-B-sym}
\sigma_1 B^{(1)} \sigma_1=B^{(1)},
\qquad
\sigma_1^{\top}=\sigma_1,
\qquad
\sigma_1^{2}=\mathbb{I}.
\end{equation}
Letting \(m=\sigma_1 n\), we compute
\begin{equation}\label{eq:theta-sigma}
\begin{aligned}
\Theta\!\left(B^{(1)},\sigma_1 z\right)
&=\sum_{n\in\mathbb{Z}^2}
\exp\!\left\{\tfrac12 n^{\top}B^{(1)}n+n^{\top}\sigma_1 z\right\} \\
&=\sum_{n\in\mathbb{Z}^2}
\exp\!\left\{\tfrac12(\sigma_1 n)^{\top}\sigma_1 B^{(1)}\sigma_1 n
+(\sigma_1 n)^{\top} z\right\} \\
&=\sum_{m\in\mathbb{Z}^2}
\exp\!\left\{\tfrac12 m^{\top}B^{(1)}m+m^{\top} z\right\}
=\Theta\!\left(B^{(1)},z\right).
\end{aligned}
\end{equation}

Using the identities
\[
\Theta^*(z)=\Theta(z^*),
\qquad
\Theta(\sigma_1 z)=\Theta(z),
\qquad
\Theta(-z)=\Theta(z),
\]
and the representation
\begin{equation}\label{eq:u-theta-ratio-1-case1}
u(x,t)
=C_{u_0}\,
\frac{\Theta\!\bigl(D+\Delta+\mathrm{i}\widehat{U}(x+vt)\bigr)}
{\Theta\!\bigl(D+\mathrm{i}\widehat{U}(x+vt)\bigr)},
\end{equation}
we obtain
\begin{equation}\label{eq:u-conj}
\begin{aligned}
u^*(x,t)
&=C_{u_0}^*\,
\frac{\Theta\!\bigl(D^*+\Delta^*-\mathrm{i}\widehat{U}(x+vt)\bigr)}
{\Theta\!\bigl(D^*-\mathrm{i}\widehat{U}(x+vt)\bigr)} \\
&=C_{u_0}^*\,
\frac{\Theta\!\bigl(\sigma_1 D^*+\sigma_1\Delta^*-\mathrm{i}\widehat{U}(x+vt)\bigr)}
{\Theta\!\bigl(\sigma_1 D^*-\mathrm{i}\widehat{U}(x+vt)\bigr)} \\
&=C_{u_0}^*\,
\frac{\Theta\!\bigl(-\sigma_1 D^*-\Delta+\mathrm{i}\widehat{U}(x+vt)\bigr)}
{\Theta\!\bigl(-\sigma_1 D^*+\mathrm{i}\widehat{U}(x+vt)\bigr)} .
\end{aligned}
\end{equation}


Requiring \(u(x,t)\in\mathbb{R}\) and comparing \eqref{eq:u-theta-ratio-2} with
\eqref{eq:u-conj}, we obtain the constraint
\begin{equation}\label{eq:D-reality}
D=-\sigma_1 D^*+2\pi\mathrm{i}\,n+B^{(1)}m,
\qquad
D=\binom{D_1}{D_2},
\qquad
n,m\in\mathbb{Z}^2 .
\end{equation}
Equivalently, this condition can be written componentwise as
\begin{equation}\label{eq:D-components}
\begin{aligned}
D_1+D_2^*
&=2\pi\mathrm{i}n_1+B_{11}m_1+B_{12}m_2,\\
D_1^*+D_2
&=2\pi\mathrm{i}n_2+B_{21}m_1+B_{22}m_2 .
\end{aligned}
\end{equation}

Using the symmetry properties of the period matrix,
\begin{equation}\label{eq:B-sym-real}
B_{11}=B_{22},
\qquad
B_{21}=B_{12},
\qquad
B_{ij}\in\mathbb{R},
\end{equation}
and noting that \(B_{11}\neq B_{12}\), we conclude that
\begin{equation}\label{eq:constraints}
n_1=-n_2,
\qquad
m_1=m_2 .
\end{equation}
Substituting these relations back into \eqref{eq:D-components} yields
\begin{equation}\label{eq:D-sum}
D_1+D_2^*
=2\pi\mathrm{i}\,n_1+\bigl(B_{11}+B_{12}\bigr)m_1 .
\end{equation}

On the other hand, rewriting \(u(x,t)\) in \eqref{eq:u-theta-ratio-2} in the form
\begin{equation}\label{eq:u-sigma1-form}
u(x,t)
=\frac{-4\,\Theta\!\bigl(\sigma_1 D-\sigma_1\Delta+\mathrm{i}\widehat{U}(x+vt)\bigr)}
{\omega_1^{2}C_{u_0}\,\Theta\!\bigl(D+\mathrm{i}\widehat{U}(x+vt)\bigr)}
=\frac{-4\,\Theta\!\bigl(\sigma_1 D+\Delta+\mathrm{i}\widehat{U}(x+vt)\bigr)}
{\omega_1^{2}C_{u_0}\,\Theta\!\bigl(D+\mathrm{i}\widehat{U}(x+vt)\bigr)},
\end{equation}
and comparing with \eqref{eq:u-theta-ratio-1-case1}, we impose the condition
\begin{equation}\label{eq:D-sigma1}
D=\sigma_1 D+2\pi\mathrm{i}\,n',
\qquad
n'\in\mathbb{Z}^2 .
\end{equation}
This is equivalent to the pair of relations
\begin{equation}\label{eq:D-diff}
D_1-D_2=2\pi\mathrm{i}\,n_1',
\qquad
D_2-D_1=2\pi\mathrm{i}\,n_2',
\end{equation}
and hence
\begin{equation}\label{eq:nprime}
n_2'=-n_1',
\qquad
D_1-D_2=2\pi\mathrm{i}\,n_1'.
\end{equation}

Under the condition \eqref{eq:D-sigma1}, comparison of \eqref{eq:u-theta-ratio-2} with
\eqref{eq:u-theta-ratio-1-case1} yields
\begin{equation}\label{eq:Cu0-square-long}
\begin{aligned}
C_{u_0}^2
&=\frac{-4\,\Theta\!\left(D-\Delta+\mathrm{i}\widehat{U}(x+vt)\right)}
{\omega_1^2\,\Theta\!\left(D+\Delta+\mathrm{i}\widehat{U}(x+vt)\right)} 
=\frac{-4\,\Theta\!\left(\sigma_1 D+\Delta+\mathrm{i}\widehat{U}(x+vt)\right)}
{\omega_1^2\,\Theta\!\left(D+\Delta+\mathrm{i}\widehat{U}(x+vt)\right)}
=\frac{-4}{\omega_1^2},
\end{aligned}
\end{equation}
where the \(\sigma_1\)-symmetry of the theta function has been used.
Consequently,
\begin{equation}\label{eq:Cu0-final}
C_{u_0}=\frac{2\mathrm{i}}{\omega_1}.
\end{equation}

Combining \eqref{eq:D-sum} with \eqref{eq:nprime}, we obtain the constraints
\begin{equation}\label{eq:D-final-constraints}
D_1-D_2=2\pi\mathrm{i}\,n_1',
\qquad
D_1+D_2^*
=2\pi\mathrm{i}\,n_1+\bigl(B_{11}+B_{12}\bigr)m_1 .
\end{equation}
These imply
\begin{equation}\label{eq:D-real-imag}
\operatorname{Re} D_1=\operatorname{Re} D_2
=\frac{1}{2}m_1\bigl(B_{11}+B_{12}\bigr),
\qquad
\operatorname{Im} D_1=\operatorname{Im} D_2+2\pi n_1 .
\end{equation}
Equivalently, the vector \(D\) admits the parameterization
\begin{equation}\label{eq:D-vector}
\binom{D_1}{D_2}
=\frac{m_1}{2}\,B^{(1)}\binom{1}{1}
+\mathrm{i}\binom{1}{1}\alpha
+\binom{2\pi\mathrm{i}n_1}{0},
\qquad
\alpha\in\mathbb{R}.
\end{equation}

By \eqref{eq:D-reality}, we have
\begin{equation}\label{eq:D-conj-assumption}
-\sigma_1 D^*=D-2\pi\mathrm{i}\,n-B^{(1)}m .
\end{equation}
Under this condition, the complex conjugate of \(u(x,t)\) in \eqref{eq:u-conj} can be written as
\begin{equation}\label{eq:u-conj-long}
\begin{aligned}
u^*(x,t)
&=C_{u_0}^*
\frac{\Theta\!\left(D-2\pi\mathrm{i}n-B^{(1)}m-\Delta+\mathrm{i}\widehat{U}(x+vt)\right)}
{\Theta\!\left(D-2\pi\mathrm{i}n-B^{(1)}m+\mathrm{i}\widehat{U}(x+vt)\right)} \\[4pt]
&=C_{u_0}^*
\frac{
\Theta\!\left(D-\Delta+\mathrm{i}\widehat{U}(x+vt)\right)
\exp\!\left\{\langle D-\Delta+\mathrm{i}\widehat{U}(x+vt),m\rangle
+\tfrac12\bigl(B_{11}m_1+B_{22}m_2\bigr)\right\}}
{
\Theta\!\left(D+\mathrm{i}\widehat{U}(x+vt)\right)
\exp\!\left\{\langle D+\mathrm{i}\widehat{U}(x+vt),m\rangle + \tfrac12\bigl(B_{11}m_1+B_{22}m_2\bigr)\right\}
} \\[4pt]
&=C_{u_0}^*
\frac{\Theta\!\left(D-\Delta+\mathrm{i}\widehat{U}(x+vt)\right)
\exp\!\left\{-\langle\Delta,m\rangle\right\}}
{\Theta\!\left(D+\mathrm{i}\widehat{U}(x+vt)\right)} .
\end{aligned}
\end{equation}

Here \(m_2=m_1\), and therefore
\begin{equation}\label{eq:Delta-m-exp}
\begin{aligned}
\exp\!\left\{-\langle\Delta,m\rangle\right\}
&=\exp\!\left\{-\bigl(\pi\mathrm{i}+\nu^{(1)}\bigr)m_1
-\bigl(\pi\mathrm{i}-\nu^{(1)}\bigr)m_1\right\} \\
&=\exp(-2\pi\mathrm{i}m_1)=1 .
\end{aligned}
\end{equation}
It then follows from \eqref{eq:u-theta-ratio-2} and \eqref{eq:u-conj-long} that
\begin{equation}\label{eq:Cu0-modulus}
\lvert C_{u_0}\rvert^2=\frac{-4}{\omega_1^{2}} .
\end{equation}





\Week{Algebro-Geometric Solutions for the mKdV Equation (12)}{2025-12-09}

\obj{Focus on Case~1. Introduce the effective integration method.}

Recall that
\begin{equation}\label{eq:L-def}
L(\lambda;x,t)=\frac14\,V(\lambda;Q)+\alpha_1\,U(\lambda;Q),
\end{equation}
whose entries are given explicitly by
\begin{equation}\label{eq:L-entries}
\begin{aligned}
&L_{11}(\lambda)
=-\mathrm{i}\bigl(\lambda^{3}+\lambda(\alpha_1-\tfrac12 u^{2})\bigr),
\qquad
L_{22}(\lambda)=-L_{11}(\lambda),\\
&L_{12}(\lambda)
=u\!\left(\lambda^{2}+\mathrm{i}\frac{u_x}{2u}\lambda-\frac{u_{xx}}{4u}
+\alpha_1-\frac{u^{2}}{2}\right)
=u(\lambda-\mu_1)(\lambda-\mu_2),\\
&L_{21}(\lambda)
=-u\!\left(\lambda^{2}-\mathrm{i}\frac{u_x}{2u}\lambda-\frac{u_{xx}}{4u}
+\alpha_1-\frac{u^{2}}{2}\right)
=-u(\lambda-\mu_1^*)(\lambda-\mu_2^*).
\end{aligned}
\end{equation}

The effective integration method amounts to exploiting the fact that the
\(x\)-part of \eqref{eq:Lax-12} is an ordinary differential equation,
\[
u_{xxx}+6u^2u_x-4\alpha_1 u_x=0,
\]
which is encoded in the equations
\begin{equation}\label{eq:Lax-12}
L_x=[U(\lambda,Q),L],
\qquad
L_t=[V(\lambda,Q),L].
\end{equation}

We compute the determinant of \(L\). Using the structure of the matrix,
\begin{equation}\label{eq:detL-start}
\begin{aligned}
\det L
&=-L_{11}^2-L_{12}L_{21}\\
&=\lambda^2\bigl(\lambda^2+\alpha_1-\tfrac12 u^2\bigr)^2
+u^2\!\left[
\left(\lambda^2-\frac{u_{xx}}{4u}+\alpha_1-\frac{u^2}{2}\right)^2
+\frac{u_x^2}{4u^2}\lambda^2
\right].
\end{aligned}
\end{equation}
A direct expansion yields the polynomial form
\begin{equation}\label{eq:detL-poly}
\begin{aligned}
\det L
&=\lambda^6+2\alpha_1\lambda^4
+\left(
\frac14(u_x^2-2u u_{xx}-3u^4)
+\alpha_1u^2+\alpha_1^2
\right)\lambda^2
+\left(\frac{u_{xx}}{4}-\alpha_1 u+\frac{u^3}{2}\right)^2\\
&=\lambda^6+2\alpha_1\lambda^4+s_2\lambda^2+s_0 .
\end{aligned}
\end{equation}

Identifying coefficients in \eqref{eq:detL-poly} gives, at order \(O(1)\),
\begin{equation}\label{eq:O1}
u_{xx}-4\alpha_1 u+2u^3=4\sqrt{s_0},
\end{equation}
and, at order \(O(\lambda^2)\),
\begin{equation}\label{eq:O2}
u_x^2-2u u_{xx}-3u^4+4\alpha_1 u^2+4\alpha_1^2=4s_2 .
\end{equation}

Substituting \eqref{eq:O1} into \eqref{eq:O2} eliminates \(u_{xx}\) and yields
\begin{equation}\label{eq:ux2-step1}
u_x^2
-2u\bigl(4\sqrt{s_0}+4\alpha_1 u-2u^3\bigr)
-3u^4+4\alpha_1 u^2+4\alpha_1^2
=4s_2,
\end{equation}
which simplifies to
\begin{equation}\label{eq:ux2-step2}
u_x^2-8u\sqrt{s_0}-8\alpha_1 u^2+u^4+4\alpha_1 u^2+4\alpha_1^2=4s_2 .
\end{equation}

Equivalently,
\begin{equation}\label{eq:ux2-final}
\begin{aligned}
u_x^2
&=-\bigl(u^4-4\alpha_1 u^2-8u\sqrt{s_0}
+4\alpha_1^2-4s_2\bigr)\\
&=-R(u)
=-\prod_{i=1}^4 (u-u_i),
\end{aligned}
\end{equation}
from which it follows immediately that
\begin{equation}\label{eq:sum-ui}
\sum_{i=1}^4 u_i=0 .
\end{equation}

Multiplying \eqref{eq:O1} by \(u_x\) yields
\begin{equation}\label{eq:uxx-note}
u_x u_{xx}
=4\alpha_1 u u_x+2u^3u_x
=4\sqrt{s_0}\,u_x .
\end{equation}
Integrating once with respect to \(x\), we obtain
\begin{equation}\label{eq:ux2-int}
u_x^2
=-\bigl(-4\alpha_1 u^2+u^4-8\sqrt{s_0}\,u\bigr)+c_0,
\qquad
c_0=4s_2-4\alpha_1^2 ,
\end{equation}
which is precisely equivalent to \eqref{eq:ux2-final}.
From a different perspective, the conserved energy is
\begin{equation}\label{eq:energy}
E=4s_2-4\alpha_1^2
=u_x^2+\bigl(u^4-4\alpha_1 u^2-8\sqrt{s_0}\,u\bigr).
\end{equation}



Since \(u_x^2=-R(u)\), we may write equivalently
\begin{equation}\label{eq:du-dx}
\frac{du}{dx}=\pm\sqrt{-R(u)}.
\end{equation}
Separating variables gives
\begin{equation}\label{eq:du-int}
\int\frac{du}{\sqrt{-R(u)}}=\pm x+C .
\end{equation}


Note that \(u\) and \(u_x\) are real-valued.
\paragraph{Case I.}
\(s_0\neq 0\), \(u_i\in\mathbb{R}\), \(u_1>u_2>u_3>u_4\).
In this case there are two types of real solutions, corresponding to the admissible intervals
\begin{equation}\label{eq:case1}
u_2\le u^{[1]}\le u_1,
\qquad
u_4\le u^{[2]}\le u_3 .
\end{equation}

\paragraph{Case II.}
\(s_0\neq 0\), \(u_{1,3}\in\mathbb{R}\), \(u_1>u_3\), and \(u_{2,4}\in\mathbb{C}\setminus\mathbb{R}\).
Then the real solution satisfies
\begin{equation}\label{eq:case2}
u_3\le u^{[3]}\le u_1 .
\end{equation}

\paragraph{Case III.}
\(s_0=0\), \(u_4=-u_1\), \(u_3=-u_2\) (genus one).


We now describe the correspondence between the branch points \(\lambda_i\) and the roots \(u_i\).
If \(R(u_i)=0\), then \(\mu_1^*=\mu_1\) and \(\mu_2^*=\mu_2\), and the determinant of \(L\) takes the form
\begin{equation}\label{eq:detL-ui}
\begin{aligned}
\det L
&=\lambda^2\left(\lambda^2+\alpha_1-\frac{u_i^2}{2}\right)^2
+u_i^2(\lambda-\mu_1)^2(\lambda-\mu_2)^2 \\
&=\Bigl(\lambda^3+\lambda\!\left(\alpha_1-\frac{u_i^2}{2}\right)
+\mathrm{i}u_i(\lambda-\mu_1)(\lambda-\mu_2)\Bigr) \\
&\qquad\times
\Bigl(\lambda^3+\lambda\!\left(\alpha_1-\frac{u_i^2}{2}\right)
-\mathrm{i}u_i(\lambda-\mu_1)(\lambda-\mu_2)\Bigr).
\end{aligned}
\end{equation}
We wish to decompose
\(\det L=\prod_{j=1}^3(\lambda^2-\lambda_j^2)\)
into two cubic polynomials of the above form. This requires that the two factors have opposite coefficients in the quadratic and constant terms.
In the Case~1, all \(\lambda_j\) lie on the imaginary axis, \(\lambda_j\in\mathrm{i}\mathbb{R}\).
In the Case~2, one has
\(\lambda_1\in\mathbb{C}\setminus(\mathbb{R}\cup\mathrm{i}\mathbb{R})\),
\(\lambda_2\in\mathrm{i}\mathbb{R}\), and \(\lambda_3=-\lambda_1^*\).

For distinct \(\lambda_i,\lambda_j,\lambda_k\), the identities
\begin{equation}\label{eq:poly-id1}
\begin{aligned}
(\lambda^2-\lambda_i^2)(\lambda-\lambda_j)
&=\lambda^3-\lambda_j\lambda^2-\lambda_i^2\lambda+\lambda_i^2\lambda_j,\\
(\lambda^2-\lambda_k^2)(\lambda+\lambda_j)
&=\lambda^3+\lambda_j\lambda^2-\lambda_k^2\lambda-\lambda_k^2\lambda_j
\end{aligned}
\end{equation}
show that the constant terms cannot be matched in this manner.

Instead, consider \(S_{i},S_{j},S_{k}\in\{\pm1\}\) and distinct \(\lambda_i,\lambda_j,\lambda_k\).
Then
\begin{equation}\label{eq:sign-prod1}
(\lambda-S_i\lambda_i)(\lambda-S_j\lambda_j)(\lambda-S_k\lambda_k)
=\lambda^3-(S_i\lambda_i+S_j\lambda_j+S_k\lambda_k)\lambda^2+\cdots
-S_iS_jS_k\lambda_i\lambda_j\lambda_k,
\end{equation}
and
\begin{equation}\label{eq:sign-prod2}
(\lambda+S_i\lambda_i)(\lambda+S_j\lambda_j)(\lambda+S_k\lambda_k)
=\lambda^3+(S_i\lambda_i+S_j\lambda_j+S_k\lambda_k)\lambda^2+\cdots
+S_iS_jS_k\lambda_i\lambda_j\lambda_k .
\end{equation}
Comparing coefficients with \eqref{eq:detL-ui}, we obtain the relations
\begin{equation}\label{eq:ui-lambdai}
\begin{array}{ll}
\mathrm{i}u_1=\lambda_1+\lambda_2+\lambda_3,
& \mathrm{i}u_2=-(\lambda_1+\lambda_2-\lambda_3),\\[4pt]
\mathrm{i}u_3=-(\lambda_1-\lambda_2+\lambda_3),
& \mathrm{i}u_4=-(-\lambda_1+\lambda_2+\lambda_3).
\end{array}
\end{equation}

\extra{}


Set \(\sqrt{s_0}=\mathrm{i}\lambda_1\lambda_2\lambda_3\). With this choice, the quartic equation admits four roots given explicitly by
\[
u_1=-\mathrm{i}\bigl(\lambda_1+\lambda_2+\lambda_3\bigr),\qquad
u_2=\mathrm{i}\bigl(\lambda_1+\lambda_2-\lambda_3\bigr),\qquad
u_3=\mathrm{i}\bigl(\lambda_1-\lambda_2+\lambda_3\bigr),\qquad
u_4=\mathrm{i}\bigl(-\lambda_1+\lambda_2+\lambda_3\bigr).
\]
Pairing the factors gives
\[
\begin{aligned}
(u-u_1)(u-u_2)
&=\bigl(u+\mathrm{i}(\lambda_1+\lambda_2+\lambda_3)\bigr)
   \bigl(u-\mathrm{i}(\lambda_1+\lambda_2-\lambda_3)\bigr)
   =u^2+2\mathrm{i}\lambda_3\,u+(\lambda_1+\lambda_2)^2-\lambda_3^2 .
\end{aligned}
\]
\[
\begin{aligned}
(u-u_3)(u-u_4)
&=\bigl(u-\mathrm{i}(\lambda_1-\lambda_2+\lambda_3)\bigr)
   \bigl(u-\mathrm{i}(-\lambda_1+\lambda_2+\lambda_3)\bigr)
   =u^2-2\mathrm{i}\lambda_3\,u+(\lambda_1-\lambda_2)^2-\lambda_3^2 .
\end{aligned}
\]
Hence
\[
\begin{aligned}
\prod_{i=1}^4 (u-u_i)
&=\bigl(u^2+2\mathrm{i}\lambda_3 u+(\lambda_1+\lambda_2)^2-\lambda_3^2\bigr)
  \bigl(u^2-2\mathrm{i}\lambda_3 u+(\lambda_1-\lambda_2)^2-\lambda_3^2\bigr)\\
&=u^4+2(\lambda_1^2+\lambda_2^2+\lambda_3^2)u^2
   -8\mathrm{i}\lambda_1\lambda_2\lambda_3\,u
   +\bigl((\lambda_1^2+\lambda_2^2+\lambda_3^2)^2
          -4(\lambda_1^2\lambda_2^2+\lambda_1^2\lambda_3^2+\lambda_2^2\lambda_3^2)\bigr).
\end{aligned}
\]

Using \(\lambda_1^2+\lambda_2^2+\lambda_3^2=-2\alpha_1\),
\(\lambda_1^2\lambda_2^2+\lambda_1^2\lambda_3^2+\lambda_2^2\lambda_3^2=s_2\),
and \(\sqrt{s_0}=\mathrm{i}\lambda_1\lambda_2\lambda_3\), we obtain
\[
\prod_{i=1}^4 (u-u_i)
= u^4-4\alpha_1 u^2-8\sqrt{s_0}\,u+4\alpha_1^2-4s_2
=R(u),
\]
so each \(u_i\) is a root of \(R(u)=0\).
